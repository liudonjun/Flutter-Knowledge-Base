## 🔬 单元测试

### 1. 基本单元测试
```dart
// 被测试的类
class Calculator {
  int add(int a, int b) => a + b;
  int subtract(int a, int b) => a - b;
  int multiply(int a, int b) => a * b;
  double divide(int a, int b) {
    if (b == 0) throw ArgumentError('除数不能为零');
    return a / b;
  }
}

// 单元测试
import 'package:flutter_test/flutter_test.dart';

void main() {
  group('Calculator', () {
    late Calculator calculator;
    
    setUp(() {
      calculator = Calculator();
    });
    
    test('should add two numbers', () {
      expect(calculator.add(2, 3), equals(5));
      expect(calculator.add(-1, 1), equals(0));
      expect(calculator.add(0, 0), equals(0));
    });
    
    test('should subtract two numbers', () {
      expect(calculator.subtract(5, 3), equals(2));
      expect(calculator.subtract(3, 5), equals(-2));
      expect(calculator.subtract(0, 0), equals(0));
    });
    
    test('should multiply two numbers', () {
      expect(calculator.multiply(2, 3), equals(6));
      expect(calculator.multiply(-2, 3), equals(-6));
      expect(calculator.multiply(0, 5), equals(0));
    });
    
    test('should divide two numbers', () {
      expect(calculator.divide(6, 3), equals(2.0));
      expect(calculator.divide(5, 2), equals(2.5));
      expect(calculator.divide(0, 5), equals(0.0));
    });
    
    test('should throw error when dividing by zero', () {
      expect(() => calculator.divide(5, 0), throwsArgumentError);
    });
  });
}
```

### 2. 异步单元测试
```dart
// 异步函数测试
class DataService {
  final ApiClient _apiClient;
  
  DataService(this._apiClient);
  
  Future<List<User>> getUsers() async {
    final response = await _apiClient.get('/users');
    return (response.data as List)
        .map((json) => User.fromJson(json))
        .toList();
  }
  
  Future<User> getUser(int id) async {
    final response = await _apiClient.get('/users/$id');
    return User.fromJson(response.data);
  }
}

// 异步测试
void main() {
  group('DataService', () {
    late DataService dataService;
    late MockApiClient mockApiClient;
    
    setUp(() {
      mockApiClient = MockApiClient();
      dataService = DataService(mockApiClient);
    });
    
    test('should get users list', () async {
      // Arrange
      final mockUsers = [
        {'id': 1, 'name': 'John'},
        {'id': 2, 'name': 'Jane'},
      ];
      when(mockApiClient.get('/users')).thenAnswer(
        (_) async => Response(data: mockUsers, statusCode: 200),
      );
      
      // Act
      final users = await dataService.getUsers();
      
      // Assert
      expect(users, isA<List<User>>());
      expect(users.length, equals(2));
      expect(users[0].name, equals('John'));
      verify(mockApiClient.get('/users')).called(1);
    });
    
    test('should get user by id', () async {
      // Arrange
      final mockUser = {'id': 1, 'name': 'John'};
      when(mockApiClient.get('/users/1')).thenAnswer(
        (_) async => Response(data: mockUser, statusCode: 200),
      );
      
      // Act
      final user = await dataService.getUser(1);
      
      // Assert
      expect(user, isA<User>());
      expect(user.id, equals(1));
      expect(user.name, equals('John'));
      verify(mockApiClient.get('/users/1')).called(1);
    });
    
    test('should throw error when request fails', () async {
      // Arrange
      when(mockApiClient.get('/users')).thenThrow(
        DioError(requestOptions: RequestOptions(path: '/users')),
      );
      
      // Act & Assert
      expect(() => dataService.getUsers(), throwsA(isA<DioError>()));
    });
  });
}
```

```dart
class CounterWidget extends StatefulWidget {
  @override
  _CounterWidgetState createState() => _CounterWidgetState();
}

class _CounterWidgetState extends State<CounterWidget> {
  int _counter = 0;
  
  void _increment() {
    setState(() {
      _counter++;
    });
  }
  
  void _decrement() {
    setState(() {
      _counter--;
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('Counter: $_counter', key: Key('counter_text')),
        ElevatedButton(
          key: Key('increment_button'),
          onPressed: _increment,
          child: Text('增加'),
        ),
        ElevatedButton(
          key: Key('decrement_button'),
          onPressed: _decrement,
          child: Text('减少'),
        ),
      ],
    );
  }
}

// Widget 测试
void main() {
  group('CounterWidget', () {
    testWidgets('should display initial counter value', (WidgetTester tester) async {
      // Arrange
      await tester.pumpWidget(MaterialApp(home: CounterWidget()));
      
      // Act & Assert
      expect(find.text('Counter: 0'), findsOneWidget);
      expect(find.text('Counter: 1'), findsNothing);
    });
    
    testWidgets('should increment counter when button pressed', (WidgetTester tester) async {
      // Arrange
      await tester.pumpWidget(MaterialApp(home: CounterWidget()));
      
      // Act
      await tester.tap(find.byKey(Key('increment_button')));
      await tester.pump(); // 重新渲染
      
      // Assert
      expect(find.text('Counter: 1'), findsOneWidget);
      expect(find.text('Counter: 0'), findsNothing);
    });
    
    testWidgets('should decrement counter when button pressed', (WidgetTester tester) async {
      // Arrange
      await tester.pumpWidget(MaterialApp(home: CounterWidget()));
      
      // Act
      await tester.tap(find.byKey(Key('decrement_button')));
      await tester.pump();
      
      // Assert
      expect(find.text('Counter: -1'), findsOneWidget);
    });
    
    testWidgets('should increment multiple times', (WidgetTester tester) async {
      // Arrange
      await tester.pumpWidget(MaterialApp(home: CounterWidget()));
      
      // Act
      for (int i = 0; i < 5; i++) {
        await tester.tap(find.byKey(Key('increment_button')));
        await tester.pump();
      }
      
      // Assert

  });
```

```dart
// 被测试的列表
class UserList extends StatelessWidget {
  final List<User> users;
  final Function(User) onUserTap;
  
  UserList({required this.users, required this.onUserTap});
  
  @override
  Widget build(BuildContext context) {
    return ListView.builder(
      itemCount: users.length,
      itemBuilder: (context, index) {
        final user = users[index];
        return ListTile(
          key: Key('user_${user.id}'),
          title: Text(user.name),
          subtitle: Text(user.email),
          onTap: () => onUserTap(user),
        );
      },
    );
  }
}

// 列表测试
void main() {
  group('UserList', () {
    testWidgets('should display list of users', (WidgetTester tester) async {
      // Arrange
      final users = [
        User(id: 1, name: 'John', email: 'john@example.com'),
        User(id: 2, name: 'Jane', email: 'jane@example.com'),
        User(id: 3, name: 'Bob', email: 'bob@example.com'),
      ];
      
      await tester.pumpWidget(
        MaterialApp(
          home: Scaffold(
            body: UserList(users: users, onUserTap: (_) {}),
          ),
        ),
      );
      
      // Act & Assert
      expect(find.text('John'), findsOneWidget);
      expect(find.text('Jane'), findsOneWidget);
      expect(find.text('Bob'), findsOneWidget);
      expect(find.byType(ListTile), findsNWidgets(3));
    });
    
    testWidgets('should call onUserTap when user is tapped', (WidgetTester tester) async {
      // Arrange
      final users = [
        User(id: 1, name: 'John', email: 'john@example.com'),
        User(id: 2, name: 'Jane', email: 'jane@example.com'),
      ];
      
      User? tappedUser;
      
      await tester.pumpWidget(
        MaterialApp(
          home: Scaffold(
            body: UserList(
              users: users,
              onUserTap: (user) => tappedUser = user,
            ),
          ),
        ),
      );
      
      // Act
      await tester.tap(find.byKey(Key('user_1')));
      await tester.pump();
      
      // Assert
      expect(tappedUser, isNotNull);
      expect(tappedUser!.id, equals(1));
      expect(tappedUser!.name, equals('John'));
    });
    
    testWidgets('should display empty state when list is empty', (WidgetTester tester) async {
      // Arrange
      await tester.pumpWidget(
        MaterialApp(
          home: Scaffold(
            body: UserList(users: [], onUserTap: (_) {}),
          ),
        ),
      );
      
      // Act & Assert
      expect(find.byType(ListTile), findsNothing);
    });
  });
```

## 🔧 常见问题解决

### 1. 测试超时
```dart
// 问题：测试超时
// 解决：增加超时时间或使用 pumpAndSettle

testWidgets('should handle long animation', (tester) async {
```
