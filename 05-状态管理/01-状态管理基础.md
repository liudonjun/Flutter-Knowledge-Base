## 📦 内置状态管理

### 1. StatefulWidget
```dart
// 基本状态管理
class Counter extends StatefulWidget {
  @override
  _CounterState createState() => _CounterState();
}

class _CounterState extends State<Counter> {
  int _count = 0;
  
  void _increment() {
    setState(() {
      _count++;
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('计数: $_count'),
        ElevatedButton(
          onPressed: _increment,
          child: Text('增加'),
        ),
      ],
    );
  }
}

// 状态提升
class ParentWidget extends StatefulWidget {
  @override
  _ParentWidgetState createState() => _ParentWidgetState();
}

class _ParentWidgetState extends State<ParentWidget> {
  int _count = 0;
  
  void _updateCount(int newCount) {
    setState(() {
      _count = newCount;
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        ChildWidget(
          count: _count,
          onUpdate: _updateCount,
        ),
        Text('父组件计数: $_count'),
      ],
    );
  }
}

class ChildWidget extends StatelessWidget {
  final int count;
  final Function(int) onUpdate;
  
  ChildWidget({required this.count, required this.onUpdate});
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Text('子组件计数: $count'),
        ElevatedButton(
          onPressed: () => onUpdate(count + 1),
          child: Text('增加'),
        ),
      ],
    );
  }
}
```

### 2. InheritedWidget
```dart
// 数据共享
class CounterInheritedWidget extends InheritedWidget {
  final int count;
  final VoidCallback increment;
  
  const CounterInheritedWidget({
    Key? key,
    required this.count,
    required this.increment,
    required Widget child,
  }) : super(key: key, child: child);
  
  static CounterInheritedWidget? of(BuildContext context) {
    return context.dependOnInheritedWidgetOfExactType<CounterInheritedWidget>();
  }
  
  @override
  bool updateShouldNotify(CounterInheritedWidget oldWidget) {
    return count != oldWidget.count;
  }
}

// 使用示例
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    final inherited = CounterInheritedWidget.of(context);
    return Column(
      children: [
        Text('计数: ${inherited?.count}'),
        ElevatedButton(
          onPressed: inherited?.increment,
          child: Text('增加'),
        ),
      ],
    );
  }
}

// 父组件
class CounterApp extends StatefulWidget {
  @override
  _CounterAppState createState() => _CounterAppState();
}

class _CounterAppState extends State<CounterApp> {
  int _count = 0;
  
  void _increment() {
    setState(() {
      _count++;
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return CounterInheritedWidget(
      count: _count,
      increment: _increment,
      child: CounterWidget(),
    );
  }
}
```

```dart
import 'package:provider/provider.dart';

// 定义 ChangeNotifier
class CounterProvider with ChangeNotifier {
  int _count = 0;
  
  int get count => _count;
  
  void increment() {
    _count++;
    notifyListeners();
  }
  
  void decrement() {
    _count--;
    notifyListeners();
  }
  
  void reset() {
    _count = 0;
    notifyListeners();
  }
}

// 在 main.dart 中配置
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (context) => CounterProvider(),
      child: MyApp(),

  );

// 在 Widget 中使用
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {

              onPressed: counter.increment,
              child: Text('增加'),
            ),
            ElevatedButton(
              onPressed: counter.decrement,
              child: Text('减少'),
            ),
            ElevatedButton(
              onPressed: counter.reset,
              child: Text('重置'),
```

### 3. BLoC
```dart
// 安装依赖
// pubspec.yaml
// dependencies:
//   flutter_bloc: ^8.0.0
//   equatable: ^2.0.0

import 'package:flutter_bloc/flutter_bloc.dart';
import 'package:equatable/equatable.dart';

// 定义 Event
abstract class CounterEvent extends Equatable {
  const CounterEvent();
  
  @override
  List<Object> get props => [];
}

class IncrementEvent extends CounterEvent {}
class DecrementEvent extends CounterEvent {}
class ResetEvent extends CounterEvent {}

// 定义 State
class CounterState extends Equatable {
  final int count;
  
  const CounterState({this.count = 0});
  
  CounterState copyWith({int? count}) {
    return CounterState(count: count ?? this.count);
  }
  
  @override
  List<Object> get props => [count];
}

// 定义 BLoC
class CounterBloc extends Bloc<CounterEvent, CounterState> {
  CounterBloc() : super(const CounterState()) {
    on<IncrementEvent>((event, emit) {
      emit(state.copyWith(count: state.count + 1));
    });
    
    on<DecrementEvent>((event, emit) {
      emit(state.copyWith(count: state.count - 1));
    });
    
    on<ResetEvent>((event, emit) {
      emit(const CounterState());
    });
  }
}

// 在 main.dart 中配置
void main() {
  runApp(
    BlocProvider(
      create: (context) => CounterBloc(),
      child: MyApp(),
    ),
  );
}

// 在 Widget 中使用
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return BlocBuilder<CounterBloc, CounterState>(
      builder: (context, state) {
        return Column(
          children: [
            Text('计数: ${state.count}'),
            ElevatedButton(
              onPressed: () => context.read<CounterBloc>().add(IncrementEvent()),
              child: Text('增加'),
            ),
            ElevatedButton(
              onPressed: () => context.read<CounterBloc>().add(DecrementEvent()),
              child: Text('减少'),
            ),
            ElevatedButton(
              onPressed: () => context.read<CounterBloc>().add(ResetEvent()),
              child: Text('重置'),
            ),
          ],
        );
      },
    );
  }
}

// 使用 BlocListener
class CounterListener extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return BlocListener<CounterBloc, CounterState>(
      listener: (context, state) {
        if (state.count == 10) {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(content: Text('计数达到 10!')),
          );
        }
      },
      child: CounterWidget(),
    );
  }
}

// 使用 BlocConsumer
class CounterConsumer extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return BlocConsumer<CounterBloc, CounterState>(
      listener: (context, state) {
        if (state.count == 10) {
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(content: Text('计数达到 10!')),
          );
        }
      },
      builder: (context, state) {
        return Column(
          children: [
            Text('计数: ${state.count}'),
            ElevatedButton(
              onPressed: () => context.read<CounterBloc>().add(IncrementEvent()),
              child: Text('增加'),
            ),
          ],
        );
      },
    );
  }
}
```

### 3. Repository 模式
```dart
// Repository
class CounterRepository {
  int _count = 0;
  
  int get count => _count;
  
  void increment() => _count++;
  void decrement() => _count--;
  void reset() => _count = 0;
  
  Future<int> getCount() async {
    // 模拟网络请求
    await Future.delayed(Duration(seconds: 1));
    return _count;
  }
  
  Future<void> saveCount(int count) async {
    // 模拟保存到服务器
    await Future.delayed(Duration(seconds: 1));
    _count = count;
  }
}

// Provider
class CounterProvider extends ChangeNotifier {
  final CounterRepository _repository = CounterRepository();
  
  int _count = 0;
  bool _isLoading = false;
  
  int get count => _count;
  bool get isLoading => _isLoading;
  
  Future<void> loadCount() async {
    _isLoading = true;
    notifyListeners();
    
    try {
      _count = await _repository.getCount();
    } catch (e) {
      print('加载失败: $e');
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }
  
  Future<void> increment() async {
    _count++;
    notifyListeners();
    
    try {
      await _repository.saveCount(_count);
    } catch (e) {
      print('保存失败: $e');
      _count--;
      notifyListeners();
    }
  }
}

// View
class CounterView extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return ChangeNotifierProvider(
      create: (context) => CounterProvider()..loadCount(),
      child: Consumer<CounterProvider>(
        builder: (context, provider, child) {
          if (provider.isLoading) {
            return CircularProgressIndicator();
          }
          
          return Column(
            children: [
              Text('计数: ${provider.count}'),
              ElevatedButton(
                onPressed: provider.increment,
                child: Text('增加'),
              ),
            ],
          );
        },
      ),
    );
  }
}
```

### 2. 状态更新策略
```dart
// 1. 批量更新
class BatchUpdate {
  static void updateMultiple(VoidCallback updates) {
    // 在同一帧内更新多个状态
    updates();
  }
}

// 2. 防抖更新
class DebounceUpdate {
  static Timer? _timer;
  
  static void debounce(VoidCallback update, {Duration duration = const Duration(milliseconds: 300)}) {
    _timer?.cancel();
    _timer = Timer(duration, update);
  }
}

// 3. 节流更新
class ThrottleUpdate {
  static DateTime? _lastUpdate;
  
  static void throttle(VoidCallback update, {Duration duration = const Duration(milliseconds: 100)}) {
    final now = DateTime.now();
    if (_lastUpdate == null || now.difference(_lastUpdate!) > duration) {
      _lastUpdate = now;
      update();
    }
  }
}
```

### 3. 状态持久化
```dart
// 状态持久化
class PersistentState extends ChangeNotifier {
  static const String _keyCount = 'count';
  static const String _keyTheme = 'theme';
  
  int _count = 0;
  String _theme = 'light';
  
  int get count => _count;
  String get theme => _theme;
  
  PersistentState() {
    _loadState();
  }
  
  Future<void> _loadState() async {
    final prefs = await SharedPreferences.getInstance();
    _count = prefs.getInt(_keyCount) ?? 0;
    _theme = prefs.getString(_keyTheme) ?? 'light';
    notifyListeners();
  }
  
  Future<void> _saveState() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setInt(_keyCount, _count);
    await prefs.setString(_keyTheme, _theme);
  }
  
  void increment() {
    _count++;
    notifyListeners();
    _saveState();
  }
  
  void setTheme(String theme) {
    _theme = theme;
    notifyListeners();
    _saveState();
  }
}
```

```dart
// 状态缓存
class CachedState {
  static final Map<String, dynamic> _cache = {};
  
  static T get<T>(String key, T Function() factory) {
    if (!_cache.containsKey(key)) {
      _cache[key] = factory();
    }
    return _cache[key] as T;
  }
  
  static void set<T>(String key, T value) {
    _cache[key] = value;
  }
  
  static void remove(String key) {
    _cache.remove(key);
  }
  
  static void clear() {
    _cache.clear();
  }
}
```
