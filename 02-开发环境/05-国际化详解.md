# Flutter 国际化详解

> 掌握 Flutter 国际化，开发多语言应用。

## 🌍 国际化概述

### 1. 什么是国际化
```dart
// 国际化 (i18n - Internationalization)
// 将应用适配到不同语言和地区的过程

// 本地化 (l10n - Localization)
// 将应用翻译成特定语言的过程

// Flutter 国际化支持
// 1. flutter_localizations - 官方本地化包
// 2. intl - 国际化工具包
// 3. ARB 文件格式 - 应用资源包格式
```

### 2. 国际化流程
```dart
// 国际化流程
// 1. 配置国际化支持
// 2. 创建翻译文件
// 3. 生成本地化代码
// 4. 使用翻译文本
// 5. 处理语言切换
```

## 🔧 环境配置

### 1. 添加依赖
```yaml
# pubspec.yaml
dependencies:
  flutter:
    sdk: flutter
  flutter_localizations:
    sdk: flutter
  intl: ^0.18.0

# 开发依赖
dev_dependencies:
  flutter_test:
    sdk: flutter
  flutter_lints: ^2.0.0
```

```yaml
# l10n.yaml
arb-dir: lib/l10n
template-arb-file: app_en.arb
output-localization-file: app_localizations.dart
output-class: AppLocalizations
```

### 3. 配置 pubspec.yaml
```yaml
# pubspec.yaml
flutter:
  generate: true
  
  # 国际化配置
  # 使用 flutter gen-l10n 命令生成本地化代码
```

## 🎯 使用本地化

### 1. 配置 MaterialApp
```dart
// lib/main.dart
import 'package:flutter/material.dart';
import 'package:flutter_localizations/flutter_localizations.dart';
import 'l10n/app_localizations.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      
      // 国际化配置
      localizationsDelegates: [
        AppLocalizations.delegate,
        GlobalMaterialLocalizations.delegate,
        GlobalWidgetsLocalizations.delegate,
        GlobalCupertinoLocalizations.delegate,
      ],
      
      // 支持的语言
      supportedLocales: [
        const Locale('en', ''), // 英语
        const Locale('zh', ''), // 中文
        const Locale('ja', ''), // 日语
      ],
      
      // 默认语言
      localeResolutionCallback: (locale, supportedLocales) {
        for (var supportedLocale in supportedLocales) {
          if (supportedLocale.languageCode == locale?.languageCode) {
            return supportedLocale;
          }
        }
        return supportedLocales.first;
      },
      
      home: MyHomePage(),
    );
  }
}
```

### 3. 动态切换语言
```dart
// 语言切换服务
class LanguageService extends ChangeNotifier {
  Locale _locale = Locale('en', '');
  
  Locale get locale => _locale;
  
  void changeLanguage(String languageCode) {
    _locale = Locale(languageCode, '');
    notifyListeners();
  }
}

// 在应用中使用
class MyApp extends StatefulWidget {
  @override
  _MyAppState createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  final LanguageService _languageService = LanguageService();
  
  @override
  Widget build(BuildContext context) {
    return ChangeNotifierProvider.value(
      value: _languageService,
      child: Consumer<LanguageService>(
        builder: (context, languageService, child) {
          return MaterialApp(
            locale: languageService.locale,
            localizationsDelegates: [
              AppLocalizations.delegate,
              GlobalMaterialLocalizations.delegate,
              GlobalWidgetsLocalizations.delegate,
              GlobalCupertinoLocalizations.delegate,
            ],
            supportedLocales: [
              const Locale('en', ''),
              const Locale('zh', ''),
              const Locale('ja', ''),
            ],
            home: MyHomePage(),
          );
        },
      ),
    );
  }
}

// 语言切换页面
class LanguageSwitchPage extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    final languageService = Provider.of<LanguageService>(context);
    
    return Scaffold(
      appBar: AppBar(
        title: Text('切换语言'),
      ),
      body: ListView(
        children: [
          ListTile(
            title: Text('English'),
            trailing: languageService.locale.languageCode == 'en'
                ? Icon(Icons.check)
                : null,
            onTap: () {
              languageService.changeLanguage('en');
              Navigator.pop(context);
            },
          ),
          ListTile(
            title: Text('中文'),
            trailing: languageService.locale.languageCode == 'zh'
                ? Icon(Icons.check)
                : null,
            onTap: () {
              languageService.changeLanguage('zh');
              Navigator.pop(context);
            },
          ),
          ListTile(
            title: Text('日本語'),
            trailing: languageService.locale.languageCode == 'ja'
                ? Icon(Icons.check)
                : null,
            onTap: () {
              languageService.changeLanguage('ja');
              Navigator.pop(context);
            },
          ),
        ],
      ),
    );
  }
}
```

```dart
// 本地化表单
class LocalizedForm extends StatelessWidget {
  final _formKey = GlobalKey<FormState>();
  final _nameController = TextEditingController();
  final _emailController = TextEditingController();
  
  @override
  Widget build(BuildContext context) {
    final l10n = AppLocalizations.of(context)!;
    
    return Form(
      key: _formKey,
      child: Column(
        children: [
          TextFormField(
            controller: _nameController,
            decoration: InputDecoration(
              labelText: l10n.name,
              hintText: l10n.enterName,
              errorText: l10n.nameRequired,
            ),
            validator: (value) {
              if (value == null || value.isEmpty) {
                return l10n.nameRequired;
              }
              return null;
            },
          ),
          TextFormField(
            controller: _emailController,
            decoration: InputDecoration(
              labelText: l10n.email,
              hintText: l10n.enterEmail,
              errorText: l10n.emailRequired,
            ),
            validator: (value) {
              if (value == null || value.isEmpty) {
                return l10n.emailRequired;
              }
              if (!value.contains('@')) {
                return l10n.invalidEmail;
              }
              return null;
            },
          ),
          ElevatedButton(
            onPressed: () {
              if (_formKey.currentState!.validate()) {
                // 处理表单提交
              }
            },
            child: Text(l10n.submit),
          ),
        ],
      ),
    );
  }
}
```

## 🔧 高级国际化

### 1. 动态语言包加载
```dart
// 动态加载语言包
class DynamicLocalization {
  static final Map<String, Map<String, String>> _translations = {};
  
  static Future<void> loadLanguage(String languageCode) async {
    if (_translations.containsKey(languageCode)) return;
    
    final jsonString = await rootBundle.loadString(
      'assets/l10n/app_$languageCode.arb'
    );
    final Map<String, dynamic> jsonMap = jsonDecode(jsonString);
    
    _translations[languageCode] = jsonMap.map(
      (key, value) => MapEntry(key, value.toString()),
    );
  }
  
  static String translate(String key, String languageCode) {
    return _translations[languageCode]?[key] ?? key;
  }
}

// 使用动态语言包
class DynamicText extends StatelessWidget {
  final String key;
  final Map<String, String>? placeholders;
  
  const DynamicText({
    Key? key,
    required this.key,
    this.placeholders,
  }) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    final languageService = Provider.of<LanguageService>(context);
    String text = DynamicLocalization.translate(
      key,
      languageService.locale.languageCode,
    );
    
    // 替换占位符
    placeholders?.forEach((key, value) {
      text = text.replaceAll('{$key}', value);
    });
    
    return Text(text);
  }
}
```

```dart
// 本地化资源管理器
class LocalizedAssets {
  static String getImagePath(String imageName, String languageCode) {
    return 'assets/images/$languageCode/$imageName';
  }
  
  static String getIconPath(String iconName, String languageCode) {
    return 'assets/icons/$languageCode/$iconName';
  }
  
  static String getDataPath(String dataName, String languageCode) {
    return 'assets/data/$languageCode/$dataName';
  }
}

// 使用本地化资源
class LocalizedImage extends StatelessWidget {
  final String imageName;
  
  const LocalizedImage({Key? key, required this.imageName}) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    final languageService = Provider.of<LanguageService>(context);
    final imagePath = LocalizedAssets.getImagePath(
      imageName,
      languageService.locale.languageCode,
    );
    
    return Image.asset(imagePath);
  }
}
```

### 2. 翻译文本规范
```json
// 好的翻译文本
{
  "welcome": "Welcome to {appName}",
  "itemsCount": "{count, plural, =0{No items} =1{1 item} other{{count} items}}",
  "userJoined": "{gender, select, male{He joined} female{She joined} other{They joined}}"

// 不好的翻译文本
```
