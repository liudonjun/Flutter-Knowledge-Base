### 2. Flutter 中的响应式实现

```dart
// Flutter 响应式实现
class FlutterReactiveImplementation {
  void explain() {
    print('''
    Flutter 响应式实现：
    
    // 1. Widget 重建
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      int _count = 0;
      
      void _increment() {
        setState(() {
          _count++;  // 状态改变
        });  // 触发 Widget 重建
      }
      
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            Text('Count: $_count'),  // 响应式更新
            ElevatedButton(
              onPressed: _increment,
              child: Text('Increment'),
            ),
          ],
        );
      }
    }
    
    // 2. Stream 处理
    class StreamExample extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return StreamBuilder<int>(
          stream: _getStream(),
          builder: (context, snapshot) {
            if (snapshot.hasData) {
              return Text('Data: ${snapshot.data}');
            } else if (snapshot.hasError) {
              return Text('Error: ${snapshot.error}');
            } else {
              return CircularProgressIndicator();
            }
          },
        );
      }
      
      Stream<int> _getStream() async* {
        for (int i = 0; i < 10; i++) {
          await Future.delayed(Duration(seconds: 1));
          yield i;
        }
      }
    }
    
    // 3. Future 处理
    class FutureExample extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return FutureBuilder<int>(
          future: _getFuture(),
          builder: (context, snapshot) {
            if (snapshot.hasData) {
              return Text('Data: ${snapshot.data}');
            } else if (snapshot.hasError) {
              return Text('Error: ${snapshot.error}');
            } else {
              return CircularProgressIndicator();
            }
          },
        );
      }
      
      Future<int> _getFuture() async {
        await Future.delayed(Duration(seconds: 2));
        return 42;
      }
    }
    ''');
  }
}
```

## 🔧 响应式状态管理

### 1. 状态管理方案

```dart
// 状态管理方案
class StateManagementSolutions {
  void explain() {
    print('''
    状态管理方案：
    
    // 1. 内置状态管理
    // - setState：简单的状态管理
    // - InheritedWidget：跨 Widget 共享数据
    // - ValueNotifier：轻量级状态管理
    
    // 2. 第三方状态管理
    // - Provider：依赖注入
    // - BLoC：业务逻辑组件
    // - Riverpod：改进的 Provider
    // - GetX：轻量级状态管理
    // - Redux：集中式状态管理
    
    // 3. 选择状态管理方案
    // - 简单应用：setState
    // - 中等应用：Provider
    // - 复杂应用：BLoC 或 Riverpod
    // - 团队协作：Redux 或 BLoC
    
    // 4. 状态管理示例
    class Counter with ChangeNotifier {
      int _count = 0;
      
      int get count => _count;
      
      void increment() {
        _count++;
        notifyListeners();
      }
    }
    
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return ChangeNotifierProvider(
          create: (context) => Counter(),
          child: Consumer<Counter>(
            builder: (context, counter, child) {
              return Text('Count: ${counter.count}');
            },
          ),
        );
      }
    }
    ''');
  }
}
```

### 2. Stream 处理

```dart
// Stream 处理
class StreamProcessing {
  void explain() {
    print('''
    Stream 处理：
    
    // 1. Stream 类型
    // - 单订阅 Stream：只能有一个监听器
    // - 多订阅 Stream：可以有多个监听器
    
    // 2. Stream 操作
    // - map：转换数据
    // - where：过滤数据
    // - reduce：聚合数据
    // - asyncMap：异步转换
    
    // 3. Stream 示例
    class StreamExample extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return StreamBuilder<int>(
          stream: _getStream().where((n) => n % 2 == 0),
          builder: (context, snapshot) {
            if (snapshot.hasData) {
              return Text('Even: ${snapshot.data}');
            } else {
              return CircularProgressIndicator();
            }
          },
        );
      }
      
      Stream<int> _getStream() async* {
        for (int i = 0; i < 10; i++) {
          await Future.delayed(Duration(seconds: 1));
          yield i;
        }
      }
    }
    
    // 4. Stream 控制器
    class StreamControllerExample {
      final _controller = StreamController<int>();
      
      Stream<int> get stream => _controller.stream;
      
      void addData(int data) {
        _controller.add(data);
      }
      
      void dispose() {
        _controller.close();
      }
    }
    ''');
  }
}
```
