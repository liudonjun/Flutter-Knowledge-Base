### 1. 布局系统

```dart
// 布局系统
class LayoutSystem {
  void explain() {
    print('''
    布局系统：
    
    // 1. 布局过程
    // - 约束传递：父到子
    // - 尺寸确定：子到父
    // - 位置确定：父到子
    // - 布局缓存：避免重复计算
    
    // 2. 约束系统
    // - BoxConstraints：盒模型约束
    // - SliverConstraints：滚动约束
    // - 约束传递规则
    // - 约束解决算法
    
    // 3. 布局优化
    // - 布局缓存
    // - 布局边界
    // - 布局委托
    // - 布局构建器
    
    // 4. 布局示例
    class MyRenderBox extends RenderBox {
      @override
      void performLayout() {
        // 获取约束
        final constraints = this.constraints;
        
        // 确定大小
        size = constraints.constrain(Size(100, 100));
        
        // 布局子对象
        if (child != null) {
          child!.layout(constraints, parentUsesSize: true);
        }
      }
    }
    ''');
  }
}
```

```dart
// 绘制系统
class PaintingSystem {
  void explain() {
    print('''
    绘制系统：
    
    // 1. 绘制过程
    // - 绘制上下文：PaintingContext
    // - 绘制画布：Canvas
    // - 绘制属性：Paint
    // - 绘制顺序：深度优先
    
    // 2. 绘制优化
    // - 绘制边界：RepaintBoundary
    // - 绘制缓存：Picture
    // - 绘制合成：Layer
    // - 绘制裁剪：Clip
    
    // 3. 绘制示例
    class MyRenderBox extends RenderBox {
      @override
      void paint(PaintingContext context, Offset offset) {
        // 获取画布
        final canvas = context.canvas;
        
        // 创建画笔
        final paint = Paint()
          ..color = Colors.blue
          ..style = PaintingStyle.fill;
        
        // 绘制矩形
        canvas.drawRect(offset & size, paint);
        
        // 绘制子对象
        if (child != null) {
          context.paintChild(child!, offset);
        }
      }
    }
    
    // 4. 绘制层次
    // - 背景层：background
    // - 内容层：content
    // - 前景层：foreground
    // - 装饰层：decoration
    ''');
  }
}
```
