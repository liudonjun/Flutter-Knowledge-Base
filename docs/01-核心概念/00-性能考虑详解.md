```dart
    // - 使用合适的 Widget
    // - 使用 const 构造函数
    // - 使用 RepaintBoundary

      child: Padding(
        padding: EdgeInsets.all(8),
        child: Column(
          children: [
            Container(
              child: Padding(
                padding: EdgeInsets.all(8),
                child: Text('Hello'),
```

```dart
// 绘制性能优化
class PaintingPerformance {
  void explain() {
    print('''
    绘制性能优化：
    
    // 1. 绘制性能优化
    // - 使用 RepaintBoundary
    // - 使用 CustomPaint
    // - 使用 Canvas
    // - 使用 Picture
    
    // 2. 绘制性能示例
    // 错误：未使用 RepaintBoundary
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            ComplexWidget(),  // 错误：复杂 Widget 未隔离
            SimpleWidget(),
          ],
        );
      }
    }
    
    // 正确：使用 RepaintBoundary
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            RepaintBoundary(
              child: ComplexWidget(),  // 正确：隔离复杂 Widget
            ),
            SimpleWidget(),
          ],
        );
      }
    }
    
    // 3. 绘制性能最佳实践
    // - 使用 CustomPaint
    // - 使用 Canvas
```

### 2. 内存性能

```dart
// 内存性能
class MemoryPerformance {
  void explain() {
    print('''
    内存性能：
    
    // 1. 内存性能优化
    // - 及时释放资源
    // - 使用 WeakReference
    // - 优化图片加载
    // - 使用缓存策略
    
    // 2. 内存性能示例
    // 错误：未释放资源
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      // 错误：未释放资源
    }
    
    // 正确：释放资源
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      @override
      void dispose() {
        _controller.dispose();  // 正确：释放控制器
        _subscription?.cancel();  // 正确：取消订阅
        super.dispose();
      }
    }
    
    // 3. 内存性能最佳实践
    // - 使用 const 构造函数
    // - 使用 WeakReference
    // - 使用缓存策略
    // - 使用懒加载
    
    // 4. 内存性能监控
    // - 使用 DevTools Memory
    // - 使用内存快照
    // - 使用内存分配跟踪
    // - 使用内存泄漏检测
    ''');
  }
}
```
