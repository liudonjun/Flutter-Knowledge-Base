### 2. 帧率分析

```dart
// 帧率分析
class FrameRateAnalysis {
  void explain() {
    print('''
    帧率分析：
    
    // 1. 帧率概念
    // - 帧率：每秒渲染的帧数
    // - 目标帧率：60 FPS
    // - 最低帧率：30 FPS
    // - 帧时间：每帧的渲染时间
    
    // 2. 帧率监控
    // - 使用 Performance Overlay
    // - 使用 DevTools
    // - 使用 Timeline
    // - 使用自定义监控
    
    // 3. 帧率优化
    // - 减少 Widget 重建
    // - 优化布局计算
    // - 减少绘制次数
    // - 优化图片加载
    
    // 4. 帧率示例
    class FrameRateMonitor {
      final List<int> _frameTimes = [];
      
      void startTracking() {
        SchedulerBinding.instance.addTimingsCallback((timings) {
          for (final timing in timings) {
            _frameTimes.add(timing.totalSpan.inMilliseconds);
          }
        });
      }
      
      double get averageFrameTime {
        if (_frameTimes.isEmpty) return 0;
        return _frameTimes.reduce((a, b) => a + b) / _frameTimes.length;
      }
      
      double get averageFps {
        if (averageFrameTime == 0) return 0;
        return 1000 / averageFrameTime;
      }
    }
    ''');
  }
}
```
