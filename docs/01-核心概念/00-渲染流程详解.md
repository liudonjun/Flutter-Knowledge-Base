    // - 约束传递
    // - 尺寸计算
    
    // 2. 布局算法
    // - 约束向下传递

    // - 布局缓存
    
    // 3. 布局优化

    // 4. 布局示例
    class MyRenderBox extends RenderBox {
      @override
      void performLayout() {
        // 获取约束

        // 确定大小

        // 布局子对象

### 2. 布局约束

```dart
// 布局约束
class LayoutConstraints {
  void explain() {
    print('''
    布局约束：
    
    // 1. BoxConstraints
    // - minWidth：最小宽度
    // - maxWidth：最大宽度
    // - minHeight：最小高度
    // - maxHeight：最大高度
    
    // 2. 约束类型
    // - 宽松约束：无最小限制
    // - 严格约束：有最小和最大限制
    // - 无限约束：无最大限制
    // - 固定约束：固定尺寸
    
    // 3. 约束传递规则
    // - 父级传递约束给子级
    // - 子级返回尺寸给父级
    // - 父级确定子级位置
    // - 子级必须在约束范围内
    
    // 4. 约束示例
    class MyRenderBox extends RenderBox {
      @override
      void performLayout() {
        // 获取约束
        final constraints = this.constraints;
        
        // 检查约束类型
        if (constraints.hasBoundedWidth && constraints.hasBoundedHeight) {
          // 有界约束
          size = constraints.constrain(Size(100, 100));
        } else if (constraints.hasTightWidth && constraints.hasTightHeight) {
          // 紧约束
          size = constraints.smallest;
        } else {
          // 松约束
          size = constraints.constrain(Size(100, 100));
        }
      }
    }
    ''');
  }
}
```
