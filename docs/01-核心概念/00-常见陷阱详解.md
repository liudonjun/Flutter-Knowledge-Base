# 常见陷阱

> 深入理解 Flutter 开发中的常见陷阱。

## 📖 常见陷阱概述

### 1. 布局陷阱

```dart
// 布局陷阱
class LayoutPitfalls {
  void explain() {
    print('''
    布局陷阱：
    
    // 1. 无限尺寸陷阱
    // 错误：在 ListView 中使用 Column
    ListView(
      children: [
        Column(  // 错误：Column 会尝试无限扩展
          children: [
            Text('Item 1'),
            Text('Item 2'),
          ],
        ),
      ],
    )
    
    // 正确：使用 shrinkWrap 或 Expanded
    ListView(
      shrinkWrap: true,  // 正确：限制 ListView 尺寸
      children: [
        Column(
          children: [
            Text('Item 1'),
            Text('Item 2'),
          ],
        ),
      ],
    )
    
    // 2. 约束传递陷阱
    // 错误：在 Container 中使用无限宽度
    Container(
      width: double.infinity,  // 错误：可能导致布局问题
      child: Text('Hello'),
    )
    
    // 正确：使用合适的约束
    Container(
      width: 200,  // 正确：固定宽度
      child: Text('Hello'),
    )
    
    // 3. 嵌套过深陷阱
    // 错误：Widget 嵌套过深
    Container(
      child: Padding(
        padding: EdgeInsets.all(8),
        child: Column(
          children: [
            Container(
              child: Padding(
                padding: EdgeInsets.all(8),
                child: Text('Hello'),
              ),
            ),
          ],
        ),
      ),
    )
    
    // 正确：简化嵌套
    Padding(
      padding: EdgeInsets.all(8),
      child: Column(
        children: [
          Padding(
            padding: EdgeInsets.all(8),
            child: Text('Hello'),
          ),
        ],
      ),
    )
    ''');
  }
}
```

### 2. 状态管理陷阱

```dart
// 状态管理陷阱
class StateManagementPitfalls {
  void explain() {
    print('''
    状态管理陷阱：
    
    // 1. 在 initState 中使用 context
    // 错误：在 initState 中使用 context
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      @override
      void initState() {
        super.initState();
        final theme = Theme.of(context);  // 错误：context 不可用
      }
    }
    
    // 正确：在 didChangeDependencies 中使用 context
    class _MyWidgetState extends State<MyWidget> {
      @override
      void didChangeDependencies() {
        super.didChangeDependencies();
        final theme = Theme.of(context);  // 正确：context 可用
      }
    }
    
    // 2. 内存泄漏陷阱
    // 错误：未释放资源
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      // 错误：未释放资源
    }
    
    // 正确：释放资源
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      @override
      void dispose() {
        _controller.dispose();  // 正确：释放控制器
        _subscription?.cancel();  // 正确：取消订阅
        super.dispose();
      }
    }
    
    // 3. 状态提升陷阱
    // 错误：状态位置不当
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      int _count = 0;  // 错误：状态位置不当
      
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            Text('Count: $_count'),
            ElevatedButton(
              onPressed: () => setState(() => _count++),
              child: Text('Increment'),
            ),
          ],
        );
      }
    }
    
    // 正确：提升状态
    class ParentWidget extends StatefulWidget {
      @override
      _ParentWidgetState createState() => _ParentWidgetState();
    }
    
    class _ParentWidgetState extends State<ParentWidget> {
      int _count = 0;  // 正确：提升状态
      
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            ChildWidget(count: _count),
            ElevatedButton(
              onPressed: () => setState(() => _count++),
              child: Text('Increment'),
            ),
          ],
        );
      }
    }
    ''');
  }
}
```

## 🔧 性能陷阱

### 1. 重建陷阱

```dart
// 重建陷阱
class RebuildPitfalls {
  void explain() {
    print('''
    重建陷阱：
    
    // 1. 不必要的重建
    // 错误：在 build 中创建对象
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        final style = TextStyle(fontSize: 16);  // 错误：每次重建都创建
        
        return Text('Hello', style: style);
      }
    }
    
    // 正确：使用 const 或提取对象
    class MyWidget extends StatelessWidget {
      static const style = TextStyle(fontSize: 16);  // 正确：静态常量
      
      @override
      Widget build(BuildContext context) {
        return Text('Hello', style: style);
      }
    }
    
    // 2. 未使用 const
    // 错误：未使用 const 构造函数
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            Text('Hello'),  // 错误：未使用 const
            Icon(Icons.star),  // 错误：未使用 const
          ],
        );
      }
    }
    
    // 正确：使用 const 构造函数
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: const [
            Text('Hello'),  // 正确：使用 const
            Icon(Icons.star),  // 正确：使用 const
          ],
        );
      }
    }
    
    // 3. 未使用 RepaintBoundary
    // 错误：复杂 Widget 未隔离
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            ComplexWidget(),  // 错误：复杂 Widget 未隔离
            SimpleWidget(),
          ],
        );
      }
    }
    
    // 正确：使用 RepaintBoundary 隔离
    class MyWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            RepaintBoundary(
              child: ComplexWidget(),  // 正确：隔离复杂 Widget
            ),
            SimpleWidget(),
          ],
        );
      }
    }
    ''');
  }
}
```

### 2. 内存陷阱

```dart
// 内存陷阱
class MemoryPitfalls {
  void explain() {
    print('''
    内存陷阱：
    
    // 1. 内存泄漏
    // 错误：未释放资源
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      // 错误：未释放资源
    }
    
    // 正确：释放资源
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      @override
      void dispose() {
        _controller.dispose();  // 正确：释放控制器
        _subscription?.cancel();  // 正确：取消订阅
        super.dispose();
      }
    }
    
    // 2. 图片内存
    // 错误：未优化图片加载
    Image.network(
      'https://example.com/image.jpg',  // 错误：未优化
    )
    
    // 正确：优化图片加载
    Image.network(
      'https://example.com/image.jpg',
      width: 100,
      height: 100,
      cacheWidth: 100,  // 正确：缓存尺寸
      cacheHeight: 100,  // 正确：缓存尺寸
    )
    
    // 3. 列表内存
    // 错误：未使用懒加载
    ListView(
      children: items.map((item) => ItemWidget(item)).toList(),  // 错误：一次性加载
    )
    
    // 正确：使用懒加载
    ListView.builder(
      itemCount: items.length,
      itemBuilder: (context, index) {
        return ItemWidget(items[index]);  // 正确：懒加载
      },
    )
    ''');
  }
}
```
