# 路由动画详解

> 掌握 Flutter 中的各种页面转场动画。

## 📖 内置转场动画

### 1. 默认动画

```dart
// 内置转场动画
class BuiltInAnimations {
  void explain() {
    print('''
    内置转场动画：
    
    // 1. MaterialPageRoute - 滑动动画（Android 风格）
    Navigator.push(
      context,
      MaterialPageRoute(builder: (context) => NextPage()),
    );
    
    // 2. CupertinoPageRoute - 滑动动画（iOS 风格）
    Navigator.push(
      context,
      CupertinoPageRoute(builder: (context) => NextPage()),
    );
    
    // 3. 全屏对话框
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => DialogPage(),
        fullscreenDialog: true,
      ),
    );
    ''');
  }
}
```

### 2. 自定义转场

```dart
// 自定义转场动画
class CustomTransitions {
  void explain() {
    print('''
    自定义转场动画：
    
    // 1. 淡入淡出
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        return FadeTransition(opacity: animation, child: child);
      },
    )
    
    // 2. 滑动
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        var begin = Offset(1.0, 0.0);
        var end = Offset.zero;
        var tween = Tween(begin: begin, end: end);
        var offsetAnimation = animation.drive(tween);
        return SlideTransition(position: offsetAnimation, child: child);
      },
    )
    
    // 3. 缩放
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        return ScaleTransition(scale: animation, child: child);
      },
    )
    
    // 4. 旋转
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        return RotationTransition(turns: animation, child: child);
      },
    )
    ''');
  }
}
```

### 3. 组合动画

```dart
// 组合动画
class CombinedAnimations {
  void explain() {
    print('''
    组合动画：
    
    // 淡入 + 滑动
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        var fadeAnimation = Tween(begin: 0.0, end: 1.0).animate(animation);
        var slideAnimation = Tween(begin: Offset(0.0, 0.3), end: Offset.zero).animate(animation);
        
        return FadeTransition(
          opacity: fadeAnimation,
          child: SlideTransition(
            position: slideAnimation,
            child: child,
          ),
        );
      },
    )
    
    // 自定义曲线
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        var curvedAnimation = CurvedAnimation(
          parent: animation,
          curve: Curves.easeInOutBack,
        );
        return ScaleTransition(scale: curvedAnimation, child: child);
      },
    )
    
    // 不同方向的滑动
    PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => NextPage(),
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        // 从底部滑入
        var begin = Offset(0.0, 1.0);
        var end = Offset.zero;
        var tween = Tween(begin: begin, end: end);
        var offsetAnimation = animation.drive(tween.chain(
          ChainTween(curve: Curves.ease),
        ));
        return SlideTransition(position: offsetAnimation, child: child);
      },
    )
    ''');
  }
}
```

## 📖 封装工具

### 1. 路由工具类

```dart
// 路由工具类
class RouteUtils {
  // 淡入淡出
  static Route<T> fadeTransition<T>(Widget page) {
    return PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => page,
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        return FadeTransition(opacity: animation, child: child);
      },
    );
  }
  
  // 从底部滑入
  static Route<T> slideFromBottom<T>(Widget page) {
    return PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => page,
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        var begin = Offset(0.0, 1.0);
        var end = Offset.zero;
        var tween = Tween(begin: begin, end: end);
        var offsetAnimation = animation.drive(tween);
        return SlideTransition(position: offsetAnimation, child: child);
      },
    );
  }
  
  // 缩放
  static Route<T> scaleTransition<T>(Widget page) {
    return PageRouteBuilder(
      pageBuilder: (context, animation, secondaryAnimation) => page,
      transitionsBuilder: (context, animation, secondaryAnimation, child) {
        return ScaleTransition(scale: animation, child: child);
      },
    );
  }
  
  // 使用
  static void navigateWithFade(BuildContext context, Widget page) {
    Navigator.push(context, fadeTransition(page));
  }
}
''');
  }
}
```
