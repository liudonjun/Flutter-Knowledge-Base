### 2. 路由状态管理实现

```dart
// 路由状态管理实现
class RouteStateManagementImplementation {
  void explain() {
    print('''
    路由状态管理实现：
    
    // 1. 使用 Navigator 2.0
    class RouterDelegate extends ChangeNotifier implements RouterDelegate<RouteInformation> {
      List<Page> _pages = [];
      
      @override
      Widget build(BuildContext context) {
        return Navigator(
          pages: _pages,
          onPopPage: (route, result) {
            if (!route.didPop(result)) {
              return false;
            }
            _pages.removeLast();
            notifyListeners();
            return true;
          },
        );
      }
      
      @override
      Future<void> setNewRoutePath(RouteInformation configuration) async {
        // 解析路由信息
        final uri = Uri.parse(configuration.uri.toString());
        _pages = _parsePages(uri);
        notifyListeners();
      }
      
      @override
      RouteInformation? get currentConfiguration {
        // 返回当前路由信息
        return RouteInformation(
          uri: Uri.parse(_pages.last.name ?? '/'),
        );
      }
    }
    
    // 2. 使用 GoRouter
    final router = GoRouter(
      routes: [
        GoRoute(
          path: '/',
          builder: (context, state) => HomePage(),
        ),
        GoRoute(
          path: '/details/:id',
          builder: (context, state) {
            final id = state.params['id'];
            return DetailsPage(id: id!);
          },
        ),
      ],
    );
    ''');
  }
}
```

## 🔧 路由状态持久化

### 1. 状态保存

```dart
// 状态保存
class StateSaving {
  void explain() {
    print('''
    状态保存：
    
    // 1. 使用 SharedPreferences
    class RouteStateStorage {
      static const _key = 'route_state';
      
      static Future<void> save(List<String> routes) async {
        final prefs = await SharedPreferences.getInstance();
        await prefs.setStringList(_key, routes);
      }
      
      static Future<List<String>> load() async {
        final prefs = await SharedPreferences.getInstance();
        return prefs.getStringList(_key) ?? ['/'];
      }
    }
    
    // 2. 使用 Hive
    class RouteStateBox {
      static late Box _box;
      
      static Future<void> init() async {
        _box = await Hive.openBox('route_state');
      }
      
      static Future<void> save(List<String> routes) async {
        await _box.put('routes', routes);
      }
      
      static List<String> load() {
        return _box.get('routes', defaultValue: ['/']);
      }
    }
    
    // 3. 使用 JSON 序列化
    class RouteState {
      final List<String> routes;
      final Map<String, dynamic> params;
      
      RouteState(this.routes, this.params);
      
      Map<String, dynamic> toJson() => {
        'routes': routes,
        'params': params,
      };
      
      factory RouteState.fromJson(Map<String, dynamic> json) {
        return RouteState(
          List<String>.from(json['routes']),
          Map<String, dynamic>.from(json['params']),
        );
      }
    }
    ''');
  }
}
```
