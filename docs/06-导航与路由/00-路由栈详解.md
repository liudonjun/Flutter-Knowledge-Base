### 2. 路由栈配置

```dart
// 路由栈配置
class RouteStackConfiguration {
  void explain() {
    print('''
    路由栈配置：
    
    // 1. 基本配置
    Navigator(
      initialRoute: '/',
      onGenerateRoute: (settings) {
        return MaterialPageRoute(
          builder: (context) => getPage(settings.name),
        );
      },
    )
    
    // 2. 使用 Navigator 管理栈
    // push：添加页面到栈顶
    Navigator.push(context, MaterialPageRoute(...));
    
    // pop：移除栈顶页面
    Navigator.pop(context);
    
    // pushReplacement：替换栈顶页面
    Navigator.pushReplacement(context, MaterialPageRoute(...));
    
    // pushAndRemoveUntil：清除栈并添加新页面
    Navigator.pushAndRemoveUntil(
      context,
      MaterialPageRoute(...),
      (route) => false,
    );
    
    // 3. 使用 Navigator 2.0
    class MyRouterDelegate extends RouterDelegate<RouteInformation> {
      List<Page> _pages = [];
      
      @override
      Widget build(BuildContext context) {
        return Navigator(
          pages: _pages,
          onPopPage: (route, result) {
            if (!route.didPop(result)) {
              return false;
            }
            _pages.removeLast();
            notifyListeners();
            return true;
          },
        );
      }
    }
    ''');
  }
}
```

## 🔧 路由栈使用

### 1. 栈操作

```dart
// 栈操作
class StackOperations {
  void explain() {
    print('''
    栈操作：
    
    // 1. push：添加页面
    Navigator.push(context, MaterialPageRoute(
      builder: (context) => DetailsPage(),
    ));
    
    // 2. pop：移除页面
    Navigator.pop(context);
    
    // 3. pushReplacement：替换页面
    Navigator.pushReplacement(context, MaterialPageRoute(
      builder: (context) => NewPage(),
    ));
    
    // 4. pushAndRemoveUntil：清除栈
    Navigator.pushAndRemoveUntil(
      context,
      MaterialPageRoute(builder: (context) => HomePage()),
      (route) => false,  // 清除所有页面
    );
    
    // 5. popUntil：弹出到指定页面
    Navigator.popUntil(context, ModalRoute.withName('/home'));
    
    // 6. canPop：检查是否可以返回
    if (Navigator.canPop(context)) {
      Navigator.pop(context);
    }
    ''');
  }
}
```

### 2. 栈状态

```dart
// 栈状态
class StackState {
  void explain() {
    print('''
    栈状态：
    
    // 1. 获取栈信息
    final route = ModalRoute.of(context);
    final name = route?.settings.name;
    final arguments = route?.settings.arguments;
    
    // 2. 检查栈状态
    final canPop = Navigator.canPop(context);
    final isFirst = route?.isFirst ?? false;
    final isCurrent = route?.isCurrent ?? false;
    
    // 3. 栈状态管理
    class StackManager {
      static List<String> getRouteStack(BuildContext context) {
        final routes = <String>[];
        Navigator.of(context).popUntil((route) {
          routes.add(route.settings.name ?? '/');
          return true;
        });
        return routes;
      }
      
      static bool isRouteInStack(BuildContext context, String routeName) {
        final routes = getRouteStack(context);
        return routes.contains(routeName);
      }
    }
    ''');
  }
}
```
