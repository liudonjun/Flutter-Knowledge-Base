### 2. 路由中间件实现

```dart
// 路由中间件实现
class RouteMiddlewareImplementation {
  void explain() {
    print('''
    路由中间件实现：
    
    // 1. 中间件接口
    abstract class RouteMiddleware {
      RouteSettings? handle(RouteSettings settings);
    }
    
    // 2. 权限中间件
    class AuthMiddleware implements RouteMiddleware {
      @override
      RouteSettings? handle(RouteSettings settings) {
        if (settings.name == '/admin' && !isAdmin) {
          return RouteSettings(name: '/login');
        }
        return null;
      }
    }
    
    // 3. 日志中间件
    class LoggingMiddleware implements RouteMiddleware {
      @override
      RouteSettings? handle(RouteSettings settings) {
        print('Navigating to: ${settings.name}');
        return null;
      }
    }
    
    // 4. 中间件管理器
    class MiddlewareManager {
      final List<RouteMiddleware> middlewares = [];
      
      void add(RouteMiddleware middleware) {
        middlewares.add(middleware);
      }
      
      RouteSettings? process(RouteSettings settings) {
        for (final middleware in middlewares) {
          final result = middleware.handle(settings);
          if (result != null) {
            return result;
          }
        }
        return null;
      }
    }
    
    // 5. 使用中间件
    final manager = MiddlewareManager();
    manager.add(AuthMiddleware());
    manager.add(LoggingMiddleware());
    
    onGenerateRoute: (settings) {
      final processed = manager.process(settings);
      if (processed != null) {
        return MaterialPageRoute(
          builder: (context) => getPage(processed.name),
        );
      }
      return MaterialPageRoute(
        builder: (context) => getPage(settings.name),
      );
    }
    ''');
  }
}
```

## 🔧 中间件模式

### 1. 链式中间件

```dart
// 链式中间件
class ChainMiddleware {
  void explain() {
    print('''
    链式中间件：
    
    // 1. 中间件链
    class MiddlewareChain {
      final List<RouteMiddleware> _middlewares = [];
      
      void add(RouteMiddleware middleware) {
        _middlewares.add(middleware);
      }
      
      RouteSettings? process(RouteSettings settings) {
        RouteSettings? current = settings;
        
        for (final middleware in _middlewares) {
          if (current == null) break;
          current = middleware.handle(current);
        }
        
        return current;
      }
    }
    
    // 2. 使用中间件链
    final chain = MiddlewareChain();
    chain.add(AuthMiddleware());
    chain.add(LoggingMiddleware());
    chain.add(DataMiddleware());
    
    onGenerateRoute: (settings) {
      final processed = chain.process(settings);
      if (processed != null) {
        return MaterialPageRoute(
          builder: (context) => getPage(processed.name),
        );
      }
      return MaterialPageRoute(
        builder: (context) => getPage(settings.name),
      );
    }
    
    // 3. 中间件优先级
    class PriorityMiddleware implements RouteMiddleware {
      final int priority;
      
      PriorityMiddleware(this.priority);
      
      @override
      RouteSettings? handle(RouteSettings settings) {
        // 实现中间件逻辑
        return null;
      }
    }
    
    // 4. 按优先级排序
    middlewares.sort((a, b) => b.priority.compareTo(a.priority));
    ''');
  }
}
```

### 2. 条件中间件

```dart
// 条件中间件
class ConditionalMiddleware {
  void explain() {
    print('''
    条件中间件：
    
    // 1. 条件中间件
    class ConditionalMiddleware implements RouteMiddleware {
      final bool Function(RouteSettings) condition;
      final RouteMiddleware middleware;
      
      ConditionalMiddleware({
        required this.condition,
        required this.middleware,
      });
      
      @override
      RouteSettings? handle(RouteSettings settings) {
        if (condition(settings)) {
          return middleware.handle(settings);
        }
        return null;
      }
    }
    
    // 2. 使用条件中间件
    final middleware = ConditionalMiddleware(
      condition: (settings) => settings.name == '/admin',
      middleware: AuthMiddleware(),
    );
    
    // 3. 路由模式匹配
    class PatternMiddleware implements RouteMiddleware {
      final String pattern;
      final RouteMiddleware middleware;
      
      PatternMiddleware({
        required this.pattern,
        required this.middleware,
      });
      
      @override
      RouteSettings? handle(RouteSettings settings) {
        if (settings.name?.startsWith(pattern) == true) {
          return middleware.handle(settings);
        }
        return null;
      }
    }
    
    // 4. 使用模式匹配
    final middleware = PatternMiddleware(
      pattern: '/admin',
      middleware: AuthMiddleware(),
    );
    ''');
  }
}
```
