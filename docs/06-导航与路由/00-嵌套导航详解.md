# 嵌套导航详解

> 深入理解 Flutter 中的嵌套导航。

## 📖 嵌套导航基础

### 1. 什么是嵌套导航

```dart
// 嵌套导航概念
class NestedNavigationConcept {
  void explain() {
    print('''
    嵌套导航概念：
    
    // 1. 什么是嵌套导航
    // - 在 Navigator 中嵌套 Navigator
    // - 每个 Navigator 有独立的导航栈
    // - 支持复杂的导航结构
    // - 实现模块化导航
    
    // 2. 嵌套导航的特点
    // - 独立栈：每个 Navigator 有独立的栈
    // - 模块化：支持模块化导航
    // - 灵活性：支持复杂导航结构
    // - 状态隔离：导航状态隔离
    
    // 3. 嵌套导航的使用场景
    // - Tab 导航：每个 Tab 独立导航
    // - 模块化应用：每个模块独立导航
    // - 复杂应用：支持复杂导航结构
    // - 多级导航：支持多级导航
    ''');
  }
}
```

### 2. 嵌套导航配置

```dart
// 嵌套导航配置
class NestedNavigationConfiguration {
  void explain() {
    print('''
    嵌套导航配置：
    
    // 1. 基本配置
    Navigator(
      key: _navigatorKey,
      onGenerateRoute: (settings) {
        return MaterialPageRoute(
          builder: (context) => getPage(settings.name),
        );
      },
    )
    
    // 2. 使用 Navigator 嵌套
    Scaffold(
      body: Navigator(
        key: _navigatorKey,
        onGenerateRoute: (settings) {
          return MaterialPageRoute(
            builder: (context) => getPage(settings.name),
          );
        },
      ),
      bottomNavigationBar: BottomNavigationBar(
        items: [
          BottomNavigationBarItem(icon: Icon(Icons.home), label: 'Home'),
          BottomNavigationBarItem(icon: Icon(Icons.search), label: 'Search'),
        ],
        onTap: (index) {
          _navigatorKey.currentState?.pushNamed('/$index');
        },
      ),
    )
    
    // 3. 使用 TabBar 嵌套
    DefaultTabController(
      length: 3,
      child: Scaffold(
        appBar: AppBar(
          bottom: TabBar(
            tabs: [
              Tab(text: 'Home'),
              Tab(text: 'Search'),
              Tab(text: 'Profile'),
            ],
          ),
        ),
        body: TabBarView(
          children: [
            Navigator(
              onGenerateRoute: (settings) {
                return MaterialPageRoute(
                  builder: (context) => HomePage(),
                );
              },
            ),
            Navigator(
              onGenerateRoute: (settings) {
                return MaterialPageRoute(
                  builder: (context) => SearchPage(),
                );
              },
            ),
            Navigator(
              onGenerateRoute: (settings) {
                return MaterialPageRoute(
                  builder: (context) => ProfilePage(),
                );
              },
            ),
          ],
        ),
      ),
    )
    ''');
  }
}
```

## 🔧 嵌套导航使用

### 1. 独立导航栈

```dart
// 独立导航栈
class IndependentNavigationStack {
  void explain() {
    print('''
    独立导航栈：
    
    // 1. 每个 Navigator 独立
    Navigator(
      key: _homeNavigatorKey,
      onGenerateRoute: (settings) {
        return MaterialPageRoute(
          builder: (context) => HomePage(),
        );
      },
    )
    
    // 2. 在嵌套 Navigator 中导航
    _homeNavigatorKey.currentState?.pushNamed('/details');
    
    // 3. 返回嵌套 Navigator
    _homeNavigatorKey.currentState?.pop();
    
    // 4. 管理多个 Navigator
    class MultiNavigatorManager {
      final Map<String, GlobalKey<NavigatorState>> _navigators = {};
      
      void addNavigator(String key, GlobalKey<NavigatorState> navigator) {
        _navigators[key] = navigator;
      }
      
      void navigate(String navigatorKey, String route) {
        _navigators[navigatorKey]?.currentState?.pushNamed(route);
      }
    }
    ''');
  }
}
```

### 2. 模块化导航

```dart
// 模块化导航
class ModularNavigation {
  void explain() {
    print('''
    模块化导航：
    
    // 1. 模块化 Navigator
    class HomeModule extends StatelessWidget {
      final GlobalKey<NavigatorState> navigatorKey;
      
      const HomeModule({required this.navigatorKey});
      
      @override
      Widget build(BuildContext context) {
        return Navigator(
          key: navigatorKey,
          onGenerateRoute: (settings) {
            return MaterialPageRoute(
              builder: (context) => getPage(settings.name),
            );
          },
        );
      }
    }
    
    // 2. 模块化路由
    class HomeRoutes {
      static const String HOME = '/';
      static const String DETAILS = '/details';
      static const String SETTINGS = '/settings';
    }
    
    // 3. 模块化导航管理
    class ModuleNavigator {
      final String moduleName;
      final GlobalKey<NavigatorState> navigatorKey;
      
      ModuleNavigator(this.moduleName, this.navigatorKey);
      
      void navigate(String route) {
        navigatorKey.currentState?.pushNamed(route);
      }
    }
    ''');
  }
}
```

## ⚠️ 注意事项

1. **嵌套导航会增加复杂度**
2. **注意导航栈管理**
3. **考虑状态隔离**
4. **测试不同场景**
