## 📖 路由守卫基础

### 1. 基本概念

class RouteGuardBasics {

### 2. 登录验证

```dart
// 登录验证
class LoginGuard {
  void explain() {
    print('''
    登录验证实现：
    
    // 1. 使用 onGenerateRoute 拦截
    MaterialApp(
      onGenerateRoute: (settings) {
        // 检查是否需要登录
        if (settings.name?.startsWith('/protected') == true) {
          if (!AuthService.isLoggedIn) {
            // 重定向到登录页
            return MaterialPageRoute(
              builder: (context) => LoginPage(),
              settings: RouteSettings(name: '/login'),
            );
          }
        }
        
        // 正常路由
        switch (settings.name) {
          case '/':
            return MaterialPageRoute(builder: (context) => HomePage());
          case '/protected/profile':
            return MaterialPageRoute(builder: (context) => ProfilePage());
          default:
            return MaterialPageRoute(builder: (context) => NotFoundPage());
        }
      },
    )
    
    // 2. 使用 RouteGuard 类
    class RouteGuard {
      static Route<dynamic> guard(RouteSettings settings, Widget page) {
        if (!AuthService.isLoggedIn && settings.name != '/login') {
          return MaterialPageRoute(
            builder: (context) => LoginPage(),
            settings: RouteSettings(name: '/login'),
          );
        }
        return MaterialPageRoute(
          builder: (context) => page,
          settings: settings,
        );
      }
    }
    ''');
  }
}
```

### 3. 权限检查

```dart
// 权限检查
class PermissionGuard {
  void explain() {
    print('''
    权限检查：
    
    // 1. 基于角色的权限
    class RoleGuard {
      static bool hasPermission(String role, String route) {
        Map<String, List<String>> permissions = {
          '/admin': ['admin'],
          '/settings': ['admin', 'user'],
          '/profile': ['admin', 'user', 'guest'],
        };
        
        return permissions[route]?.contains(role) ?? false;
      }
      
      static Route<dynamic> guard(RouteSettings settings, Widget page) {
        final userRole = AuthService.currentUser?.role ?? 'guest';
        
        if (!hasPermission(userRole, settings.name ?? '')) {
          return MaterialPageRoute(
            builder: (context) => AccessDeniedPage(),
          );
        }
        
        return MaterialPageRoute(
          builder: (context) => page,
          settings: settings,
        );
      }
    }
    
    // 2. 使用示例
    MaterialApp(
      onGenerateRoute: (settings) {
        final page = getPage(settings.name);
        return RoleGuard.guard(settings, page);
      },
    )
    ''');
  }
}
```

## 📖 高级路由守卫

### 1. 多重守卫

```dart
// 多重守卫
class MultiGuard {
  void explain() {
    print('''
    多重守卫：
    
    // 守卫链
    class GuardChain {
      static Route<dynamic> apply(
        RouteSettings settings,
        Widget page,
        List<Guard> guards,
      ) {
        for (final guard in guards) {
          final result = guard(settings);
          if (result != null) return result;
        }
        return MaterialPageRoute(
          builder: (context) => page,
          settings: settings,
        );
      }
    }
    
    // 守卫定义
    typedef Guard = Route<dynamic>? Function(RouteSettings settings);
    
    final authGuard = (RouteSettings settings) {
      if (!AuthService.isLoggedIn) {
        return MaterialPageRoute(
          builder: (context) => LoginPage(),
          settings: RouteSettings(name: '/login'),
        );
      }
      return null;  // 通过
    };
    
    final roleGuard = (RouteSettings settings) {
      if (!AuthService.hasRole('admin')) {
        return MaterialPageRoute(
          builder: (context) => AccessDeniedPage(),
        );
      }
      return null;
    };
    
    // 使用
    MaterialApp(
      onGenerateRoute: (settings) {
        return GuardChain.apply(
          settings,
          getPage(settings.name),
          [authGuard, roleGuard],
        );
      },
    )
    ''');
  }
}
```

### 2. 页面级守卫

```dart
// 页面级守卫
class PageGuard {
  void explain() {
    print('''
    页面级守卫：
    
    // 1. 使用 WillPopScope
    class ProtectedPage extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return WillPopScope(
          onWillPop: () async {
            // 检查是否有未保存的更改
            if (hasUnsavedChanges) {
              return await showDialog(
                context: context,
                builder: (context) => AlertDialog(
                  title: Text('提示'),
                  content: Text('有未保存的更改，确定要离开吗？'),
                  actions: [
                    TextButton(
                      onPressed: () => Navigator.pop(context, false),
                      child: Text('取消'),
                    ),
                    TextButton(
                      onPressed: () => Navigator.pop(context, true),
                      child: Text('确定'),
                    ),
                  ],
                ),
              );
            }
            return true;
          },
          child: Scaffold(...),
        );
      }
    }
    
    // 2. 使用 NavigatorObserver
    class MyNavigatorObserver extends NavigatorObserver {
      @override
      void didPush(Route<dynamic> route, Route<dynamic>? previousRoute) {
        print('导航到: ${route.settings.name}');
      }
      
      @override
      void didPop(Route<dynamic> route, Route<dynamic>? previousRoute) {
        print('从 ${route.settings.name} 返回');
      }
    }
    ''');
  }
}
```
