### 1. 什么是页面返回

// 页面返回概念
class PageBackConcept {
  void explain() {

    // - 可以携带返回数据
    // - 支持系统返回按钮
    // - 支持手势返回
    
    // 2. 页面返回的特点
    // - 数据传递：可以返回数据

    // 3. 页面返回的使用场景

### 2. 页面返回配置

```dart
// 页面返回配置
class PageBackConfiguration {
  void explain() {
    print('''
    页面返回配置：
    
    // 1. 基本返回
    Navigator.pop(context);
    
    // 2. 带数据返回
    Navigator.pop(context, {'id': 42, 'title': 'Hello'});
    
    // 3. 返回并获取数据
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => DetailsPage(),
      ),
    ).then((result) {
      // 处理返回数据
      print('Result: $result');
    });
    
    // 4. 检查是否可以返回
    if (Navigator.canPop(context)) {
      Navigator.pop(context);
    }
    
    // 5. 使用 WillPopScope
    WillPopScope(
      onWillPop: () async {
        // 处理返回逻辑
        return true;  // 允许返回
      },
      child: Scaffold(
        appBar: AppBar(
          leading: IconButton(
            icon: Icon(Icons.arrow_back),
            onPressed: () => Navigator.pop(context),
          ),
        ),
        body: Center(child: Text('Page')),
      ),
    )
    ''');
  }
}
```

## 🔧 页面返回使用

### 1. 数据返回

```dart
// 数据返回
class DataReturn {
  void explain() {
    print('''
    数据返回：
    
    // 1. 返回简单数据
    Navigator.pop(context, 'Hello');
    
    // 2. 返回复杂数据
    Navigator.pop(context, {
      'id': 42,
      'title': 'Hello',
      'data': [1, 2, 3],
    });
    
    // 3. 返回对象
    class Result {
      final int id;
      final String title;
      
      Result(this.id, this.title);
    }
    
    Navigator.pop(context, Result(42, 'Hello'));
    
    // 4. 接收返回数据
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => DetailsPage(),
      ),
    ).then((result) {
      if (result != null) {
        final data = result as Map<String, dynamic>;
        print('ID: ${data['id']}');
        print('Title: ${data['title']}');
      }
    });
    ''');
  }
}
```

### 2. 手势返回

```dart
// 手势返回
class GestureBack {
  void explain() {
    print('''
    手势返回：
    
    // 1. 启用手势返回
    // iOS 默认支持
    // Android 需要配置
    
    // 2. 禁用手势返回
    WillPopScope(
      onWillPop: () async => false,  // 禁用返回
      child: Scaffold(
        body: Center(child: Text('Cannot go back')),
      ),
    )
    
    // 3. 自定义返回逻辑
    WillPopScope(
      onWillPop: () async {
        // 显示确认对话框
        final shouldPop = await showDialog<bool>(
          context: context,
          builder: (context) => AlertDialog(
            title: Text('确认返回？'),
            actions: [
              TextButton(
                onPressed: () => Navigator.pop(context, false),
                child: Text('取消'),
              ),
              TextButton(
                onPressed: () => Navigator.pop(context, true),
                child: Text('确定'),
              ),
            ],
          ),
        );
        return shouldPop ?? false;
      },
      child: Scaffold(
        body: Center(child: Text('Page')),
      ),
    )
    
    // 4. 使用 PopScope (Flutter 3.12+)
    PopScope(
      canPop: false,
      onPopInvoked: (didPop) {
        if (didPop) return;
        // 处理返回逻辑
      },
      child: Scaffold(
        body: Center(child: Text('Page')),
      ),
    )
    ''');
  }
}
```
