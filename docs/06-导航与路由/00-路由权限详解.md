### 2. 路由权限配置

```dart
// 路由权限配置
class RoutePermissionsConfiguration {
  void explain() {
    print('''
    路由权限配置：
    
    // 1. 定义权限
    class AppPermissions {
      static const String ADMIN = 'admin';
      static const String USER = 'user';
      static const String GUEST = 'guest';
    }
    
    // 2. 定义路由权限
    class RoutePermissions {
      static Map<String, List<String>> permissions = {
        '/admin': [AppPermissions.ADMIN],
        '/user': [AppPermissions.USER, AppPermissions.ADMIN],
        '/guest': [AppPermissions.GUEST, AppPermissions.USER, AppPermissions.ADMIN],
      };
      
      static bool hasPermission(String route, List<String> userPermissions) {
        final requiredPermissions = permissions[route];
        if (requiredPermissions == null) return true;
        
        return requiredPermissions.any(
          (permission) => userPermissions.contains(permission),
        );
      }
    }
    
    // 3. 使用路由权限
    MaterialApp(
      onGenerateRoute: (settings) {
        final userPermissions = getUserPermissions();
        
        if (!RoutePermissions.hasPermission(settings.name!, userPermissions)) {
          return MaterialPageRoute(
            builder: (context) => AccessDeniedPage(),
          );
        }
        
        return MaterialPageRoute(
          builder: (context) => getPage(settings.name),
        );
      },
    )
    ''');
  }
}
```

## 🔧 路由权限使用

### 1. 角色权限

```dart
// 角色权限
class RolePermissions {
  void explain() {
    print('''
    角色权限：
    
    // 1. 定义角色
    class AppRoles {
      static const String ADMIN = 'admin';
      static const String USER = 'user';
      static const String GUEST = 'guest';
    }
    
    // 2. 角色权限映射
    class RolePermissions {
      static Map<String, List<String>> permissions = {
        AppRoles.ADMIN: ['read', 'write', 'delete', 'admin'],
        AppRoles.USER: ['read', 'write'],
        AppRoles.GUEST: ['read'],
      };
      
      static bool hasPermission(String role, String permission) {
        final rolePermissions = permissions[role];
        if (rolePermissions == null) return false;
        
        return rolePermissions.contains(permission);
      }
    }
    
    // 3. 使用角色权限
    class PermissionGuard {
      static Widget checkRole(String role, Widget page) {
        if (RolePermissions.hasPermission(role, 'read')) {
          return page;
        } else {
          return AccessDeniedPage();
        }
      }
    }
    ''');
  }
}
```

### 2. 动态权限

```dart
// 动态权限
class DynamicPermissions {
  void explain() {
    print('''
    动态权限：
    
    // 1. 动态权限管理
    class DynamicPermissionManager {
      static Map<String, List<String>> _permissions = {};
      
      static void setPermissions(String route, List<String> permissions) {
        _permissions[route] = permissions;
      }
      
      static bool hasPermission(String route, List<String> userPermissions) {
        final requiredPermissions = _permissions[route];
        if (requiredPermissions == null) return true;
        
        return requiredPermissions.any(
          (permission) => userPermissions.contains(permission),
        );
      }
    }
    
    // 2. 动态权限更新
    void updatePermissions() {
      // 从服务器获取权限
      fetchPermissions().then((permissions) {
        DynamicPermissionManager.setPermissions('/admin', permissions);
      });
    }
    
    // 3. 动态权限验证
    void checkDynamicPermission(String route) {
      final userPermissions = getUserPermissions();
      
      if (DynamicPermissionManager.hasPermission(route, userPermissions)) {
        // 允许访问
      } else {
        // 拒绝访问
      }
    }
    ''');
  }
}
```
