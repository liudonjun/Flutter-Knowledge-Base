# 路由拦截详解

> 深入理解 Flutter 中的路由拦截。

## 📖 路由拦截基础

### 1. 什么是路由拦截

```dart
// 路由拦截概念
class RouteInterceptionConcept {
  void explain() {
    print('''
    路由拦截概念：
    
    // 1. 什么是路由拦截
    // - 在路由跳转前后执行逻辑
    // - 可以阻止或重定向路由
    // - 用于权限验证、日志记录等
    // - 保护特定路由
    
    // 2. 路由拦截的特点
    // - 拦截导航：可以阻止或重定向
    // - 权限控制：验证用户权限
    // - 日志记录：记录路由访问
    // - 数据预加载：预加载页面数据
    
    // 3. 路由拦截的使用场景
    // - 登录验证：未登录重定向到登录页
    // - 权限检查：无权限访问重定向
    // - 数据验证：验证路由参数
    // - 日志追踪：记录用户导航路径
    ''');
  }
}
```

### 2. 路由拦截实现

```dart
// 路由拦截实现
class RouteInterceptionImplementation {
  void explain() {
    print('''
    路由拦截实现：
    
    // 1. 使用 Navigator 的 onGenerateRoute
    MaterialApp(
      onGenerateRoute: (settings) {
        // 检查权限
        if (settings.name == '/admin' && !isAdmin) {
          return MaterialPageRoute(
            builder: (context) => AccessDeniedPage(),
          );
        }
        return MaterialPageRoute(
          builder: (context) => getPage(settings.name),
        );
      },
    )
    
    // 2. 使用路由包装器
    class ProtectedRoute extends StatelessWidget {
      final Widget child;
      final bool Function() checkPermission;
      
      const ProtectedRoute({
        required this.child,
        required this.checkPermission,
      });
      
      @override
      Widget build(BuildContext context) {
        if (checkPermission()) {
          return child;
        } else {
          return AccessDeniedPage();
        }
      }
    }
    
    // 3. 使用 MaterialPageRoute 的 builder
    MaterialPageRoute(
      builder: (context) {
        if (!isLoggedIn) {
          return LoginPage();
        }
        return AdminPage();
      },
    )
    ''');
  }
}
```

## 🔧 路由拦截模式

### 1. 登录拦截

```dart
// 登录拦截
class LoginInterception {
  void explain() {
    print('''
    登录拦截：
    
    // 1. 检查登录状态
    class AuthInterceptor {
      static bool isLoggedIn = false;
      
      static Widget checkAuth(Widget page) {
        if (isLoggedIn) {
          return page;
        } else {
          return LoginPage();
        }
      }
    }
    
    // 2. 使用登录拦截
    MaterialPageRoute(
      builder: (context) => AuthInterceptor.checkAuth(ProfilePage()),
    )
    
    // 3. 登录后重定向
    class LoginPage extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return ElevatedButton(
          onPressed: () {
            AuthInterceptor.isLoggedIn = true;
            Navigator.pushReplacementNamed(context, '/profile');
          },
          child: Text('Login'),
        );
      }
    }
    
    // 4. 登出处理
    void logout() {
      AuthInterceptor.isLoggedIn = false;
      Navigator.pushNamedAndRemoveUntil(
        context,
        '/login',
        (route) => false,
      );
    }
    ''');
  }
}
```

### 2. 权限拦截

```dart
// 权限拦截
class PermissionInterception {
  void explain() {
    print('''
    权限拦截：
    
    // 1. 权限检查
    class PermissionInterceptor {
      static bool hasPermission(String permission) {
        // 检查用户权限
        return userPermissions.contains(permission);
      }
      
      static Widget checkPermission(
        String permission,
        Widget page,
      ) {
        if (hasPermission(permission)) {
          return page;
        } else {
          return AccessDeniedPage();
        }
      }
    }
    
    // 2. 使用权限拦截
    MaterialPageRoute(
      builder: (context) => PermissionInterceptor.checkPermission(
        'admin',
        AdminPage(),
      ),
    )
    
    // 3. 多权限检查
    class MultiPermissionInterceptor {
      static Widget checkPermissions(
        List<String> permissions,
        Widget page,
      ) {
        for (final permission in permissions) {
          if (!PermissionInterceptor.hasPermission(permission)) {
            return AccessDeniedPage();
          }
        }
        return page;
      }
    }
    
    // 4. 角色拦截
    class RoleInterceptor {
      static bool hasRole(String role) {
        return userRoles.contains(role);
      }
      
      static Widget checkRole(String role, Widget page) {
        if (hasRole(role)) {
          return page;
        } else {
          return AccessDeniedPage();
        }
      }
    }
    ''');
  }
}
```

## ⚠️ 注意事项

1. **路由拦截应该简洁高效**
2. **避免在拦截中执行耗时操作**
3. **考虑用户体验**
4. **处理拦截失败的情况**
