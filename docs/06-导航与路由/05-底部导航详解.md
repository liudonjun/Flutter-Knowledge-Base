# 底部导航详解

> 掌握 Flutter 中底部导航栏（BottomNavigationBar）的使用。

## 📖 BottomNavigationBar

### 1. 基本用法

```dart
// BottomNavigationBar 基本用法
class BottomNavigationBarBasics {
  void explain() {
    print('''
    BottomNavigationBar 基本用法：
    
    class HomePage extends StatefulWidget {
      @override
      _HomePageState createState() => _HomePageState();
    }
    
    class _HomePageState extends State<HomePage> {
      int _currentIndex = 0;
      
      final List<Widget> _pages = [
        HomeScreen(),
        SearchScreen(),
        ProfileScreen(),
      ];
      
      @override
      Widget build(BuildContext context) {
        return Scaffold(
          body: _pages[_currentIndex],
          bottomNavigationBar: BottomNavigationBar(
            currentIndex: _currentIndex,
            onTap: (index) => setState(() => _currentIndex = index),
            items: [
              BottomNavigationBarItem(
                icon: Icon(Icons.home),
                label: '首页',
              ),
              BottomNavigationBarItem(
                icon: Icon(Icons.search),
                label: '搜索',
              ),
              BottomNavigationBarItem(
                icon: Icon(Icons.person),
                label: '我的',
              ),
            ],
          ),
        );
      }
    }
    ''');
  }
}
```

### 2. 样式配置

```dart
// BottomNavigationBar 样式
class BottomNavigationBarStyles {
  void explain() {
    print('''
    BottomNavigationBar 样式：
    
    BottomNavigationBar(
      // 类型
      type: BottomNavigationBarType.fixed,        // 固定（4个以下）
      type: BottomNavigationBarType.shifting,     // 移动（4个以上）
      
      // 颜色
      backgroundColor: Colors.white,
      selectedItemColor: Colors.blue,
      unselectedItemColor: Colors.grey,
      
      // 图标大小
      iconSize: 24,
      
      // 文字样式
      selectedLabelStyle: TextStyle(fontSize: 14),
      unselectedLabelStyle: TextStyle(fontSize: 12),
      
      // 显示
      showSelectedLabels: true,
      showUnselectedLabels: true,
      
      //  elevation
      elevation: 8,
      
      items: [...],
    )
    
    // 使用 Theme 自定义
    Theme(
      data: Theme.of(context).copyWith(
        canvasColor: Colors.blue[900],
      ),
      child: BottomNavigationBar(...),
    )
    ''');
  }
}
```

### 3. 徽章

```dart
// 带徽章的 BottomNavigationBar
class BadgeBottomNavigationBar {
  void explain() {
    print('''
    带徽章的底部导航：
    
    // 使用 badges 包
    // badges: ^3.0.0
    
    import 'package:badges/badges.dart';
    
    BottomNavigationBar(
      items: [
        BottomNavigationBarItem(
          icon: Badge(
            badgeContent: Text('3', style: TextStyle(color: Colors.white)),
            child: Icon(Icons.home),
          ),
          label: '首页',
        ),
        BottomNavigationBarItem(
          icon: Badge(
            badgeContent: Text('99+'),
            badgeColor: Colors.red,
            child: Icon(Icons.message),
          ),
          label: '消息',
        ),
        BottomNavigationBarItem(
          icon: Icon(Icons.person),
          label: '我的',
        ),
      ],
    )
    
    // 自定义徽章
    BottomNavigationBarItem(
      icon: Stack(
        children: [
          Icon(Icons.notifications),
          Positioned(
            right: 0,
            child: Container(
              padding: EdgeInsets.all(2),
              decoration: BoxDecoration(
                color: Colors.red,
                borderRadius: BorderRadius.circular(10),
              ),
              constraints: BoxConstraints(minWidth: 16, minHeight: 16),
              child: Text('5', style: TextStyle(color: Colors.white, fontSize: 10)),
            ),
          ),
        ],
      ),
      label: '通知',
    )
    ''');
  }
}
```
