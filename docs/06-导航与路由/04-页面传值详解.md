# 页面传值详解

> 掌握 Flutter 中页面间数据传递的各种方式。

## 📖 传值方式

### 1. 构造函数传值

```dart
// 构造函数传值
class ConstructorPassing {
  void explain() {
    print('''
    构造函数传值：
    
    // 定义页面
    class DetailPage extends StatelessWidget {
      final String title;
      final int id;
      
      const DetailPage({
        Key? key,
        required this.title,
        required this.id,
      }) : super(key: key);
      
      @override
      Widget build(BuildContext context) {
        return Scaffold(
          appBar: AppBar(title: Text(title)),
          body: Center(child: Text('ID: $id')),
        );
      }
    }
    
    // 跳转时传值
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => DetailPage(
          title: '详情页',
          id: 123,
        ),
      ),
    );
    
    // 优点：类型安全，编译时检查
    // 缺点：只能传简单数据
    ''');
  }
}
```

### 2. 命名路由传值

```dart
// 命名路由传值
class NamedRoutePassing {
  void explain() {
    print('''
    命名路由传值：
    
    // 定义路由
    MaterialApp(
      routes: {
        '/detail': (context) => DetailPage(),
      },
      onGenerateRoute: (settings) {
        if (settings.name == '/detail') {
          final args = settings.arguments as Map<String, dynamic>;
          return MaterialPageRoute(
            builder: (context) => DetailPage(
              title: args['title'],
              id: args['id'],
            ),
          );
        }
        return null;
      },
    )
    
    // 跳转时传值
    Navigator.pushNamed(
      context,
      '/detail',
      arguments: {'title': '详情页', 'id': 123},
    );
    
    // 在页面中获取
    final args = ModalRoute.of(context)!.settings.arguments as Map<String, dynamic>;
    final title = args['title'];
    final id = args['id'];
    ''');
  }
}
```

### 3. 返回值传递

```dart
// 返回值传递
class ReturnValuePassing {
  void explain() {
    print('''
    返回值传递：
    
    // 页面1：发送页面
    class SendPage extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return ElevatedButton(
          onPressed: () async {
            // 跳转并等待返回值
            final result = await Navigator.push<String>(
              context,
              MaterialPageRoute(builder: (context) => ReceivePage()),
            );
            
            // 处理返回值
            if (result != null) {
              print('收到返回值: $result');
            }
          },
          child: Text('选择'),
        );
      }
    }
    
    // 页面2：返回页面
    class ReceivePage extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        return Column(
          children: [
            ElevatedButton(
              onPressed: () => Navigator.pop(context, '选项A'),
              child: Text('选项A'),
            ),
            ElevatedButton(
              onPressed: () => Navigator.pop(context, '选项B'),
              child: Text('选项B'),
            ),
          ],
        );
      }
    }
    ''');
  }
}
```

## 📖 复杂数据传递

### 1. 对象传递

```dart
// 对象传递
class ObjectPassing {
  void explain() {
    print('''
    对象传递：
    
    // 定义数据模型
    class User {
      final String name;
      final int age;
      final String email;
      
      User({required this.name, required this.age, required this.email});
    }
    
    // 传递对象
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => UserDetailPage(
          user: User(name: 'Alice', age: 25, email: 'alice@example.com'),
        ),
      ),
    );
    
    // 使用 JSON 传递
    Navigator.pushNamed(
      context,
      '/user/detail',
      arguments: {'user': user.toJson()},
    );
    
    // 在页面中解析
    final args = ModalRoute.of(context)!.settings.arguments as Map<String, dynamic>;
    final user = User.fromJson(args['user']);
    ''');
  }
}
```

### 2. 回调传递

```dart
// 回调传递
class CallbackPassing {
  void explain() {
    print('''
    回调传递：
    
    // 定义带回调的页面
    class SelectionPage extends StatelessWidget {
      final Function(String) onSelected;
      
      SelectionPage({required this.onSelected});
      
      @override
      Widget build(BuildContext context) {
        return ListView.builder(
          itemCount: 10,
          itemBuilder: (context, index) {
            return ListTile(
              title: Text('选项 $index'),
              onTap: () {
                onSelected('选项 $index');
                Navigator.pop(context);
              },
            );
          },
        );
      }
    }
    
    // 使用
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => SelectionPage(
          onSelected: (value) {
            print('选择了: $value');
          },
        ),
      ),
    );
    ''');
  }
}
```
