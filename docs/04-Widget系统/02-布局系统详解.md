# 布局系统详解

> 掌握 Flutter 布局系统，创建精美的用户界面。

## 📐 布局基础

### 1. 布局约束原理
```dart
// Flutter 布局约束系统
// 父 Widget 向子 Widget 传递约束
// 子 Widget 根据约束决定自己的大小
// 父 Widget 根据子 Widget 的大小决定位置

// 约束示例
class ConstraintExample extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    // 父 Widget 设置约束
    return Container(
      width: 200,
      height: 200,
      color: Colors.blue,
      child: Container(
        // 子 Widget 可以决定自己的大小（在约束范围内）
        width: 100,
        height: 100,
        color: Colors.red,
      ),
    );
  }
}

// 约束传递
class ConstraintPassing extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Container(
      width: 300,
      height: 300,
      color: Colors.blue,
      child: Container(
        // 子 Widget 接收父 Widget 的约束
        // 可以设置更小的尺寸，但不能超过父 Widget
        width: 200,
        height: 200,
        color: Colors.red,
        child: Container(
          // 孙 Widget 也接收约束
          width: 100,
          height: 100,
          color: Colors.green,
        ),
      ),
    );
  }
}
```

```dart

    child: Text('约束容器'),

  ),
```

### 2. Flow Widget
```dart
// 自定义布局
class FlowExample extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Flow(
      delegate: FlowDelegateExample(),
      children: [
        Container(width: 50, height: 50, color: Colors.red),
        Container(width: 50, height: 50, color: Colors.green),
        Container(width: 50, height: 50, color: Colors.blue),
        Container(width: 50, height: 50, color: Colors.yellow),
      ],
    );
  }
}

class FlowDelegateExample extends FlowDelegate {
  @override
  void paintChildren(FlowPaintingContext context) {
    // 自定义绘制逻辑
    for (int i = 0; i < context.childCount; i++) {
      context.paintChild(
        i,
        transform: Matrix4.translationValues(
          (i % 3) * 60.0,
          (i ~/ 3) * 60.0,
          0,
        ),
      );
    }
  }
  
  @override
  bool shouldRepaint(covariant FlowDelegate oldDelegate) => false;
}
```

## 🎯 布局技巧

### 1. 响应式布局
```dart
// 使用 MediaQuery
class ResponsiveLayout extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    final screenWidth = MediaQuery.of(context).size.width;
    final screenHeight = MediaQuery.of(context).size.height;
    
    if (screenWidth < 600) {
      return _buildMobileLayout();
    } else if (screenWidth < 1200) {
      return _buildTabletLayout();
    } else {
      return _buildDesktopLayout();
    }
  }
  
  Widget _buildMobileLayout() {
    return Column(
      children: [
        Container(height: 100, color: Colors.red),
        Expanded(child: Container(color: Colors.green)),
      ],
    );
  }
  
  Widget _buildTabletLayout() {
    return Row(
      children: [
        Container(width: 200, color: Colors.red),
        Expanded(child: Container(color: Colors.green)),
      ],
    );
  }
  
  Widget _buildDesktopLayout() {
    return Row(
      children: [
        Container(width: 300, color: Colors.red),
        Expanded(child: Container(color: Colors.green)),
        Container(width: 200, color: Colors.blue),
      ],
    );
  }
}

// 使用 LayoutBuilder
class AdaptiveLayout extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return LayoutBuilder(
      builder: (context, constraints) {
        if (constraints.maxWidth < 600) {
          return _buildNarrowLayout();
        } else {
          return _buildWideLayout();
        }
      },
    );
  }
  
  Widget _buildNarrowLayout() {
    return Column(
      children: [
        _buildHeader(),
        _buildContent(),
      ],
    );
  }
  
  Widget _buildWideLayout() {
    return Row(
      children: [
        _buildSidebar(),
        Expanded(child: _buildContent()),
      ],
    );
  }
  
  Widget _buildHeader() => Container(height: 100, color: Colors.red);
  Widget _buildSidebar() => Container(width: 200, color: Colors.green);
  Widget _buildContent() => Container(color: Colors.blue);
}
```

### 2. 布局动画
```dart
// 布局动画
class AnimatedLayout extends StatefulWidget {
  @override
  _AnimatedLayoutState createState() => _AnimatedLayoutState();
}

class _AnimatedLayoutState extends State<AnimatedLayout> {
  bool _isExpanded = false;
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        AnimatedContainer(
          duration: Duration(milliseconds: 300),
          height: _isExpanded ? 200 : 100,
          color: Colors.blue,
          child: Center(
            child: Text(
              _isExpanded ? '展开' : '收缩',
              style: TextStyle(color: Colors.white),
            ),
          ),
        ),
        ElevatedButton(
          onPressed: () {
            setState(() {
              _isExpanded = !_isExpanded;
            });
          },
          child: Text(_isExpanded ? '收缩' : '展开'),
        ),
      ],
    );
  }
}
```

## 📚 最佳实践

### 1. 布局原则
```dart
// 1. 使用合适的布局 Widget
// - 简单布局: Container, Padding, Center
// - 复杂布局: Row, Column, Stack
// - 滚动布局: ListView, GridView

  child: Container(
    child: Container(

// 好的实践
Container(
  padding: EdgeInsets.all(16),
  child: Text('简洁布局'),

// 3. 使用 const 构造函数
const myLayout = const Column(
  children: const [
    Text('Hello'),
    Text('World'),

)
```
