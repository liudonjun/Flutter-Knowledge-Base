    // 4. 自定义性能分析
    class DelayedLoadingTracker {
      final List<DelayedLoadingTiming> _timings = [];
      
      void startTracking() {
        SchedulerBinding.instance.addTimingsCallback((timings) {
          for (final timing in timings) {
            _timings.add(DelayedLoadingTiming(
              buildDuration: timing.buildDuration,
              rasterDuration: timing.rasterDuration,
            ));
          }
        });
      }
      
      void printReport() {
        print('延迟加载性能报告:');
        for (final timing in _timings) {
          print('  build: ${timing.buildDuration.inMilliseconds}ms');
          print('  raster: ${timing.rasterDuration.inMilliseconds}ms');
        }
      }
    }
    ''');
  }
}

### 1. Widget 延迟加载

```dart
// Widget 延迟加载
class WidgetDelayedLoading {
  void explain() {
    print('''
    Widget 延迟加载：
    
    // 1. 为什么 Widget 延迟加载
    // - 减少初始加载时间
    // - 减少内存使用
    // - 提高性能
    // - 改善用户体验
    
    // 2. Widget 延迟加载示例
    class DelayedWidget extends StatefulWidget {
      @override
      _DelayedWidgetState createState() => _DelayedWidgetState();
    }
    
    class _DelayedWidgetState extends State<DelayedWidget> {
      bool _isLoaded = false;
      
      @override
      void initState() {
        super.initState();
        _loadWidget();
      }
      
      Future<void> _loadWidget() async {
        // 延迟加载 Widget
        await Future.delayed(Duration(seconds: 1));
        setState(() {
          _isLoaded = true;
        });
      }
      
      @override
      Widget build(BuildContext context) {
        if (!_isLoaded) {
          return CircularProgressIndicator();
        }
        
        return ExpensiveWidget();
      }
    }
    
    // 3. Widget 延迟加载最佳实践
    // - 使用 Future
    // - 使用 Stream
    // - 使用 Isolate
    // - 使用缓存
    
    // 4. Widget 延迟加载注意事项
    // - 延迟加载时机要适当
    // - 使用占位符
    // - 使用错误处理
    // - 考虑性能影响
    ''');
  }
}
```
