## 🔧 缓存策略实现

### 1. LRU 缓存

```dart
// LRU 缓存
class LRUCache {
  void explain() {
    print('''
    LRU 缓存：
    
    // 1. LRU 缓存实现
    class LRUCache<K, V> {
      final int maxSize;
      final Map<K, V> _cache = {};
      final List<K> _keys = [];
      
      LRUCache({this.maxSize = 100});
      
      V? get(K key) {
        if (!_cache.containsKey(key)) {
          return null;
        }
        
        // 移动到最近使用
        _keys.remove(key);
        _keys.add(key);
        
        return _cache[key];
      }
      
      void put(K key, V value) {
        if (_cache.containsKey(key)) {
          // 更新现有键
          _keys.remove(key);
        } else if (_cache.length >= maxSize) {
          // 淘汰最久未使用的
          final oldestKey = _keys.removeAt(0);
          _cache.remove(oldestKey);
        }
        
        _cache[key] = value;
        _keys.add(key);
      }
      
      void remove(K key) {
        _cache.remove(key);
        _keys.remove(key);
      }
      
      void clear() {
        _cache.clear();
        _keys.clear();
      }
    }
    
    // 2. 使用 LRU 缓存
    final cache = LRUCache<String, Widget>(maxSize: 50);
    
    // 缓存 Widget
    cache.put('key', Widget());
    
    // 获取 Widget
    final widget = cache.get('key');
    
    // 3. LRU 缓存最佳实践
    // - 设置最大大小
    // - 使用适当的数据结构
    // - 定期清理
    // - 监控使用情况
    
    // 4. LRU 缓存注意事项
    // - 键必须唯一
    // - 值必须可复用
    // - 考虑线程安全
    // - 考虑性能影响
    ''');
  }
}
```

### 2. TTL 缓存

```dart
// TTL 缓存
class TTLCache {
  void explain() {
    print('''
    TTL 缓存：
    
    // 1. TTL 缓存实现
    class TTLCache<K, V> {
      final Duration defaultTTL;
      final Map<K, _CacheEntry<V>> _cache = {};
      
      TTLCache({this.defaultTTL = const Duration(minutes: 5)});
      
      V? get(K key) {
        if (!_cache.containsKey(key)) {
          return null;
        }
        
        final entry = _cache[key]!;
        if (entry.isExpired) {
          _cache.remove(key);
          return null;
        }
        
        return entry.value;
      }
      
      void put(K key, V value, {Duration? ttl}) {
        _cache[key] = _CacheEntry(
          value: value,
          expiry: DateTime.now().add(ttl ?? defaultTTL),
        );
      }
      
      void remove(K key) {
        _cache.remove(key);
      }
      
      void clear() {
        _cache.clear();
      }
      
      void cleanup() {
        final now = DateTime.now();
        _cache.removeWhere((key, entry) => entry.expiry.isBefore(now));
      }
    }
    
    class _CacheEntry<V> {
      final V value;
      final DateTime expiry;
      
      _CacheEntry({required this.value, required this.expiry});
      
      bool get isExpired => DateTime.now().isAfter(expiry);
    }
    
    // 2. 使用 TTL 缓存
    final cache = TTLCache<String, Widget>(
      defaultTTL: Duration(minutes: 10),
    );
    
    // 缓存 Widget
    cache.put('key', Widget());
    
    // 获取 Widget
    final widget = cache.get('key');
    
    // 3. TTL 缓存最佳实践
    // - 设置默认 TTL
    // - 使用适当的数据结构
    // - 定期清理
    // - 监控使用情况
    
    // 4. TTL 缓存注意事项
    // - TTL 要适当
    // - 键必须唯一
    // - 值必须可复用
    // - 考虑性能影响
    ''');
  }
}
```
