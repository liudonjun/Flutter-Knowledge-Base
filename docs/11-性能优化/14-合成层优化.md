### 2. 合成性能分析

```dart
// 合成性能分析
class CompositingPerformanceAnalysis {
  void explain() {
    print('''
    合成性能分析：
    
    // 1. 使用 Performance Overlay
    MaterialApp(
      showPerformanceOverlay: true,
      home: MyHomePage(),
    )
    
    // 2. 使用 DevTools
    // - 打开 DevTools
    // - 切换到 Performance 面板
    // - 查看帧图表
    // - 分析合成性能
    
    // 3. 使用 Flutter Inspector
    // - 查看 Widget 树
    // - 分析合成结构
    // - 识别性能问题
    // - 优化合成
    
    // 4. 自定义性能分析
    class CompositingPerformanceTracker {
      final List<CompositingTiming> _timings = [];
      
      void startTracking() {
        SchedulerBinding.instance.addTimingsCallback((timings) {
          for (final timing in timings) {
            _timings.add(CompositingTiming(
              buildDuration: timing.buildDuration,
              rasterDuration: timing.rasterDuration,
            ));
          }
        });
      }
      
      void printReport() {
        print('合成性能报告:');
        for (final timing in _timings) {
          print('  build: ${timing.buildDuration.inMilliseconds}ms');
          print('  raster: ${timing.rasterDuration.inMilliseconds}ms');
        }
      }
    }
    ''');
  }
}
```

### 1. 减少图层数量

```dart
// 减少图层数量
class ReduceLayers {
  void explain() {
    print('''
    减少图层数量：
    
    // 1. 为什么减少图层
    // - 减少合成计算
    // - 提高性能
    // - 减少内存使用
    // - 改善用户体验
    
    // 2. 减少图层的方法
    // - 使用 RepaintBoundary
    // - 使用 CustomPaint
    // - 使用 Canvas
    // - 使用 Picture
    
    // 3. 减少图层的示例
    // 不好
    Stack(
      children: [
        Positioned(child: Widget1()),
        Positioned(child: Widget2()),
        Positioned(child: Widget3()),
      ],
    )
    
    // 好
    RepaintBoundary(
      child: Stack(
        children: [
          Widget1(),
          Widget2(),
          Widget3(),
        ],
      ),
    )
    
    // 4. 减少图层的最佳实践
    // - 使用 RepaintBoundary
    // - 使用 CustomPaint
    // - 使用 Canvas
    // - 保持图层简单
    ''');
  }
}
```
