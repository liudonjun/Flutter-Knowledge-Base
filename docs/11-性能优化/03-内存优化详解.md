## 🔍 内存泄漏检测

### 1. 内存监控
```dart
// 内存监控工具
class MemoryMonitor {
  static final Map<String, WeakReference<Object>> _references = {};
  static final Map<String, int> _counts = {};
  
  // 跟踪对象
  static void track(Object object, String name) {
    _references[name] = WeakReference(object);
    _counts[name] = (_counts[name] ?? 0) + 1;
  }
  
  // 检查内存泄漏
  static void checkLeaks() {
    for (final entry in _references.entries) {
      if (entry.value.target == null) {
        print('内存泄漏: ${entry.key} (${_counts[entry.key]} 次)');
      }
    }
  }
  
  // 打印内存使用情况
  static void printMemoryUsage() {
    print('内存使用情况:');
    for (final entry in _counts.entries) {
      final ref = _references[entry.key];
      final status = ref?.target != null ? '活跃' : '已释放';
      print('  ${entry.key}: ${entry.value} 次 ($status)');
    }
  }
  
  // 获取内存统计
  static Map<String, dynamic> getStatistics() {
    int active = 0;
    int released = 0;
    
    for (final ref in _references.values) {
      if (ref.target != null) {
        active++;
      } else {
        released++;
      }
    }
    
    return {
      'total': _references.length,
      'active': active,
      'released': released,
      'leakRate': released / _references.length * 100,
    };
  }
}
```

### 2. 对象生命周期跟踪
```dart
// 对象生命周期跟踪
class LifecycleTracker {
  static final Map<String, DateTime> _creationTimes = {};
  static final Map<String, DateTime> _destructionTimes = {};
  
  // 记录创建
  static void recordCreation(String name) {
    _creationTimes[name] = DateTime.now();
    print('创建: $name at ${DateTime.now()}');
  }
  
  // 记录销毁
  static void recordDestruction(String name) {
    _destructionTimes[name] = DateTime.now();
    print('销毁: $name at ${DateTime.now()}');
    
    // 计算生命周期
    final creation = _creationTimes[name];
    final destruction = _destructionTimes[name];
    if (creation != null && destruction != null) {
      final duration = destruction.difference(creation);
      print('生命周期: ${duration.inMilliseconds} ms');
    }
  }
  
  // 检查未销毁的对象
  static void checkUndestroyed() {
    final now = DateTime.now();
    for (final entry in _creationTimes.entries) {
      if (!_destructionTimes.containsKey(entry.key)) {
        final age = now.difference(entry.value);
        if (age.inMinutes > 5) {
          print('警告: ${entry.key} 已创建 ${age.inMinutes} 分钟但未销毁');
        }
      }
    }
  }
}
```

## 🎨 图片内存优化

### 1. 图片缓存管理
```dart
// 图片缓存管理
class ImageCacheManager {
  static final Map<String, ImageProvider> _cache = {};
  static final Map<String, DateTime> _cacheTime = {};
  static const int _maxCacheSize = 100;
  static const Duration _maxAge = Duration(hours: 1);
  
  // 获取缓存图片
  static ImageProvider? getCachedImage(String url) {
    final cached = _cache[url];
    final cachedTime = _cacheTime[url];
    
    if (cached != null && cachedTime != null) {
      final age = DateTime.now().difference(cachedTime);
      if (age < _maxAge) {
        return cached;
      } else {
        // 过期，移除缓存
        _cache.remove(url);
        _cacheTime.remove(url);
      }
    }
    
    return null;
  }
  
  // 缓存图片
  static void cacheImage(String url, ImageProvider provider) {
    // 检查缓存大小
    if (_cache.length >= _maxCacheSize) {
      _removeOldest();
    }
    
    _cache[url] = provider;
    _cacheTime[url] = DateTime.now();
  }
  
  // 移除最旧的缓存
  static void _removeOldest() {
    if (_cacheTime.isEmpty) return;
    
    String? oldestKey;
    DateTime? oldestTime;
    
    for (final entry in _cacheTime.entries) {
      if (oldestTime == null || entry.value.isBefore(oldestTime)) {
        oldestTime = entry.value;
        oldestKey = entry.key;
      }
    }
    
    if (oldestKey != null) {
      _cache.remove(oldestKey);
      _cacheTime.remove(oldestKey);
    }
  }
  
  // 清除所有缓存
  static void clearCache() {
    _cache.clear();
    _cacheTime.clear();
  }
  
  // 获取缓存统计
  static Map<String, dynamic> getStatistics() {
    return {
      'cacheSize': _cache.length,
      'maxCacheSize': _maxCacheSize,
      'oldestEntry': _cacheTime.values.isEmpty ? null : 
          _cacheTime.values.reduce((a, b) => a.isBefore(b) ? a : b),
    };
  }
}
```

### 2. 图片加载优化
```dart
// 图片加载优化
class OptimizedImage extends StatelessWidget {
  final String imageUrl;
  final double? width;
  final double? height;
  final BoxFit fit;
  
  const OptimizedImage({
    Key? key,
    required this.imageUrl,
    this.width,
    this.height,
    this.fit = BoxFit.cover,
  }) : super(key: key);
  
  @override
  Widget build(BuildContext context) {
    // 检查缓存
    final cachedImage = ImageCacheManager.getCachedImage(imageUrl);
    if (cachedImage != null) {
      return Image(
        image: cachedImage,
        width: width,
        height: height,
        fit: fit,
      );
    }
    
    // 加载新图片
    return Image.network(
      imageUrl,
      width: width,
      height: height,
      fit: fit,
      // 设置缓存尺寸
      cacheWidth: width?.toInt(),
      cacheHeight: height?.toInt(),
      // 加载占位符
      loadingBuilder: (context, child, loadingProgress) {
        if (loadingProgress == null) {
          // 缓存图片
          ImageCacheManager.cacheImage(imageUrl, NetworkImage(imageUrl));
          return child;
        }
        return Center(
          child: CircularProgressIndicator(
            value: loadingProgress.expectedTotalBytes != null
                ? loadingProgress.cumulativeBytesLoaded /
                    loadingProgress.expectedTotalBytes!
                : null,
          ),
        );
      },
      // 错误处理
      errorBuilder: (context, error, stackTrace) {
        return Icon(Icons.error, size: 50);
      },
    );
  }
}
```

```dart
// 列表懒加载
class LazyLoadingList extends StatefulWidget {
  @override
  _LazyLoadingListState createState() => _LazyLoadingListState();
}

class _LazyLoadingListState extends State<LazyLoadingList> {
  final List<String> _items = [];
  bool _isLoading = false;
  int _page = 0;
  
  @override
  void initState() {
    super.initState();
    _loadMore();
  }
  
  Future<void> _loadMore() async {
    if (_isLoading) return;
    
    setState(() {
      _isLoading = true;
    });
    
    // 模拟网络请求
    await Future.delayed(Duration(seconds: 1));
    
    final newItems = List.generate(20, (index) => 'Item ${_page * 20 + index}');
    
    setState(() {
      _items.addAll(newItems);
      _page++;
      _isLoading = false;
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return NotificationListener<ScrollNotification>(
      onNotification: (ScrollNotification scrollInfo) {
        if (scrollInfo.metrics.pixels == scrollInfo.metrics.maxScrollExtent) {
          _loadMore();
        }
        return false;
      },
      child: ListView.builder(
        itemCount: _items.length + (_isLoading ? 1 : 0),
        itemBuilder: (context, index) {
          if (index == _items.length) {
            return Center(child: CircularProgressIndicator());
          }
          
          return ListTile(
            title: Text(_items[index]),
          );
        },
      ),
    );
  }
}
```

### 3. 使用对象池
```dart
// 对象池模式
class ObjectPool<T> {
  final List<T> _pool = [];
  final T Function() _factory;
  final int _maxSize;
  
  ObjectPool(this._factory, {int maxSize = 100}) : _maxSize = maxSize;
  
  // 获取对象
  T acquire() {
    if (_pool.isNotEmpty) {
      return _pool.removeLast();
    }
    return _factory();
  }
  
  // 释放对象
  void release(T object) {
    if (_pool.length < _maxSize) {
      _pool.add(object);
    }
  }
  
  // 清空池
  void clear() {
    _pool.clear();
  }
  
  // 获取池大小
  int get size => _pool.length;
}

// 使用示例
class MyObject {
  final String id;
  MyObject(this.id);
}

final objectPool = ObjectPool<MyObject>(() => MyObject(DateTime.now().toString()));

void useObjectPool() {
  // 获取对象
  final obj = objectPool.acquire();
  
  // 使用对象
  print('使用对象: ${obj.id}');
  
  // 释放对象
  objectPool.release(obj);
}
```

```dart
// 内存分析工具
class MemoryAnalyzer {
  static final Map<String, int> _allocations = {};
  static final Map<String, int> _deallocations = {};
  
  // 记录分配
  static void recordAllocation(String type) {
    _allocations[type] = (_allocations[type] ?? 0) + 1;
  }
  
  // 记录释放
  static void recordDeallocation(String type) {
    _deallocations[type] = (_deallocations[type] ?? 0) + 1;
  }
  
  // 打印统计
  static void printStatistics() {
    print('内存分配统计:');
    for (final type in _allocations.keys) {
      final allocated = _allocations[type] ?? 0;
      final deallocated = _deallocations[type] ?? 0;
      final leaked = allocated - deallocated;
      
      print('  $type: 分配 $allocated, 释放 $deallocated, 泄漏 $leaked');
    }
  }
  
  // 检查泄漏
  static List<String> checkLeaks() {
    final leaks = <String>[];
    
    for (final type in _allocations.keys) {
      final allocated = _allocations[type] ?? 0;
      final deallocated = _deallocations[type] ?? 0;
      
      if (allocated > deallocated) {
        leaks.add('$type: ${allocated - deallocated} 个泄漏');
      }
    }
    
    return leaks;
  }
}
```

### 2. 内存快照
```dart
// 内存快照
class MemorySnapshot {
  final DateTime timestamp;
  final Map<String, int> allocations;
  final Map<String, int> activeObjects;
  
  MemorySnapshot({
    required this.timestamp,
    required this.allocations,
    required this.activeObjects,
  });
  
  // 比较快照
  static MemorySnapshot diff(MemorySnapshot a, MemorySnapshot b) {
    final diffAllocations = <String, int>{};
    final diffActive = <String, int>{};
    
    for (final key in {...a.allocations.keys, ...b.allocations.keys}) {
      final aCount = a.allocations[key] ?? 0;
      final bCount = b.allocations[key] ?? 0;
      diffAllocations[key] = bCount - aCount;
    }
    
    for (final key in {...a.activeObjects.keys, ...b.activeObjects.keys}) {
      final aCount = a.activeObjects[key] ?? 0;
      final bCount = b.activeObjects[key] ?? 0;
      diffActive[key] = bCount - aCount;
    }
    
    return MemorySnapshot(
      timestamp: b.timestamp,
      allocations: diffAllocations,
      activeObjects: diffActive,
    );
  }
  
  // 打印差异
  void printDiff() {
    print('内存变化:');
    for (final entry in allocations.entries) {
      if (entry.value != 0) {
        print('  ${entry.key}: ${entry.value > 0 ? "+" : ""}${entry.value}');
      }
    }
  }
}
```

### 1. 内存监控

```dart
// 内存监控
class MemoryPerformanceMonitor {
  static final List<MemorySnapshot> _snapshots = [];
  static const int _maxSnapshots = 100;
  
  // 创建快照
  static MemorySnapshot takeSnapshot() {
    final snapshot = MemorySnapshot(
      timestamp: DateTime.now(),
      allocations: Map.from(MemoryAnalyzer._allocations),
      activeObjects: Map.from(MemoryMonitor._counts),
    );
    
    _snapshots.add(snapshot);
    if (_snapshots.length > _maxSnapshots) {
      _snapshots.removeAt(0);
    }
    
    return snapshot;
  }
  
  // 分析趋势
  static void analyzeTrends() {
    if (_snapshots.length < 2) return;
    
    final first = _snapshots.first;
    final last = _snapshots.last;
    final diff = MemorySnapshot.diff(first, last);
    
    print('内存趋势分析:');
    print('  时间跨度: ${last.timestamp.difference(first.timestamp).inMinutes} 分钟');
    
    for (final entry in diff.allocations.entries) {
      if (entry.value > 100) {
        print('  警告: ${entry.key} 分配增长 ${entry.value}');
      }
    }
  }
  
  // 检测内存峰值
  static void detectPeaks() {
    if (_snapshots.isEmpty) return;
    
    int maxAllocations = 0;
    MemorySnapshot? peakSnapshot;
    
    for (final snapshot in _snapshots) {
      final total = snapshot.allocations.values.fold(0, (a, b) => a + b);
      if (total > maxAllocations) {
        maxAllocations = total;
        peakSnapshot = snapshot;
      }
    }
    
    if (peakSnapshot != null) {
      print('内存峰值: $maxAllocations at ${peakSnapshot.timestamp}');
    }
  }
}
```

```dart
// 内存优化流程
class MemoryOptimization {
  static Future<void> optimize() async {
    // 1. 内存分析
    await _analyzeMemory();
    
    // 2. 识别问题
    final issues = await _identifyIssues();
    
    // 3. 实施优化
    await _applyOptimizations(issues);
    
    // 4. 验证结果
    await _verifyResults();
  }
  
  static Future<void> _analyzeMemory() async {
    // 使用 DevTools 分析
    // 1. 打开内存面板
    // 2. 记录内存快照
    // 3. 分析内存使用
  }
  
  static Future<List<String>> _identifyIssues() async {
    // 识别内存问题
    return MemoryAnalyzer.checkLeaks();
  }
  
  static Future<void> _applyOptimizations(List<String> issues) async {
    // 应用优化
    for (final issue in issues) {
      print('修复: $issue');
    }
  }
  
  static Future<void> _verifyResults() async {
    // 验证优化结果
    final snapshots = MemoryPerformanceMonitor._snapshots;
    if (snapshots.length >= 2) {
      final diff = MemorySnapshot.diff(snapshots.first, snapshots.last);
      diff.printDiff();
    }
  }
```
