### 2. 内存监控

```dart
// 内存监控
class MemoryMonitor {
  void explain() {
    print('''
    内存监控：
    
    // 1. 使用 DevTools Memory 面板
    // - 打开 DevTools
    // - 切换到 Memory 面板
    // - 点击 Snapshot 按钮
    // - 分析内存快照
    
    // 2. 监听内存事件
    // - 添加内存压力监听器
    // - 在内存不足时释放资源
    // - 监控内存变化
    
    // 3. 自定义内存监控
    class MemoryTracker {
      final List<MemoryUsage> _usageHistory = [];
      
      void startTracking() {
        // 定期记录内存使用
        Timer.periodic(Duration(seconds: 5), (timer) {
          final usage = _getCurrentMemoryUsage();
          _usageHistory.add(usage);
        });
      }
      
      MemoryUsage _getCurrentMemoryUsage() {
        // 获取当前内存使用
        return MemoryUsage(
          timestamp: DateTime.now(),
          dartHeap: 0,  // Dart 堆内存
          nativeHeap: 0,  // 原生内存
          gpuMemory: 0,  // GPU 内存
        );
      }
      
      List<MemoryUsage> get usageHistory => _usageHistory;
    }
    
    // 4. 使用 dart:developer
    import 'dart:developer';
    
    void logMemoryUsage() {
      final current = Service.getIsolateID();
      print('当前 Isolate: $current');
    }
    ''');
  }
}
```

## 🔧 内存优化

### 1. 避免内存泄漏

```dart
// 避免内存泄漏
class AvoidMemoryLeaks {
  void explain() {
    print('''
    避免内存泄漏：
    
    // 1. 及时释放资源
    class MyController {
      StreamSubscription? _subscription;
      Timer? _timer;
      
      void init() {
        _subscription = stream.listen((data) {});
        _timer = Timer.periodic(Duration(seconds: 1), (timer) {});
      }
      
      void dispose() {
        _subscription?.cancel();  // 取消订阅
        _timer?.cancel();         // 取消定时器
      }
    }
    
    // 2. 取消监听器
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      
      @override
      void dispose() {
        _controller.dispose();  // 释放控制器
        super.dispose();
      }
    }
    
    // 3. 关闭控制器
    class MyBloc {
      final _controller = StreamController<int>();
      
      void dispose() {
        _controller.close();  // 关闭流控制器
      }
    }
    
    // 4. 使用 WeakReference
    import 'dart:core';
    
    class MyCache {
      final Map<String, WeakReference<Object>> _cache = {};
      
      void set(String key, Object value) {
        _cache[key] = WeakReference(value);
      }
      
      Object? get(String key) {
        return _cache[key]?.target;
      }
    }
    ''');
  }
}
```

### 2. 优化图片内存

```dart
// 优化图片内存
class OptimizeImageMemory {
  void explain() {
    print('''
    优化图片内存：
    
    // 1. 使用适当尺寸
    Image.network(
      'https://example.com/image.jpg',
      width: 100,
      height: 100,
      cacheWidth: 100,   // 缓存尺寸
      cacheHeight: 100,  // 缓存尺寸
    )
    
    // 2. 使用缩略图
    Image.network(
      'https://example.com/image_small.jpg',
      width: 50,
      height: 50,
    )
    
    // 3. 使用缓存
    CachedNetworkImage(
      imageUrl: 'https://example.com/image.jpg',
      width: 100,
      height: 100,
      placeholder: (context, url) => CircularProgressIndicator(),
      errorWidget: (context, url, error) => Icon(Icons.error),
    )
    
    // 4. 使用内存缓存
    class ImageCache {
      final Map<String, Image> _cache = {};
      final int maxSize;
      
      ImageCache({this.maxSize = 100});
      
      Image? get(String url) {
        return _cache[url];
      }
      
      void set(String url, Image image) {
        if (_cache.length >= maxSize) {
          // 移除最早添加的图片
          final oldest = _cache.keys.first;
          _cache.remove(oldest);
        }
        _cache[url] = image;
      }
    }
    
    // 5. 使用低内存模式
    // 在 pubspec.yaml 中配置
    flutter:
      assets:
        - assets/images/
    ''');
  }
}
```
