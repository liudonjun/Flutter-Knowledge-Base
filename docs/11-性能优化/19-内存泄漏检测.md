## 🔧 内存泄漏修复

### 1. 常见内存泄漏

```dart
// 常见内存泄漏
class CommonMemoryLeaks {
  void explain() {
    print('''
    常见内存泄漏：
    
    // 1. 未释放资源
    class MyController {
      StreamSubscription? _subscription;
      
      void init() {
        _subscription = stream.listen((data) {});
      }
      
      void dispose() {
        _subscription?.cancel();  // 必须取消订阅
      }
    }
    
    // 2. 未取消监听器
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      
      @override
      void dispose() {
        _controller.dispose();  // 必须释放控制器
        super.dispose();
      }
    }
    
    // 3. 未关闭控制器
    class MyBloc {
      final _controller = StreamController<int>();
      
      void dispose() {
        _controller.close();  // 必须关闭流控制器
      }
    }
    
    // 4. 未释放缓存
    class MyCache {
      final Map<String, Object> _cache = {};
      
      void clear() {
        _cache.clear();  // 必须清理缓存
      }
    }
    ''');
  }
}
```

### 2. 内存泄漏修复方法

```dart
// 内存泄漏修复方法
class MemoryLeakFix {
  void explain() {
    print('''
    内存泄漏修复方法：
    
    // 1. 及时释放资源
    // - 在 dispose 方法中释放
    // - 使用 try-finally
    // - 使用自动释放
    // - 使用 WeakReference
    
    // 2. 使用 WeakReference
    import 'dart:core';
    
    class MyCache {
      final Map<String, WeakReference<Object>> _cache = {};
      
      void set(String key, Object value) {
        _cache[key] = WeakReference(value);
      }
      
      Object? get(String key) {
        return _cache[key]?.target;
      }
    }
    
    // 3. 使用自动释放
    class MyResource {
      void useResource() {
        try {
          // 使用资源
        } finally {
          // 释放资源
        }
      }
    }
    
    // 4. 使用内存管理最佳实践
    // - 及时释放资源
    // - 使用 WeakReference
    // - 使用自动释放
    // - 定期清理缓存
    ''');
  }
}
```
