## 🔧 内存回收实现

### 1. 使用 dispose 方法

```dart
// 使用 dispose 方法
class UseDispose {
  void explain() {
    print('''
    使用 dispose 方法：
    
    // 1. 为什么使用 dispose
    // - 释放资源
    // - 避免内存泄漏
    // - 提高性能
    // - 改善用户体验
    
    // 2. dispose 示例
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      final _controller = TextEditingController();
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {});
      }
      
      @override
      void dispose() {
        _controller.dispose();  // 释放控制器
        _subscription?.cancel();  // 取消订阅
        super.dispose();
      }
      
      @override
      Widget build(BuildContext context) {
        return TextField(controller: _controller);
      }
    }
    
    // 3. dispose 最佳实践
    // - 在 dispose 中释放所有资源
    // - 使用 try-finally
    // - 使用自动释放
    // - 使用 WeakReference
    
    // 4. dispose 注意事项
    // - 必须调用 super.dispose()
    // - 必须释放所有资源
    // - 避免在 dispose 后使用资源
    // - 考虑性能影响
    ''');
  }
}
```

### 2. 使用 WeakReference

```dart
// 使用 WeakReference
class UseWeakReference {
  void explain() {
    print('''
    使用 WeakReference：
    
    // 1. 为什么使用 WeakReference
    // - 允许垃圾回收
    // - 避免内存泄漏
    // - 提高性能
    // - 改善用户体验
    
    // 2. WeakReference 示例
    import 'dart:core';
    
    class MyCache {
      final Map<String, WeakReference<Object>> _cache = {};
      
      void set(String key, Object value) {
        _cache[key] = WeakReference(value);
      }
      
      Object? get(String key) {
        return _cache[key]?.target;
      }
      
      void clear() {
        _cache.clear();
      }
    }
    
    // 3. WeakReference 最佳实践
    // - 用于缓存
    // - 用于临时引用
    // - 用于可回收对象
    // - 使用适当的数据结构
    
    // 4. WeakReference 注意事项
    // - WeakReference 可能为 null
    // - 需要检查 target
    // - 考虑性能影响
    // - 使用适当的数据结构
    ''');
  }
}
```
