### 2. 初始化分析

```dart
// 初始化分析
class InitializationAnalysis {
  void explain() {
    print('''
    初始化分析：
    
    // 1. 使用 Performance Overlay
    MaterialApp(
      showPerformanceOverlay: true,
      home: MyHomePage(),
    )
    
    // 2. 使用 DevTools
    // - 打开 DevTools
    // - 切换到 Performance 面板
    // - 查看初始化帧
    // - 分析初始化流程
    
    // 3. 使用 Flutter Inspector
    // - 查看 Widget 树
    // - 分析初始化结构
    // - 识别性能问题
    // - 优化初始化流程
    
    // 4. 自定义性能分析
    class InitializationTracker {
      final List<InitializationTiming> _timings = [];
      
      void startTracking() {
        SchedulerBinding.instance.addTimingsCallback((timings) {
          for (final timing in timings) {
            _timings.add(InitializationTiming(
              buildDuration: timing.buildDuration,
              rasterDuration: timing.rasterDuration,
            ));
          }
        });
      }
      
      void printReport() {
        print('初始化性能报告:');
        for (final timing in _timings) {
          print('  build: ${timing.buildDuration.inMilliseconds}ms');
          print('  raster: ${timing.rasterDuration.inMilliseconds}ms');
        }
      }
    }
    ''');
  }
}
```

### 1. 延迟初始化

```dart
// 延迟初始化
class DelayedInitialization {
  void explain() {
    print('''
    延迟初始化：
    
    // 1. 为什么延迟初始化
    // - 减少启动时间
    // - 提高性能
    // - 减少内存使用
    // - 改善用户体验
    
    // 2. 延迟初始化示例
    class App {
      late final Database _database;
      late final ApiService _apiService;
      
      Future<void> init() async {
        // 延迟初始化
        _database = await Database.init();
        _apiService = await ApiService.init();
      }
    }
    
    // 3. 延迟初始化最佳实践

    // - 确保初始化完成
    // - 避免空指针
    // - 考虑线程安全
    // - 考虑性能影响
    ''');
  }
}
```
