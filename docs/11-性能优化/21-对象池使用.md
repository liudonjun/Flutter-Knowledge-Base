## 🔧 对象池实现

### 1. 简单对象池

```dart
// 简单对象池
class SimpleObjectPool {
  void explain() {
    print('''
    简单对象池：
    
    // 1. 简单对象池实现
    class ObjectPool<T> {
      final List<T> _pool = [];
      final T Function() _factory;
      final int maxSize;
      
      ObjectPool(this._factory, {this.maxSize = 100});
      
      T get() {
        if (_pool.isNotEmpty) {
          return _pool.removeLast();
        }
        return _factory();
      }
      
      void release(T object) {
        if (_pool.length < maxSize) {
          _pool.add(object);
        }
      }
      
      void clear() {
        _pool.clear();
      }
    }
    
    // 2. 使用对象池
    final pool = ObjectPool<Widget>(() => Container());
    
    // 获取对象
    final widget = pool.get();
    
    // 使用对象
    // ...
    
    // 释放对象
    pool.release(widget);
    
    // 3. 对象池最佳实践
    // - 设置最大大小
    // - 使用工厂方法
    // - 定期清理
    // - 监控使用情况
    
    // 4. 对象池注意事项
    // - 对象必须可复用
    // - 对象必须可重置
    // - 考虑线程安全
    // - 考虑性能影响
    ''');
  }
}
```

### 2. Widget 对象池

```dart
// Widget 对象池
class WidgetObjectPool {
  void explain() {
    print('''
    Widget 对象池：
    
    // 1. Widget 对象池实现
    class WidgetPool {
      final Map<Type, List<Widget>> _pool = {};
      final int maxSize;
      
      WidgetPool({this.maxSize = 50});
      
      Widget get(Type type, Widget Function() factory) {
        if (!_pool.containsKey(type)) {
          _pool[type] = [];
        }
        
        if (_pool[type]!.isNotEmpty) {
          return _pool[type]!.removeLast();
        }
        
        return factory();
      }
      
      void release(Widget widget) {
        final type = widget.runtimeType;
        if (!_pool.containsKey(type)) {
          _pool[type] = [];
        }
        
        if (_pool[type]!.length < maxSize) {
          _pool[type]!.add(widget);
        }
      }
      
      void clear() {
        _pool.clear();
      }
    }
    
    // 2. 使用 Widget 对象池
    final pool = WidgetPool();
    
    // 获取 Widget
    final widget = pool.get(Container, () => Container());
    
    // 使用 Widget
    // ...
    
    // 释放 Widget
    pool.release(widget);
    
    // 3. Widget 对象池最佳实践
    // - 使用 const 构造函数
    // - 使用 RepaintBoundary
    // - 使用 child 参数
    // - 定期清理
    
    // 4. Widget 对象池注意事项
    // - Widget 必须可复用
    // - Widget 必须可重置
    // - 考虑性能影响
    // - 考虑内存使用
    ''');
  }
}
```
