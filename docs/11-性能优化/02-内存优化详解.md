## 📖 内存管理

### 1. 资源释放

```dart
// 资源释放
class ResourceDisposal {
  void explain() {
    print('''
    资源释放：
    
    class MyWidget extends StatefulWidget {
      @override
      _MyWidgetState createState() => _MyWidgetState();
    }
    
    class _MyWidgetState extends State<MyWidget> {
      // 声明资源
      late TextEditingController _controller;
      late AnimationController _animationController;
      StreamSubscription? _subscription;
      Timer? _timer;
      
      @override
      void initState() {
        super.initState();
        
        // 初始化资源
        _controller = TextEditingController();
        _animationController = AnimationController(
          vsync: this,
          duration: Duration(seconds: 1),
        );
        
        // 订阅流
        _subscription = stream.listen((data) {});
        
        // 创建定时器
        _timer = Timer.periodic(Duration(seconds: 1), (_) {});
      }
      
      @override
      void dispose() {
        // 释放所有资源
        _controller.dispose();
        _animationController.dispose();
        _subscription?.cancel();
        _timer?.cancel();
        
        super.dispose();
      }
      
      @override
      Widget build(BuildContext context) {
        return TextField(controller: _controller);
      }
    }
    ''');
  }
}
```

### 2. 内存泄漏检测

```dart
// 内存泄漏检测
class MemoryLeakDetection {
  void explain() {
    print('''
    内存泄漏检测：
    
    // 1. 常见内存泄漏
    // - 未取消的监听器
    // - 未释放的控制器
    // - 未关闭的流
    // - 循环引用
    
    // 2. 检测方法
    // - 使用 DevTools Memory 面板
    // - 拍摄堆快照
    // - 比较快照
    
    // 3. 避免泄漏
    class SafeWidget extends StatefulWidget {
      @override
      _SafeWidgetState createState() => _SafeWidgetState();
    }
    
    class _SafeWidgetState extends State<SafeWidget> {
      StreamSubscription? _subscription;
      
      @override
      void initState() {
        super.initState();
        _subscription = stream.listen((data) {
          if (mounted) {  // 检查是否还在树中
            setState(() {
              // 更新状态
            });
          }
        });
      }
      
      @override
      void dispose() {
        _subscription?.cancel();
        super.dispose();
      }
    }
    ''');
  }
}
```

## 📖 图片优化

### 1. 图片内存

```dart
// 图片内存优化
class ImageMemory {
  void explain() {
    print('''
    图片内存优化：
    
    // 1. 指定缓存尺寸
    Image.network(
      'https://example.com/image.jpg',
      width: 200,
      height: 200,
      cacheWidth: 200,   // 缓存宽度
      cacheHeight: 200,  // 缓存高度
    )
    
    // 2. 使用缓存库
    CachedNetworkImage(
      imageUrl: 'https://example.com/image.jpg',
      width: 200,
      height: 200,
      memCacheWidth: 200,
      memCacheHeight: 200,
    )
    
    // 3. 预加载
    precacheImage(
      NetworkImage('https://example.com/image.jpg'),
      context,
    )
    
    // 4. 清理缓存
    imageCache.clear();
    imageCache.maximumSize = 100;
    imageCache.maximumSizeBytes = 50 * 1024 * 1024;  // 50MB
    ''');
  }
}
```

### 2. 列表优化

```dart
// 列表内存优化
class ListMemory {
  void explain() {
    print('''
    列表内存优化：
    
    // 1. 使用 builder
    ListView.builder(
      itemCount: items.length,
      itemBuilder: (context, index) => ItemWidget(items[index]),
    )
    
    // 2. 固定高度
    ListView.builder(
      itemExtent: 80.0,  // 固定每项高度
      itemCount: items.length,
      itemBuilder: (context, index) => ItemWidget(items[index]),
    )
    
    // 3. 缓存范围
    ListView.builder(
      cacheExtent: 500,  // 缓存区域
      itemCount: items.length,
      itemBuilder: (context, index) => ItemWidget(items[index]),
    )
    
    // 4. 保持状态
    class KeepAliveItem extends StatefulWidget {
      @override
      _KeepAliveItemState createState() => _KeepAliveItemState();
    }
    
    class _KeepAliveItemState extends State<KeepAliveItem>
        with AutomaticKeepAliveClientMixin {
      @override
      bool get wantKeepAlive => true;
      
      @override
      Widget build(BuildContext context) {
        super.build(context);
        return Container();
      }
    }
    ''');
  }
}
```
