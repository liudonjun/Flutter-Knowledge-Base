```dart
// 常见布局问题
class LayoutIssues {
  /*
  常见布局问题：
  
  1. 溢出问题
     - RenderFlex overflow
     - 黄黑条纹警告
     - 内容超出边界
  
  2. 约束冲突
     - 无限高度/宽度
     - 约束不匹配
     - 布局循环
  
  3. 性能问题
     - 过度重建
     - 布局抖动
     - 深层嵌套
  */
  
  void explain() {
    print('''
    常见布局问题：
    
    1. 溢出问题
       // RenderFlex overflow 错误
       Row(
         children: [
           Text('很长的文本...'), // 可能溢出
         ],
       )
       
       // 解决方案
       Row(
         children: [
           Expanded(
             child: Text('很长的文本...'),
           ),
         ],
       )
    
    2. 约束冲突
       // 无限高度问题
       Column(
         children: [
           ListView( // ListView 需要有限高度
             children: [...],
           ),
         ],
       )
       
       // 解决方案
       Column(
         children: [
           Expanded(
             child: ListView(
               children: [...],
             ),
           ),
         ],
       )
    
    3. 布局调试工具
       // 启用布局调试
       debugPaintSizeEnabled = true;
       debugPrintLayouts = true;
       
       // 使用 LayoutBuilder
       LayoutBuilder(
         builder: (context, constraints) {
           print('约束: $constraints');
           return Container();
         },
       )
    ''');
  }
}
```

```dart
class PaintingPerformance {

  绘制性能分析：
  
  1. 重绘问题
     - 不必要的重绘
     - 过度重绘
     - 重绘区域过大
  
  2. 绘制优化
     - 使用 RepaintBoundary
     - 减少透明度
     - 优化绘制操作
  
  3. 性能监控
     - Performance Overlay
     - DevTools
     - 帧率监控
  */
  
  void explain() {
    print('''
    绘制性能分析：
    
    1. 重绘问题识别
       // 启用重绘可视化
       debugRepaintRainbowEnabled = true;
       
       // 每次重绘会显示不同颜色
       // 频繁重绘的区域需要优化
    
    2. 使用 RepaintBoundary
       // 隔离重绘区域
       RepaintBoundary(
         child: Container(
           color: Colors.blue,
           child: CustomPaint(
             painter: MyPainter(),
           ),
         ),
       )
       
       // 复杂动画使用 RepaintBoundary
       RepaintBoundary(
         child: AnimatedWidget(),
       )
    
    3. 减少透明度
       // 不推荐：使用 Opacity
       Opacity(
         opacity: 0.5,
         child: Container(...),
       )
       
       // 推荐：使用 Colors.transparent
       Container(
         color: Colors.blue.withOpacity(0.5),
       )
       
       // 或使用 DecoratedBox
       DecoratedBox(
         decoration: BoxDecoration(
           color: Colors.blue,
           borderRadius: BorderRadius.circular(8),
         ),
       )
    ''');
  }
}
```

```dart
// 合成优化
class CompositingOptimization {
  /*
  合成优化：
  
  1. 图层优化
     - 减少图层数量
     - 使用缓存
     - 避免过度图层
  
  2. 绘制优化
     - 批量绘制
     - 使用 Canvas 高效 API
     - 避免复杂路径
  
  3. 硬件加速
     - 使用 Transform
     - 使用 Opacity (谨慎)
     - 使用 Shader
  */
  
  void explain() {
    print('''
    合成优化：
    
    1. 图层优化
       // 不推荐：创建过多图层
       for (int i = 0; i < 100; i++) {
         Opacity(
           opacity: 0.5,
           child: Container(),
         )
       }
       
       // 推荐：合并图层
       CustomPaint(
         painter: BatchPainter(),
       )
    
    2. 绘制优化
       class OptimizedPainter extends CustomPainter {
         @override
         void paint(Canvas canvas, Size size) {
           // 批量绘制
           final paint = Paint();
           final path = Path();
           
           // 合并多个绘制操作
           for (int i = 0; i < 100; i++) {
             path.addRect(Rect.fromLTWH(i * 10.0, 0, 8, 8));
           }
           
           // 一次绘制
           canvas.drawPath(path, paint);
         }
         
         @override
         bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
       }
    
    3. 缓存复杂绘制
       // 使用 cache
       class CachedWidget extends StatelessWidget {
         @override
         Widget build(BuildContext context) {
           return RepaintBoundary(
             child: Container(
               // 复杂绘制
               child: CustomPaint(
                 painter: ComplexPainter(),
               ),
             ),
           );
         }
       }
    ''');
  }
}
```

```dart
// 帧率优化
class FrameRateOptimization {
  /*
  帧率优化：
  
  1. 60 FPS 目标
     - 每帧 16ms
     - 避免卡顿
     - 流畅动画
  
  2. 性能监控
     - Performance Overlay
     - 帧率统计
     - 识别瓶颈
  
  3. 优化策略
     - 减少重建
     - 优化布局
     - 优化绘制
  */
  
  void explain() {
    print('''
    帧率优化：
    
    1. 启用 Performance Overlay
       // 在 main() 中设置
       void main() {
         WidgetsFlutterBinding.ensureInitialized();
         
         // 启用性能叠加
         debugProfileBuildsEnabled = true;
         
         runApp(MyApp());
       }
       
       // 或使用 MaterialApp 的 showPerformanceOverlay
       MaterialApp(
         showPerformanceOverlay: true,
         home: MyHomePage(),
       )
    
    2. 帧率监控
       // 使用 SchedulerBinding
       import 'package:flutter/scheduler.dart';
       
       class FrameRateMonitor {
         int _frameCount = 0;
         DateTime _lastTime = DateTime.now();
         
         void startMonitoring() {
           SchedulerBinding.instance.addTimingsCallback((timings) {
             for (final timing in timings) {
               final duration = timing.totalSpan.inMilliseconds;
               if (duration > 16) {
                 print('慢帧: ${duration}ms');
               }
             }
           });
         }
       }
    
    3. 优化策略
       // 1. 使用 const
       const Text('静态文本');
       
       // 2. 提取 Widget
       Widget _buildExpensiveWidget() {
         return RepaintBoundary(
           child: ComplexWidget(),
         );
       }
       
       // 3. 使用 Builder
       Builder(
         builder: (context) {
           // 避免不必要的重建
           return Text('内容');
         },
       )
    ''');
  }
}
```

```dart
// 响应式布局
class ResponsiveLayout {
  /*
  响应式布局：
  
  1. MediaQuery
     - 获取屏幕尺寸
     - 获取方向
     - 获取安全区域
  
  2. LayoutBuilder
     - 获取约束
     - 响应式构建
  
  3. 自适应组件
     - Flexible
     - Expanded
     - FractionallySizedBox
  */
  
  void explain() {
    print('''
    响应式布局：
    
    1. 使用 MediaQuery
       class ResponsiveWidget extends StatelessWidget {
         @override
         Widget build(BuildContext context) {
           final mediaQuery = MediaQuery.of(context);
           final screenWidth = mediaQuery.size.width;
           final screenHeight = mediaQuery.size.height;
           final orientation = mediaQuery.orientation;
           
           if (screenWidth > 600) {
             return _buildWideLayout();
           } else {
             return _buildNarrowLayout();
           }
         }
       }
    
    2. 使用 LayoutBuilder
       LayoutBuilder(
         builder: (context, constraints) {
           if (constraints.maxWidth > 600) {
             return Row(
               children: [
                 Expanded(child: LeftPanel()),
                 Expanded(child: RightPanel()),
               ],
             );
           } else {
             return Column(
               children: [
                 LeftPanel(),
                 RightPanel(),
               ],
             );
           }
         },
       )
    
    3. 使用自适应组件
       // Flexible
       Row(
         children: [
           Flexible(
             flex: 2,
             child: Container(color: Colors.red),
           ),
           Flexible(
             flex: 1,
             child: Container(color: Colors.blue),
           ),
         ],
       )
       
       // FractionallySizedBox

         child: Container(color: Colors.green),
```

## 📖 总结

### 布局调试核心技巧

| 问题 | 调试方法 | 解决方案 |
|------|----------|----------|
| **溢出** | debugPaintSizeEnabled | Expanded, Flexible |
| **约束冲突** | LayoutBuilder | 合理设置约束 |
| **重绘** | debugRepaintRainbowEnabled | RepaintBoundary |
| **性能** | Performance Overlay | 优化绘制逻辑 |

### 绘制优化策略

1. **减少重绘**：使用 RepaintBoundary 隔离
2. **优化透明度**：减少 Opacity 使用
3. **批量绘制**：合并绘制操作
4. **使用缓存**：缓存复杂绘制结果
