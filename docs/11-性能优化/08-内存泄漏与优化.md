       class PaginatedList extends StatefulWidget {
         @override
         _PaginatedListState createState() => _PaginatedListState();
       }

         final List<Item> _items = [];
         final int _pageSize = 20;
         bool _isLoading = false;
         bool _hasMore = true;
         
         @override
         void initState() {
           super.initState();
           _loadMore();
         }
         
         Future<void> _loadMore() async {
           if (_isLoading || !_hasMore) return;
           
           setState(() => _isLoading = true);
           
           final newItems = await fetchItems(_items.length, _pageSize);
           
           setState(() {
             _items.addAll(newItems);
             _isLoading = false;
             _hasMore = newItems.length == _pageSize;
           });
         }
         
         @override
         Widget build(BuildContext context) {
           return ListView.builder(
             itemCount: _items.length + (_hasMore ? 1 : 0),
             itemBuilder: (context, index) {
               if (index == _items.length) {
                 _loadMore();
                 return CircularProgressIndicator();
               }
               return ListItem(item: _items[index]);
             },
           );
         }
       }
    ''');

class ObjectPoolPattern {
  /*
  对象池模式：
  
  1. 功能
     - 复用对象
     - 减少创建销毁
     - 控制内存占用
  
  2. 使用场景

     - 创建成本高的对象
     - 需要限制数量的对象
  
  3. 实现
     - 维护对象池
     - 获取和归还
     - 清理策略
  */

    print('''
    对象池模式：
    
    1. 基本实现
       class ObjectPool<T> {
         final List<T> _pool = [];
         final T Function() _factory;
         final int maxSize;
         
         ObjectPool({
           required T Function() factory,
           this.maxSize = 100,
         }) : _factory = factory;
         
         T acquire() {
           if (_pool.isNotEmpty) {
             return _pool.removeLast();
           }
           return _factory();
         }
         
         void release(T object) {
           if (_pool.length < maxSize) {
             _pool.add(object);
           }
         }
         
         void clear() {
           _pool.clear();
         }
       }
    
    2. Widget 对象池
       class WidgetPool {
         static final Map<Type, List<Widget>> _pools = {};
         static const int maxSize = 50;
         
         static T acquire<T extends Widget>(T Function() factory) {
           final pool = _pools[T] ?? [];
           
           if (pool.isNotEmpty) {
             return pool.removeLast() as T;
           }
           
           return factory();
         }
         
         static void release<T extends Widget>(T widget) {
           final pool = _pools.putIfAbsent(T, () => []);
           
           if (pool.length < maxSize) {
             pool.add(widget);
           }
         }
       }
    
    3. 使用示例
       class OptimizedList extends StatelessWidget {
         @override
         Widget build(BuildContext context) {
           return ListView.builder(
             itemBuilder: (context, index) {
               // 从池中获取
               return WidgetPool.acquire<ReusableItem>(
                 () => ReusableItem(),
               );
             },
           );
         }
       }
    ''');
  }
}

## 📖 总结

### 内存泄漏核心概念

| 概念 | 描述 | 影响 |
|------|------|------|
| **内存泄漏** | 分配的内存未释放 | 内存占用增长 |
| **GC** | 垃圾回收 | 自动回收内存 |
| **引用** | 对象被引用 | GC 无法回收 |
| **dispose** | 释放资源 | 手动管理资源 |

### 最佳实践

1. **及时释放**：在 dispose 中释放所有资源
2. **检查 mounted**：异步操作后检查 mounted
3. **使用清单**：建立资源管理清单
4. **定期检测**：使用 DevTools 检测泄漏
