### 2. 渲染性能分析

```dart
// 渲染性能分析
class RenderingPerformanceAnalysis {
  void explain() {
    print('''
    渲染性能分析：
    
    // 1. 使用 Performance Overlay
    MaterialApp(
      showPerformanceOverlay: true,
      home: MyHomePage(),
    )
    
    // 2. 使用 DevTools
    // - 打开 DevTools
    // - 切换到 Performance 面板
    // - 查看帧图表
    // - 分析渲染性能
    
    // 3. 使用 Flutter Inspector
    // - 查看 Widget 树
    // - 分析渲染结构
    // - 识别性能问题
    // - 优化渲染
    
    // 4. 自定义性能分析
    class RenderingPerformanceTracker {
      final List<RenderingTiming> _timings = [];
      
      void startTracking() {
        SchedulerBinding.instance.addTimingsCallback((timings) {
          for (final timing in timings) {
            _timings.add(RenderingTiming(
              buildDuration: timing.buildDuration,
              rasterDuration: timing.rasterDuration,
            ));
          }
        });
      }
      
      void printReport() {
        print('渲染性能报告:');
        for (final timing in _timings) {
          print('  build: ${timing.buildDuration.inMilliseconds}ms');
          print('  raster: ${timing.rasterDuration.inMilliseconds}ms');
        }
      }
    }
    ''');
  }
}
```

### 2. 优化绘制

```dart
// 优化绘制
class OptimizePainting {
  void explain() {
    print('''
    优化绘制：
    
    // 1. 为什么优化绘制
    // - 减少绘制次数
    // - 提高性能
    // - 减少内存使用
    // - 改善用户体验
    
    // 2. 优化绘制的方法
    // - 使用 CustomPaint
    // - 使用 Canvas
    // - 使用 Picture
    // - 使用 ImageFilter
    
    // 3. 优化绘制的示例
    CustomPaint(
      painter: MyPainter(),
      size: Size(200, 200),
    )
    
    class MyPainter extends CustomPainter {
      @override
      void paint(Canvas canvas, Size size) {
        final paint = Paint()
          ..color = Colors.blue
          ..style = PaintingStyle.fill;
        
        canvas.drawCircle(
          Offset(size.width / 2, size.height / 2),
          50,
          paint,
        );
      }
      
      @override
      bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
    }
    
    // 4. 优化绘制的最佳实践
    // - 使用 CustomPaint
    // - 使用 Canvas
    // - 使用 Picture
    // - 保持绘制简单
    ''');
  }
}
```
