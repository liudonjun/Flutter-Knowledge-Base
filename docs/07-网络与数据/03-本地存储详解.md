# 本地存储详解

> 掌握 Flutter 中的各种本地数据存储方式。

## 📖 SharedPreferences

### 1. 基本用法

```dart
// SharedPreferences 基本用法
import 'package:shared_preferences/shared_preferences.dart';

class SharedPreferencesBasics {
  void explain() {
    print('''
    SharedPreferences 基本用法：
    
    // 依赖
    // shared_preferences: ^2.0.0
    
    // 1. 保存数据
    Future<void> saveData() async {
      final prefs = await SharedPreferences.getInstance();
      await prefs.setString('name', 'Alice');
      await prefs.setInt('age', 25);
      await prefs.setBool('isLoggedIn', true);
      await prefs.setDouble('score', 95.5);
      await prefs.setStringList('hobbies', ['reading', 'swimming']);
    }
    
    // 2. 读取数据
    Future<void> readData() async {
      final prefs = await SharedPreferences.getInstance();
      final name = prefs.getString('name') ?? '';
      final age = prefs.getInt('age') ?? 0;
      final isLoggedIn = prefs.getBool('isLoggedIn') ?? false;
      final score = prefs.getDouble('score') ?? 0.0;
      final hobbies = prefs.getStringList('hobbies') ?? [];
    }
    
    // 3. 删除数据
    Future<void> removeData() async {
      final prefs = await SharedPreferences.getInstance();
      await prefs.remove('name');
      await prefs.clear();  // 清空所有
    }
    
    // 4. 检查键是否存在
    Future<bool> hasKey(String key) async {
      final prefs = await SharedPreferences.getInstance();
      return prefs.containsKey(key);
    }
    ''');
  }
}
```

### 2. 封装工具类

```dart
// SharedPreferences 工具类
class SpUtils {
  static SharedPreferences? _prefs;
  
  static Future<void> init() async {
    _prefs = await SharedPreferences.getInstance();
  }
  
  // 保存
  static Future<bool> setString(String key, String value) async {
    return await _prefs?.setString(key, value) ?? false;
  }
  
  static Future<bool> setInt(String key, int value) async {
    return await _prefs?.setInt(key, value) ?? false;
  }
  
  static Future<bool> setBool(String key, bool value) async {
    return await _prefs?.setBool(key, value) ?? false;
  }
  
  // 读取
  static String getString(String key, {String defaultValue = ''}) {
    return _prefs?.getString(key) ?? defaultValue;
  }
  
  static int getInt(String key, {int defaultValue = 0}) {
    return _prefs?.getInt(key) ?? defaultValue;
  }
  
  static bool getBool(String key, {bool defaultValue = false}) {
    return _prefs?.getBool(key) ?? defaultValue;
  }
  
  // 删除
  static Future<bool> remove(String key) async {
    return await _prefs?.remove(key) ?? false;
  }
  
  // 清空
  static Future<bool> clear() async {
    return await _prefs?.clear() ?? false;
  }
}
''');
  }
}
```

## 📖 文件存储

### 1. 基本文件操作

```dart
// 文件存储
import 'dart:io';
import 'package:path_provider/path_provider.dart';

class FileStorage {
  void explain() {
    print('''
    文件存储：
    
    // 依赖
    // path_provider: ^2.0.0
    
    // 1. 获取目录
    Future<void> getDirectories() async {
      final appDir = await getApplicationDocumentsDirectory();
      final tempDir = await getTemporaryDirectory();
      final externalDir = await getExternalStorageDirectory();
      
      print('应用目录: ${appDir.path}');
      print('临时目录: ${tempDir.path}');
    }
    
    // 2. 写入文件
    Future<void> writeFile() async {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}/data.txt');
      await file.writeAsString('Hello, Flutter!');
    }
    
    // 3. 读取文件
    Future<String> readFile() async {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}/data.txt');
      return await file.readAsString();
    }
    
    // 4. 检查文件是否存在
    Future<bool> fileExists() async {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}/data.txt');
      return await file.exists();
    }
    
    // 5. 删除文件
    Future<void> deleteFile() async {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}/data.txt');
      await file.delete();
    }
    ''');
  }
}
```

### 2. JSON 文件存储

```dart
// JSON 文件存储
class JsonFileStorage {
  Future<void> saveJson(String fileName, Map<String, dynamic> data) async {
    final dir = await getApplicationDocumentsDirectory();
    final file = File('${dir.path}/$fileName.json');
    await file.writeAsString(jsonEncode(data));
  }
  
  Future<Map<String, dynamic>?> loadJson(String fileName) async {
    try {
      final dir = await getApplicationDocumentsDirectory();
      final file = File('${dir.path}/$fileName.json');
      if (await file.exists()) {
        final content = await file.readAsString();
        return jsonDecode(content) as Map<String, dynamic>;
      }
    } catch (e) {
      print('读取失败: $e');
    }
    return null;
  }
  
  void explain() {
    print('''
    JSON 文件存储：
    
    // 保存
    await saveJson('user', {'name': 'Alice', 'age': 25});
    
    // 读取
    final data = await loadJson('user');
    if (data != null) {
      print(data['name']);
    }
    ''');
  }
}
```
