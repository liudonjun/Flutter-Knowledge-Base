### 2. 文件上传实现

```dart
// 文件上传实现
class FileUploadImplementation {
  void explain() {
    print('''
    文件上传实现：
    
    // 1. 使用 Dio 上传
    final dio = Dio();
    final formData = FormData.fromMap({
      'file': await MultipartFile.fromFile('path/to/file.jpg'),
    });
    final response = await dio.post('/upload', data: formData);
    
    // 2. 带进度的上传
    final response = await dio.post(
      '/upload',
      data: formData,
      onSendProgress: (sent, total) {
        final progress = sent / total;
        print('Upload progress: ${(progress * 100).toStringAsFixed(0)}%');
      },
    );
    
    // 3. 多文件上传
    final formData = FormData.fromMap({
      'files': [
        await MultipartFile.fromFile('file1.jpg'),
        await MultipartFile.fromFile('file2.jpg'),
      ],
    });
    final response = await dio.post('/upload', data: formData);
    
    // 4. 使用 http 包上传
    final request = http.MultipartRequest('POST', Uri.parse('/upload'));
    request.files.add(await http.MultipartFile.fromPath('file', 'path/to/file.jpg'));
    final response = await request.send();
    
    // 5. 文件上传最佳实践
    // - 错误处理
    // - 进度显示
    // - 文件验证
    // - 安全性
    ''');
  }
}
```
