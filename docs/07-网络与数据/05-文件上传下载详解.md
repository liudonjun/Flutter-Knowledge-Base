# 文件上传下载详解

> 掌握 Flutter 中的文件上传和下载实现。

## 📖 文件下载

### 1. 基本下载

```dart
// 文件下载
import 'dart:io';
import 'package:dio/dio.dart';
import 'package:path_provider/path_provider.dart';

class FileDownload {
  void explain() {
    print('''
    文件下载方式：
    
    // 1. 使用 Dio 下载
    Future<void> downloadFile(String url, String fileName) async {
      final dir = await getApplicationDocumentsDirectory();
      final savePath = '${dir.path}/$fileName';
      
      final dio = Dio();
      await dio.download(url, savePath);
      print('下载完成: $savePath');
    }
    
    // 2. 带进度的下载
    Future<void> downloadWithProgress(String url, String fileName) async {
      final dir = await getApplicationDocumentsDirectory();
      final savePath = '${dir.path}/$fileName';
      
      final dio = Dio();
      await dio.download(
        url,
        savePath,
        onReceiveProgress: (received, total) {
          if (total != -1) {
            final progress = (received / total * 100).toStringAsFixed(0);
            print('下载进度: $progress%');
          }
        },
      );
    }
    
    // 3. 取消下载
    CancelToken cancelToken = CancelToken();
    
    Future<void> downloadCancelable(String url, String savePath) async {
      try {
        await Dio().download(url, savePath, cancelToken: cancelToken);
      } catch (e) {
        if (CancelToken.isCancel(e)) {
          print('下载已取消');
        }
      }
    }
    
    // 取消
    cancelToken.cancel('用户取消');
    ''');
  }
}
```

## 📖 文件上传

### 1. 基本上传

```dart
// 文件上传
class FileUpload {
  void explain() {
    print('''
    文件上传方式：
    
    // 1. 单文件上传
    Future<void> uploadFile(String filePath) async {
      final dio = Dio();
      final formData = FormData.fromMap({
        'file': await MultipartFile.fromFile(filePath),
      });
      
      final response = await dio.post(
        'https://api.example.com/upload',
        data: formData,
        onSendProgress: (sent, total) {
          final progress = (sent / total * 100).toStringAsFixed(0);
          print('上传进度: $progress%');
        },
      );
      
      print('上传成功: ${response.data}');
    }
    
    // 2. 多文件上传
    Future<void> uploadMultipleFiles(List<String> filePaths) async {
      final dio = Dio();
      final formData = FormData();
      
      for (final path in filePaths) {
        formData.files.add(MapEntry(
          'files',
          await MultipartFile.fromFile(path),
        ));
      }
      
      final response = await dio.post(
        'https://api.example.com/upload',
        data: formData,
      );
    }
    
    // 3. 带参数的上传
    Future<void> uploadWithParams(String filePath) async {
      final dio = Dio();
      final formData = FormData.fromMap({
        'description': '文件描述',
        'category': 'images',
        'file': await MultipartFile.fromFile(filePath),
      });
      
      await dio.post('https://api.example.com/upload', data: formData);
    }
    ''');
  }
}
```

### 2. 图片上传

```dart
// 图片上传
class ImageUpload {
  void explain() {
    print('''
    图片上传：
    
    // 使用 image_picker
    // image_picker: ^1.0.0
    
    import 'package:image_picker/image_picker.dart';
    
    Future<void> pickAndUploadImage() async {
      final picker = ImagePicker();
      final image = await picker.pickImage(source: ImageSource.gallery);
      
      if (image != null) {
        // 压缩图片
        final compressed = await FlutterImageCompress.compressWithFile(
          image.path,
          quality: 85,
        );
        
        // 上传
        final dio = Dio();
        final formData = FormData.fromMap({
          'image': await MultipartFile.fromFile(
            image.path,
            filename: 'image.jpg',
          ),
        });
        
        await dio.post('https://api.example.com/upload', data: formData);
      }
    }
    
    // 选择多张图片
    Future<void> pickMultipleImages() async {
      final picker = ImagePicker();
      final images = await picker.pickMultiImage();
      
      final formData = FormData();
      for (final image in images) {
        formData.files.add(MapEntry(
          'images',
          await MultipartFile.fromFile(image.path),
        ));
      }
      
      await Dio().post('https://api.example.com/upload', data: formData);
    }
    ''');
  }
}
```
