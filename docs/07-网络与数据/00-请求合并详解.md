### 2. 请求合并实现

```dart
// 请求合并实现
class RequestBatchingImplementation {
  void explain() {
    print('''
    请求合并实现：
    
    // 1. 批量请求
    final responses = await Future.wait([
      dio.get('/users/1'),
      dio.get('/users/2'),
      dio.get('/users/3'),
    ]);
    
    // 2. 请求队列
    class RequestQueue {
      final List<Future<Response>> _queue = [];
      final int maxConcurrent;
      
      RequestQueue({this.maxConcurrent = 5});
      
      Future<Response> add(Future<Response> Function() request) async {
        while (_queue.length >= maxConcurrent) {
          await Future.any(_queue);
        }
        
        final future = request();
        _queue.add(future);
        future.then((_) => _queue.remove(future));
        return future;
      }
    }
    
    // 3. 请求聚合
    class RequestAggregator {
      final Map<String, List<Completer<Response>>> _pending = {};
      
      Future<Response> get(String url) {
        final completer = Completer<Response>();
        _pending[url] ??= [];
        _pending[url]!.add(completer);
        
        // 延迟发送请求
        Future.delayed(Duration(milliseconds: 10), () {
          if (_pending.containsKey(url)) {
            final completers = _pending.remove(url)!;
            dio.get(url).then((response) {
              for (final c in completers) {
                c.complete(response);
              }
            });
          }
        });
        
        return completer.future;
      }
    }
    ''');
  }
}
```
