### 2. 离线缓存实现

```dart
// 离线缓存实现
class OfflineCachingImplementation {
  void explain() {
    print('''
    离线缓存实现：
    
    // 1. 使用 Hive 缓存
    final box = await Hive.openBox('cache');
    
    // 存储数据
    await box.put('users', users.map((u) => u.toJson()).toList());
    
    // 读取数据
    final cachedUsers = box.get('users');
    
    // 2. 使用 SQLite 缓存
    final db = await openDatabase('cache.db');
    await db.insert('users', {'data': jsonEncode(users)});
    final cachedData = await db.query('users');
    
    // 3. 离线优先策略
    class OfflineFirstService {
      Future<List<User>> getUsers() async {
        try {
          // 尝试网络请求
          final users = await fetchUsersFromNetwork();
          // 缓存数据
          await cacheUsers(users);
          return users;
        } catch (e) {
          // 网络失败，使用缓存
          return await getCachedUsers();
        }
      }
    }
    
    // 4. 离线缓存最佳实践
    // - 缓存策略
    // - 数据同步
    // - 冲突处理
    // - 存储管理
    ''');
  }
}
```
