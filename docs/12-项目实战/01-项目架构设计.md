# 项目架构设计

> 掌握 Flutter 项目的架构设计原则和模式。

## 📖 架构模式

### 1. 分层架构

```dart
// 分层架构
class LayeredArchitecture {
  void explain() {
    print('''
    分层架构：
    
    lib/
      ├── data/              # 数据层
      │   ├── models/        # 数据模型
      │   ├── repositories/  # 数据仓库
      │   └── datasources/   # 数据源
      ├── domain/            # 领域层
      │   ├── entities/      # 实体
      │   ├── repositories/  # 仓库接口
      │   └── usecases/      # 用例
      ├── presentation/      # 表现层
      │   ├── screens/       # 页面
      │   ├── widgets/       # 组件
      │   └── viewmodels/    # 视图模型
      └── core/              # 核心层
          ├── utils/         # 工具类
          ├── constants/     # 常量
          └── errors/        # 错误处理
    
    // 依赖方向
    // Presentation → Domain ← Data
    // 依赖倒置原则
    ''');
  }
}
```

### 2. Clean Architecture

```dart
// Clean Architecture
class CleanArchitecture {
  void explain() {
    print('''
    Clean Architecture：
    
    lib/
      ├── core/              # 核心层
      │   ├── errors/        # 错误类
      │   ├── network/       # 网络
      │   └── usecases/      # 用例基类
      ├── features/          # 功能模块
      │   ├── auth/          # 认证模块
      │   │   ├── data/
      │   │   ├── domain/
      │   │   └── presentation/
      │   ├── home/          # 首页模块
      │   │   ├── data/
      │   │   ├── domain/
      │   │   └── presentation/
      │   └── profile/       # 个人资料
      │       ├── data/
      │       ├── domain/
      │       └── presentation/
      └── shared/            # 共享模块
          ├── widgets/
          ├── utils/
          └── constants/
    
    // 优点
    // 1. 关注点分离
    // 2. 易于测试
    // 3. 易于维护
    // 4. 可扩展性强
    ''');
  }
}
```

## 📖 模块化设计

### 1. 功能模块

```dart
// 功能模块
class FeatureModule {
  void explain() {
    print('''
    功能模块设计：
    
    // 模块结构
    features/
      ├── auth/
      │   ├── data/
      │   │   ├── models/
      │   │   ├── repositories/
      │   │   └── datasources/
      │   ├── domain/
      │   │   ├── entities/
      │   │   ├── repositories/
      │   │   └── usecases/
      │   └── presentation/
      │       ├── bloc/
      │       ├── screens/
      │       └── widgets/
      ├── home/
      │   └── ...
      └── profile/
          └── ...
    
    // 模块接口
    abstract class AuthModule {
      Future<User> login(String email, String password);
      Future<void> logout();
      bool get isLoggedIn;
    }
    
    // 模块实现
    class AuthModuleImpl implements AuthModule {
      final AuthRepository _repository;
      
      AuthModuleImpl(this._repository);
      
      @override
      Future<User> login(String email, String password) {
        return _repository.login(email, password);
      }
      
      @override
      Future<void> logout() {
        return _repository.logout();
      }
      
      @override
      bool get isLoggedIn => _repository.isLoggedIn;
    }
    ''');
  }
}
```
