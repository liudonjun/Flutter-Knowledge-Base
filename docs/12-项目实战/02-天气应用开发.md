# 天气应用开发

> 开发一个完整的天气应用，掌握 Flutter 实战技能。

## 🎯 项目目标

### 核心功能
- **当前位置天气**: 自动获取当前位置天气
- **城市搜索**: 搜索并添加城市
- **天气详情**: 温度、湿度、风速等详细信息
- **天气预报**: 未来几天天气预报
- **主题切换**: 浅色/深色主题

### 技术栈
- **状态管理**: Riverpod
- **网络请求**: Dio
- **地理位置**: Geolocator
- **本地存储**: SharedPreferences
- **UI 组件**: Flutter ScreenUtil

```dart
// 天气数据模型
class Weather {
  final String city;
  final double temperature;
  final double feelsLike;
  final int humidity;
  final double windSpeed;
  final String description;
  final String icon;
  final DateTime dateTime;
  
  Weather({
    required this.city,
    required this.temperature,
    required this.feelsLike,
    required this.humidity,
    required this.windSpeed,
    required this.description,
    required this.icon,
    required this.dateTime,
  });
  
  factory Weather.fromJson(Map<String, dynamic> json) {
    return Weather(
      city: json['name'],
      temperature: json['main']['temp'].toDouble(),
      feelsLike: json['main']['feels_like'].toDouble(),
      humidity: json['main']['humidity'],
      windSpeed: json['wind']['speed'].toDouble(),
      description: json['weather'][0]['description'],
      icon: json['weather'][0]['icon'],
      dateTime: DateTime.fromMillisecondsSinceEpoch(json['dt'] * 1000),
    );
  }
  
  Map<String, dynamic> toJson() {
    return {
      'name': city,
      'main': {
        'temp': temperature,
        'feels_like': feelsLike,
        'humidity': humidity,
      },
      'wind': {'speed': windSpeed},
      'weather': [
        {
          'description': description,
          'icon': icon,
        }
      ],
      'dt': dateTime.millisecondsSinceEpoch ~/ 1000,
    };
  }
}

// 天气预报模型
class Forecast {
  final List<ForecastItem> items;
  
  Forecast({required this.items});
  
  factory Forecast.fromJson(Map<String, dynamic> json) {
    final list = json['list'] as List;
    return Forecast(
      items: list.map((item) => ForecastItem.fromJson(item)).toList(),
    );
  }
}

class ForecastItem {
  final DateTime dateTime;
  final double temperature;
  final double tempMin;
  final double tempMax;
  final String description;
  final String icon;
  
  ForecastItem({
    required this.dateTime,
    required this.temperature,
    required this.tempMin,
    required this.tempMax,
    required this.description,
    required this.icon,
  });
  
  factory ForecastItem.fromJson(Map<String, dynamic> json) {
    return ForecastItem(
      dateTime: DateTime.fromMillisecondsSinceEpoch(json['dt'] * 1000),
      temperature: json['main']['temp'].toDouble(),
      tempMin: json['main']['temp_min'].toDouble(),
      tempMax: json['main']['temp_max'].toDouble(),
      description: json['weather'][0]['description'],
      icon: json['weather'][0]['icon'],
    );
  }
}
```

### 2. 天气服务
```dart
// 天气服务
class WeatherService {
  final ApiClient _apiClient;
  final String _apiKey;
  
  WeatherService(this._apiClient, this._apiKey);
  
  // 获取当前天气
  Future<Weather> getCurrentWeather(double lat, double lon) async {
    final response = await _apiClient.get(
      '/weather',
      queryParameters: {
        'lat': lat,
        'lon': lon,
        'appid': _apiKey,
        'units': 'metric',
        'lang': 'zh_cn',
      },
    );
    return Weather.fromJson(response.data);
  }
  
  // 根据城市名获取天气
  Future<Weather> getWeatherByCity(String city) async {
    final response = await _apiClient.get(
      '/weather',
      queryParameters: {
        'q': city,
        'appid': _apiKey,
        'units': 'metric',
        'lang': 'zh_cn',
      },
    );
    return Weather.fromJson(response.data);
  }
  
  // 获取天气预报
  Future<Forecast> getForecast(double lat, double lon) async {
    final response = await _apiClient.get(
      '/forecast',
      queryParameters: {
        'lat': lat,
        'lon': lon,
        'appid': _apiKey,
        'units': 'metric',
        'lang': 'zh_cn',
        'cnt': 40, // 5天预报，每3小时一次
      },
    );
    return Forecast.fromJson(response.data);
  }
  
  // 搜索城市
  Future<List<Location>> searchCity(String query) async {
    final response = await _apiClient.get(
      '/find',
      queryParameters: {
        'q': query,
        'appid': _apiKey,
        'units': 'metric',
        'lang': 'zh_cn',
        'type': 'like',
        'cnt': 10,
      },
    );
    
    final list = response.data['list'] as List;
    return list.map((item) => Location.fromJson(item)).toList();
  }
}
```

### 3. 位置服务
```dart
// 位置服务
class LocationService {
  final Geolocator _geolocator = Geolocator.instance;
  
  // 获取当前位置
  Future<Position> getCurrentPosition() async {
    bool serviceEnabled;
    LocationPermission permission;
    
    // 检查位置服务是否启用
    serviceEnabled = await _geolocator.isLocationServiceEnabled();
    if (!serviceEnabled) {
      throw Exception('位置服务未启用');
    }
    
    // 检查权限
    permission = await _geolocator.checkPermission();
    if (permission == LocationPermission.denied) {
      permission = await _geolocator.requestPermission();
      if (permission == LocationPermission.denied) {
        throw Exception('位置权限被拒绝');
      }
    }
    
    if (permission == LocationPermission.deniedForever) {
      throw Exception('位置权限被永久拒绝');
    }
    
    // 获取位置
    return await _geolocator.getCurrentPosition(
      desiredAccuracy: LocationAccuracy.high,
    );
  }
  
  // 获取最后已知位置
  Future<Position?> getLastKnownPosition() async {
    return await _geolocator.getLastKnownPosition();
  }
  
  // 监听位置变化
  Stream<Position> getPositionStream() {
    return _geolocator.getPositionStream(
      locationSettings: LocationSettings(
        accuracy: LocationAccuracy.high,
        distanceFilter: 100,
      ),
    );
  }
  
  // 计算距离
  double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
    return _geolocator.distanceBetween(lat1, lon1, lat2, lon2);
  }
}
```

```dart
class HomePage extends ConsumerWidget {

          SizedBox(height: 20),

          WeatherDetails(weather: weather),
          SizedBox(height: 20),
          
          // 天气预报
          ForecastWidget(),
        ],
      ),
    );
  }
}
```

```dart
// 天气状态管理
final currentWeatherProvider = FutureProvider<Weather>((ref) async {
  final locationService = ref.read(locationServiceProvider);
  final weatherService = ref.read(weatherServiceProvider);
  
  try {
    final position = await locationService.getCurrentPosition();
    return await weatherService.getCurrentWeather(
      position.latitude,
      position.longitude,
    );
  } catch (e) {
    // 如果位置获取失败，使用默认城市
    return await weatherService.getWeatherByCity('北京');
  }
});

// 天气预报状态管理
final forecastProvider = FutureProvider<Forecast>((ref) async {
  final locationService = ref.read(locationServiceProvider);
  final weatherService = ref.read(weatherServiceProvider);
  
  try {
    final position = await locationService.getCurrentPosition();
    return await weatherService.getForecast(
      position.latitude,
      position.longitude,
    );
  } catch (e) {
    // 使用默认城市
    final weather = await weatherService.getWeatherByCity('北京');
    return await weatherService.getForecast(39.9042, 116.4074);
  }
});

// 城市列表状态管理
final cityListProvider = StateNotifierProvider<CityListNotifier, List<String>>((ref) {
  return CityListNotifier(ref.read(storageServiceProvider));
});

class CityListNotifier extends StateNotifier<List<String>> {
  final StorageService _storageService;
  
  CityListNotifier(this._storageService) : super([]) {
    _loadCities();
  }
  
  Future<void> _loadCities() async {
    final cities = await _storageService.getStringList('cities') ?? [];
    state = cities;
  }
  
  Future<void> addCity(String city) async {
    if (!state.contains(city)) {
      final newCities = [...state, city];
      state = newCities;
      await _storageService.saveStringList('cities', newCities);
    }
  }
  
  Future<void> removeCity(String city) async {
    final newCities = state.where((c) => c != city).toList();
    state = newCities;
    await _storageService.saveStringList('cities', newCities);
  }
}
```
