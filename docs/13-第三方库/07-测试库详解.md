# 测试库详解

> 深入了解各种 Flutter 测试库，提高代码质量和开发效率。

## 📊 测试库对比

### 1. Flutter Test
```dart
// Flutter Test - 官方测试框架
// 优点：官方维护，功能完整，集成度高
// 缺点：功能相对基础

// 安装
dev_dependencies:
  flutter_test:
    sdk: flutter

// 基本使用
import 'package:flutter_test/flutter_test.dart';

void main() {
  // 单元测试
  test('should add two numbers', () {
    expect(add(2, 3), 5);
    expect(add(-1, 1), 0);
    expect(add(0, 0), 0);
  });
  
  // 分组测试
  group('Calculator', () {
    late Calculator calculator;
    
    setUp(() {
      calculator = Calculator();
    });
    
    test('should add', () {
      expect(calculator.add(2, 3), 5);
    });
    
    test('should subtract', () {
      expect(calculator.subtract(5, 3), 2);
    });
    
    test('should multiply', () {
      expect(calculator.multiply(2, 3), 6);
    });
    
    test('should divide', () {
      expect(calculator.divide(6, 3), 2.0);
    });
  });
}
```

### 2. Mockito
```dart
// Mockito - Mock 对象
// 优点：功能强大，灵活
// 缺点：需要代码生成

// 安装
dev_dependencies:
  mockito: ^5.4.4
  build_runner: ^2.4.0

// 基本使用
import 'package:mockito/mockito.dart';
import 'package:mockito/annotations.dart';

@GenerateMocks([UserRepository])
import 'user_service_test.mocks.dart';

void main() {
  group('UserService', () {
    late UserService userService;
    late MockUserRepository mockUserRepository;
    
    setUp(() {
      mockUserRepository = MockUserRepository();
      userService = UserService(mockUserRepository);
    });
    
    test('should get users', () async {
      // Arrange
      final mockUsers = [
        User(id: 1, name: 'John'),
        User(id: 2, name: 'Jane'),
      ];
      when(mockUserRepository.getUsers()).thenAnswer(
        (_) async => mockUsers,
      );
      
      // Act
      final users = await userService.getUsers();
      
      // Assert
      expect(users, isA<List<User>>());
      expect(users.length, 2);
      verify(mockUserRepository.getUsers()).called(1);
    });
    
    test('should handle error', () async {
      // Arrange
      when(mockUserRepository.getUsers()).thenThrow(
        Exception('Network error'),
      );
      
      // Act & Assert
      expect(
        () => userService.getUsers(),
        throwsA(isA<Exception>()),
      );
    });
  });
}
```

### 3. Bloc Test
```dart
// Bloc Test - BLoC 测试
// 优点：专门为 BLoC 设计，简单易用
// 缺点：仅适用于 BLoC

// 安装
dev_dependencies:
  bloc_test: ^9.1.5

// 基本使用
import 'package:flutter_test/flutter_test.dart';
import 'package:bloc_test/bloc_test.dart';

void main() {
  group('CounterBloc', () {
    late CounterBloc counterBloc;
    
    setUp(() {
      counterBloc = CounterBloc();
    });
    
    tearDown(() {
      counterBloc.close();
    });
    
    test('initial state is 0', () {
      expect(counterBloc.state, equals(0));
    });
    
    blocTest<CounterBloc, int>(
      'emits [1] when increment is added',
      build: () => counterBloc,
      act: (bloc) => bloc.add(IncrementEvent()),
      expect: () => [1],
    );
    
    blocTest<CounterBloc, int>(
      'emits [-1] when decrement is added',
      build: () => counterBloc,
      act: (bloc) => bloc.add(DecrementEvent()),
      expect: () => [-1],
    );
    
    blocTest<CounterBloc, int>(
      'emits [1, 2, 3] when increment is added three times',
      build: () => counterBloc,
      act: (bloc) {
        bloc.add(IncrementEvent());
        bloc.add(IncrementEvent());
        bloc.add(IncrementEvent());
      },
      expect: () => [1, 2, 3],
    );
  });
}
```

## 🔧 测试工具

### 1. Widget 测试
```dart
// Widget 测试
import 'package:flutter/material.dart';
import 'package:flutter_test/flutter_test.dart';

void main() {
  group('CounterWidget', () {
    testWidgets('should display initial count', (tester) async {
      await tester.pumpWidget(
        MaterialApp(home: CounterWidget()),
      );
      
      expect(find.text('Count: 0'), findsOneWidget);
      expect(find.byType(ElevatedButton), findsOneWidget);
    });
    
    testWidgets('should increment count when button pressed', (tester) async {
      await tester.pumpWidget(
        MaterialApp(home: CounterWidget()),
      );
      
      // 点击按钮
      await tester.tap(find.byType(ElevatedButton));
      await tester.pump();
      
      // 验证计数增加
      expect(find.text('Count: 1'), findsOneWidget);
    });
    
    testWidgets('should handle multiple taps', (tester) async {
      await tester.pumpWidget(
        MaterialApp(home: CounterWidget()),
      );
      
      // 多次点击
      for (int i = 0; i < 5; i++) {
        await tester.tap(find.byType(ElevatedButton));
        await tester.pump();
      }
      
      // 验证计数
      expect(find.text('Count: 5'), findsOneWidget);
    });
  });
}
```

### 2. 集成测试
```dart
// 集成测试
import 'package:integration_test/integration_test.dart';
import 'package:flutter_test/flutter_test.dart';
import 'package:my_app/main.dart' as app;

void main() {
  IntegrationTestWidgetsFlutterBinding.ensureInitialized();
  
  group('end-to-end test', () {
    testWidgets('should complete login flow', (tester) async {
      app.main();
      await tester.pumpAndSettle();
      
      // 验证登录页面
      expect(find.text('登录'), findsOneWidget);
      expect(find.byType(TextFormField), findsNWidgets(2));
      
      // 输入邮箱
      await tester.enterText(find.byKey(Key('email_field')), 'test@example.com');
      await tester.pumpAndSettle();
      
      // 输入密码
      await tester.enterText(find.byKey(Key('password_field')), '123456');
      await tester.pumpAndSettle();
      
      // 点击登录按钮
      await tester.tap(find.byKey(Key('login_button')));
      await tester.pumpAndSettle();
      
      // 验证登录成功
      expect(find.text('欢迎'), findsOneWidget);
    });
  });
}
```

## 🎯 测试模式

### 1. AAA 模式
```dart
// Arrange-Act-Assert 模式
void main() {
  group('UserRepository', () {
    late UserRepository userRepository;
    late MockApiClient mockApiClient;
    
    setUp(() {
      mockApiClient = MockApiClient();
      userRepository = UserRepositoryImpl(mockApiClient);
    });
    
    test('should get users successfully', () async {
      // Arrange - 准备测试数据
      final mockUsers = [
        User(id: 1, name: 'John'),
        User(id: 2, name: 'Jane'),
      ];
      when(mockApiClient.getUsers()).thenAnswer(
        (_) async => mockUsers,
      );
      
      // Act - 执行测试
      final users = await userRepository.getUsers();
      
      // Assert - 验证结果
      expect(users, isA<List<User>>());
      expect(users.length, 2);
      expect(users[0].name, 'John');
      verify(mockApiClient.getUsers()).called(1);
    });
    
    test('should handle network error', () async {
      // Arrange
      when(mockApiClient.getUsers()).thenThrow(
        DioException(requestOptions: RequestOptions(path: '/users')),
      );
      
      // Act & Assert
      expect(
        () => userRepository.getUsers(),
        throwsA(isA<NetworkException>()),
      );
    });
  });
}
```

### 2. Given-When-Then 模式
```dart
// Given-When-Then 模式 (BDD 风格)
void main() {
  group('ShoppingCart', () {
    late ShoppingCart cart;
    
    setUp(() {
      cart = ShoppingCart();
    });
    
    test('should calculate total price correctly', () {
      // Given - 给定条件
      final apple = Item(name: 'Apple', price: 1.0);
      final banana = Item(name: 'Banana', price: 0.5);
      
      // When - 当执行操作
      cart.addItem(apple);
      cart.addItem(banana);
      cart.addItem(apple);
      
      // Then - 那么验证结果
      expect(cart.totalPrice, 2.5);
      expect(cart.itemCount, 3);
    });
    
    test('should remove item correctly', () {
      // Given
      final item = Item(name: 'Apple', price: 1.0);
      cart.addItem(item);
      
      // When
      cart.removeItem(item);
      
      // Then
      expect(cart.itemCount, 0);
      expect(cart.totalPrice, 0);
    });
  });
}
```

## 🔍 测试工具类

### 1. 测试辅助类
```dart
// 测试辅助类
class TestUtils {
  // 创建测试 Widget
  static Widget createTestableWidget(Widget child) {
    return MaterialApp(
      home: Scaffold(body: child),
    );
  }
  
  // 创建带 Provider 的测试 Widget
  static Widget createTestableWidgetWithProvider(
    Widget child,
    List<Provider> providers,
  ) {
    return MultiProvider(
      providers: providers,
      child: MaterialApp(
        home: Scaffold(body: child),
      ),
    );
  }
  
  // 等待动画完成
  static Future<void> waitForAnimation(WidgetTester tester) async {
    await tester.pumpAndSettle();
  }
  
  // 模拟网络延迟
  static Future<void> simulateNetworkDelay() async {
    await Future.delayed(Duration(milliseconds: 500));
  }
  
  // 生成测试数据
  static List<User> generateTestUsers(int count) {
    return List.generate(count, (i) => User(
      id: i + 1,
      name: 'User ${i + 1}',
      email: 'user${i + 1}@example.com',
    ));
  }
}
```

### 2. Mock 数据生成
```dart
// Mock 数据生成器
class MockDataGenerator {
  static final Random _random = Random();
  
  // 生成随机字符串
  static String randomString([int length = 10]) {
    const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
    return String.fromCharCodes(Iterable.generate(
      length,
      (_) => chars.codeUnitAt(_random.nextInt(chars.length)),
    ));
  }
  
  // 生成随机邮箱
  static String randomEmail() {
    return '${randomString(5)}@example.com';
  }
  
  // 生成随机用户
  static User randomUser() {
    return User(
      id: _random.nextInt(1000),
      name: 'User ${randomString(5)}',
      email: randomEmail(),
    );
  }
  
  // 生成随机用户列表
  static List<User> randomUsers(int count) {
    return List.generate(count, (_) => randomUser());
  }
  
  // 生成随机日期
  static DateTime randomDate() {
    final start = DateTime(2020);
    final end = DateTime(2026);
    return start.add(Duration(days: _random.nextInt(end.difference(start).inDays)));
  }
}
```

### 1. 性能测试
```dart
// 性能测试
void main() {
  group('Performance', () {
    test('should process 1000 items in under 100ms', () {
      final stopwatch = Stopwatch()..start();
      
      // 执行操作
      processItems(List.generate(1000, (i) => i));
      
      stopwatch.stop();
      
      // 验证性能
      expect(stopwatch.elapsedMilliseconds, lessThan(100));
    });
    
    testWidgets('should scroll list smoothly', (tester) async {
      await tester.pumpWidget(
        MaterialApp(
          home: ListView.builder(
            itemCount: 1000,
            itemBuilder: (context, index) => ListTile(
              title: Text('Item $index'),
            ),
          ),
        ),
      );
      
      // 开始性能跟踪
      await tester.binding.traceAction(() async {
        // 滚动列表
        await tester.fling(find.byType(ListView), Offset(0, -500), 1000);
        await tester.pumpAndSettle();
      });
    });
  });
}
```
