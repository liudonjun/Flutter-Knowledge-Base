## 📊 数据库库对比

### 1. SQLite (sqflite)
```dart
// SQLite - 关系型数据库，轻量级
// 优点：轻量级，高性能，支持事务
// 缺点：需要手动管理表结构

// 安装
dependencies:
  sqflite: ^2.3.0
  path: ^1.8.3

// 基本使用
import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';

class DatabaseService {
  static Database? _database;
  
  static Future<Database> get database async {
    if (_database != null) return _database!;
    _database = await _initDatabase();
    return _database!;
  }
  
  static Future<Database> _initDatabase() async {
    String path = join(await getDatabasesPath(), 'app.db');
    return await openDatabase(
      path,
      version: 1,
      onCreate: (db, version) {
        // 创建表
        db.execute('''
          CREATE TABLE users(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL,
            email TEXT NOT NULL,
            phone TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
          )
        ''');
        
        db.execute('''
          CREATE TABLE posts(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id INTEGER,
            title TEXT NOT NULL,
            content TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (user_id) REFERENCES users (id)
          )
        ''');
      },
    );
  }
  
  // 插入数据
  static Future<int> insert(String table, Map<String, dynamic> data) async {
    final db = await database;
    return await db.insert(table, data);
  }
  
  // 查询数据
  static Future<List<Map<String, dynamic>>> query(
    String table, {
    String? where,
    List<dynamic>? whereArgs,
    String? orderBy,
    int? limit,
    int? offset,
  }) async {
    final db = await database;
    return await db.query(
      table,
      where: where,
      whereArgs: whereArgs,
      orderBy: orderBy,
      limit: limit,
      offset: offset,
    );
  }
  
  // 更新数据
  static Future<int> update(
    String table,
    Map<String, dynamic> data, {
    String? where,
    List<dynamic>? whereArgs,
  }) async {
    final db = await database;
    return await db.update(
      table,
      data,
      where: where,
      whereArgs: whereArgs,
    );
  }
  
  // 删除数据
  static Future<int> delete(
    String table, {
    String? where,
    List<dynamic>? whereArgs,
  }) async {
    final db = await database;
    return await db.delete(
      table,
      where: where,
      whereArgs: whereArgs,
    );
  }
  
  // 原始查询
  static Future<List<Map<String, dynamic>>> rawQuery(String sql, [List<dynamic>? arguments]) async {
    final db = await database;
    return await db.rawQuery(sql, arguments);
  }
  
  // 关闭数据库
  static Future<void> close() async {
    final db = await database;
    db.close();
  }
}
```

### 2. Hive
```dart
// Hive - NoSQL 数据库，高性能
// 优点：高性能，类型安全，支持加密
// 缺点：不支持复杂查询

// 安装
dependencies:
  hive: ^2.2.3
  hive_flutter: ^1.1.0

dev_dependencies:
  hive_generator: ^2.0.1
  build_runner: ^2.4.0

// 基本使用
import 'package:hive/hive.dart';
import 'package:hive_flutter/hive_flutter.dart';

// 数据模型
@HiveType(typeId: 0)
class User extends HiveObject {
  @HiveField(0)
  String name;
  
  @HiveField(1)
  String email;
  
  @HiveField(2)
  String phone;
  
  @HiveField(3)
  DateTime createdAt;
  
  User({
    required this.name,
    required this.email,
    this.phone = '',
    DateTime? createdAt,
  }) : createdAt = createdAt ?? DateTime.now();
}

// 初始化
void main() async {
  await Hive.initFlutter();
  Hive.registerAdapter(UserAdapter());
  await Hive.openBox<User>('users');
  runApp(MyApp());
}

// 使用示例
class HiveService {
  static Box<User> get userBox => Hive.box<User>('users');
  
  // 添加用户
  static Future<void> addUser(User user) async {
    await userBox.add(user);
  }
  
  // 获取所有用户
  static List<User> getAllUsers() {
    return userBox.values.toList();
  }
  
  // 根据索引获取用户
  static User? getUserAt(int index) {
    if (index < userBox.length) {
      return userBox.getAt(index);
    }
    return null;
  }
  
  // 更新用户
  static Future<void> updateUser(int index, User user) async {
    await userBox.putAt(index, user);
  }
  
  // 删除用户
  static Future<void> deleteUser(int index) async {
    await userBox.deleteAt(index);
  }
  
  // 清空所有用户
  static Future<void> clearAll() async {
    await userBox.clear();
  }
  
  // 监听变化
  static Stream<BoxEvent> watch() {
    return userBox.watch();
  }
}
```

### 3. Isar
```dart
// Isar - 高性能 NoSQL 数据库
// 优点：高性能，类型安全，支持索引
// 缺点：相对较新

// 安装
dependencies:
  isar: ^3.1.0
  isar_flutter_libs: ^3.1.0

dev_dependencies:
  isar_generator: ^3.1.0
  build_runner: ^2.4.0

// 基本使用
import 'package:isar/isar.dart';

part 'user.g.dart';

@collection
class User {
  Id id = Isar.autoIncrement;
  
  @Index()
  late String name;
  
  @Index()
  late String email;
  
  late String phone;
  
  late DateTime createdAt;
}

// 初始化
late Isar isar;

void main() async {
  await Isar.initialize();
  isar = await Isar.open([UserSchema]);
  runApp(MyApp());
}

// 使用示例
class IsarService {
  static Isar get isar => Isar.getInstance()!;
  
  // 添加用户
  static Future<int> addUser(User user) async {
    return await isar.writeTxn(() async {
      return await isar.users.put(user);
    });
  }
  
  // 获取所有用户
  static Future<List<User>> getAllUsers() async {
    return await isar.users.where().findAll();
  }
  
  // 根据 ID 获取用户
  static Future<User?> getUserById(int id) async {
    return await isar.users.get(id);
  }
  
  // 根据邮箱查询用户
  static Future<User?> getUserByEmail(String email) async {
    return await isar.users.filter().emailEqualTo(email).findFirst();
  }
  
  // 更新用户
  static Future<int> updateUser(User user) async {
    return await isar.writeTxn(() async {
      return await isar.users.put(user);
    });
  }
  
  // 删除用户
  static Future<bool> deleteUser(int id) async {
    return await isar.writeTxn(() async {
      return await isar.users.delete(id);
    });
  }
  
  // 清空所有用户
  static Future<int> clearAll() async {
    return await isar.writeTxn(() async {
      return await isar.users.clear();
    });
  }
}
```

## 🔧 高级功能

### 1. 数据库迁移
```dart
// SQLite 数据库迁移
class DatabaseMigration {
  static Future<void> migrate(Database db, int oldVersion, int newVersion) async {
    if (oldVersion < 2) {
      // 版本 1 -> 2: 添加新字段
      await db.execute('ALTER TABLE users ADD COLUMN avatar TEXT');
    }
    
    if (oldVersion < 3) {
      // 版本 2 -> 3: 创建新表
      await db.execute('''
        CREATE TABLE settings(
          id INTEGER PRIMARY KEY,
          key TEXT NOT NULL,
          value TEXT NOT NULL
        )
      ''');
    }
    
    if (oldVersion < 4) {
      // 版本 3 -> 4: 修改表结构
      await db.execute('ALTER TABLE users RENAME TO users_old');
      await db.execute('''
        CREATE TABLE users(
          id INTEGER PRIMARY KEY,
          name TEXT NOT NULL,
          email TEXT NOT NULL,
          phone TEXT,
          avatar TEXT,
          created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
      ''');
      await db.execute('''
        INSERT INTO users (id, name, email, phone, created_at)
        SELECT id, name, email, phone, created_at FROM users_old
      ''');
      await db.execute('DROP TABLE users_old');
    }
  }
}

// 使用迁移
static Future<Database> _initDatabase() async {
  String path = join(await getDatabasesPath(), 'app.db');
  return await openDatabase(
    path,
    version: 4,
    onCreate: (db, version) {
      // 创建表
    },
    onUpgrade: (db, oldVersion, newVersion) {
      DatabaseMigration.migrate(db, oldVersion, newVersion);
    },
  );
}
```

### 2. 事务处理
```dart
// SQLite 事务
static Future<void> batchInsert(List<User> users) async {
  final db = await database;
  final batch = db.batch();
  
  for (var user in users) {
    batch.insert('users', user.toJson());
  }
  
  await batch.commit();
}

// Hive 事务
static Future<void> batchUpdate(List<User> users) async {
  final box = Hive.box<User>('users');
  
  await box.putAll(
    Map.fromEntries(
      users.map((user) => MapEntry(user.id, user)),
    ),
  );
}

// Isar 事务
static Future<void> batchDelete(List<int> ids) async {
  await isar.writeTxn(() async {
    await isar.users.deleteAll(ids);
  });
}
```

## 🎯 最佳实践

### 1. 数据库服务封装
```dart
// 通用数据库服务
class DatabaseService<T> {
  final String tableName;
  final T Function(Map<String, dynamic>) fromJson;
  final Map<String, dynamic> Function(T) toJson;
  
  DatabaseService({
    required this.tableName,
    required this.fromJson,
    required this.toJson,
  });
  
  // 插入
  Future<int> insert(T item) async {
    final db = await DatabaseService.database;
    return await db.insert(tableName, toJson(item));
  }
  
  // 查询所有
  Future<List<T>> getAll() async {
    final db = await DatabaseService.database;
    final maps = await db.query(tableName);
    return maps.map((map) => fromJson(map)).toList();
  }
  
  // 根据 ID 查询
  Future<T?> getById(int id) async {
    final db = await DatabaseService.database;
    final maps = await db.query(
      tableName,
      where: 'id = ?',
      whereArgs: [id],
    );
    if (maps.isNotEmpty) {
      return fromJson(maps.first);
    }
    return null;
  }
  
  // 更新
  Future<int> update(T item, int id) async {
    final db = await DatabaseService.database;
    return await db.update(
      tableName,
      toJson(item),
      where: 'id = ?',
      whereArgs: [id],
    );
  }
  
  // 删除
  Future<int> delete(int id) async {
    final db = await DatabaseService.database;
    return await db.delete(
      tableName,
      where: 'id = ?',
      whereArgs: [id],
    );
  }
  
  // 清空
  Future<int> clear() async {
    final db = await DatabaseService.database;
    return await db.delete(tableName);
  }
}
```

### 2. Repository 模式
```dart
// Repository 接口
abstract class UserRepository {
  Future<List<User>> getUsers();
  Future<User?> getUser(int id);
  Future<int> addUser(User user);
  Future<int> updateUser(User user);
  Future<int> deleteUser(int id);
  Future<int> clearUsers();
}

// Repository 实现
class UserRepositoryImpl implements UserRepository {
  final DatabaseService<User> _databaseService;
  
  UserRepositoryImpl()
      : _databaseService = DatabaseService<User>(
          tableName: 'users',
          fromJson: (json) => User.fromJson(json),
          toJson: (user) => user.toJson(),
        );
  
  @override
  Future<List<User>> getUsers() async {
    return await _databaseService.getAll();
  }
  
  @override
  Future<User?> getUser(int id) async {
    return await _databaseService.getById(id);
  }
  
  @override
  Future<int> addUser(User user) async {
    return await _databaseService.insert(user);
  }
  
  @override
  Future<int> updateUser(User user) async {
    return await _databaseService.update(user, user.id!);
  }
  
  @override
  Future<int> deleteUser(int id) async {
    return await _databaseService.delete(id);
  }
  
  @override
  Future<int> clearUsers() async {
    return await _databaseService.clear();
  }
}
```

```dart
// SQLite 索引
static Future<void> createIndexes() async {
  final db = await database;
```

## 🔄 数据同步

### 1. 离线同步
```dart
class SyncService {
  static final Map<String, DateTime> _lastSyncTime = {};
  static const Duration _syncInterval = Duration(hours: 1);
  
  static bool shouldSync(String key) {
    final lastSync = _lastSyncTime[key];
    if (lastSync == null) return true;
    return DateTime.now().difference(lastSync) > _syncInterval;
  }
  
  static void markSynced(String key) {
    _lastSyncTime[key] = DateTime.now();
  }
  
  static Future<void> syncUsers() async {
    if (!shouldSync('users')) return;
    
    try {
      // 从服务器获取数据
      final response = await http.get(
        Uri.parse('https://api.example.com/users'),
      );
      
      if (response.statusCode == 200) {
        List<dynamic> data = jsonDecode(response.body);
        List<User> users = data.map((json) => User.fromJson(json)).toList();
        
        // 保存到本地数据库
        for (var user in users) {
          await DatabaseService.insert('users', user.toJson());
        }
        
        markSynced('users');
      }
    } catch (e) {
      print('同步失败: $e');
    }
  }
}
```
