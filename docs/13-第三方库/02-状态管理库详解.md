# 状态管理库详解

> 深入了解各种 Flutter 状态管理库，选择最适合你项目的方案。

## 📊 状态管理库对比

### 1. Provider
```dart
// Provider - 简单易用，官方推荐
// 优点：简单易学，性能好，社区活跃
// 缺点：大型项目可能不够用

// 安装
dependencies:
  provider: ^6.1.1

// 基本使用
class CounterProvider with ChangeNotifier {
  int _count = 0;
  int get count => _count;
  
  void increment() {
    _count++;
    notifyListeners();
  }
}

// 在 main.dart 中配置
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (context) => CounterProvider(),
      child: MyApp(),
    ),
  );
}

// 在 Widget 中使用
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return Consumer<CounterProvider>(
      builder: (context, counter, child) {
        return Column(
          children: [
            Text('计数: ${counter.count}'),
            ElevatedButton(
              onPressed: counter.increment,
              child: Text('增加'),
            ),
          ],
        );
      },
    );
  }
}
```

### 2. Riverpod
```dart
// Riverpod - 类型安全，编译时检查
// 优点：类型安全，更好的性能，更灵活
// 缺点：学习曲线陡峭

// 安装
dependencies:
  flutter_riverpod: ^2.4.9

// 基本使用
final counterProvider = StateNotifierProvider<CounterNotifier, int>((ref) {
  return CounterNotifier();
});

class CounterNotifier extends StateNotifier<int> {
  CounterNotifier() : super(0);
  
  void increment() => state++;
  void decrement() => state--;
}

// 在 main.dart 中配置
void main() {
  runApp(
    ProviderScope(
      child: MyApp(),
    ),
  );
}

// 在 Widget 中使用
class CounterWidget extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final count = ref.watch(counterProvider);
    return Column(
      children: [
        Text('计数: $count'),
        ElevatedButton(
          onPressed: () => ref.read(counterProvider.notifier).increment(),
          child: Text('增加'),
        ),
      ],
    );
  }
}
```

### 3. BLoC
```dart
// BLoC - 企业级，清晰架构
// 优点：清晰架构，易于测试，可预测
// 缺点：代码量大，学习成本高

// 安装
dependencies:
  flutter_bloc: ^8.1.3
  equatable: ^2.0.5

// 基本使用
// Event
abstract class CounterEvent extends Equatable {
  const CounterEvent();
  
  @override
  List<Object> get props => [];
}

class IncrementEvent extends CounterEvent {}
class DecrementEvent extends CounterEvent {}

// State
class CounterState extends Equatable {
  final int count;
  
  const CounterState({this.count = 0});
  
  CounterState copyWith({int? count}) {
    return CounterState(count: count ?? this.count);
  }
  
  @override
  List<Object> get props => [count];
}

// BLoC
class CounterBloc extends Bloc<CounterEvent, CounterState> {
  CounterBloc() : super(const CounterState()) {
    on<IncrementEvent>((event, emit) {
      emit(state.copyWith(count: state.count + 1));
    });
    
    on<DecrementEvent>((event, emit) {
      emit(state.copyWith(count: state.count - 1));
    });
  }
}

// 在 main.dart 中配置
void main() {
  runApp(
    BlocProvider(
      create: (context) => CounterBloc(),
      child: MyApp(),
    ),
  );
}

// 在 Widget 中使用
class CounterWidget extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return BlocBuilder<CounterBloc, CounterState>(
      builder: (context, state) {
        return Column(
          children: [
            Text('计数: ${state.count}'),
            ElevatedButton(
              onPressed: () => context.read<CounterBloc>().add(IncrementEvent()),
              child: Text('增加'),
            ),
          ],
        );
      },
    );
  }
}
```

### 4. GetX
```dart
// GetX - 功能丰富，简单易用
// 优点：功能丰富，简单易用，性能好
// 缺点：社区相对较小

// 安装
dependencies:
  get: ^4.6.6

// 基本使用
class CounterController extends GetxController {
  var count = 0.obs;
  
  void increment() => count++;
  void decrement() => count--;
}

// 在 main.dart 中配置
void main() {
  runApp(GetMaterialApp(home: Home()));
}

// 在 Widget 中使用
class CounterWidget extends StatelessWidget {
  final CounterController controller = Get.put(CounterController());
  
  @override
  Widget build(BuildContext context) {
    return Column(
      children: [
        Obx(() => Text('计数: ${controller.count}')),
        ElevatedButton(
          onPressed: controller.increment,
          child: Text('增加'),
        ),
      ],
    );
  }
}
```

## 🔧 高级用法

### 1. Provider 高级用法
```dart
// 多个 Provider
MultiProvider(
  providers: [
    ChangeNotifierProvider(create: (context) => UserProvider()),
    ChangeNotifierProvider(create: (context) => ThemeProvider()),
    ChangeNotifierProvider(create: (context) => CartProvider()),
  ],
  child: MyApp(),
)

// ProxyProvider - 依赖其他 Provider
ProxyProvider<UserProvider, AuthProvider>(
  update: (context, userProvider, previous) => AuthProvider(userProvider),
  child: MyApp(),
)

// Selector - 优化性能
Selector<CounterProvider, int>(
  selector: (context, provider) => provider.count,
  builder: (context, count, child) {
    return Text('计数: $count');
  },
)
```

### 2. Riverpod 高级用法
```dart
// FutureProvider - 异步数据
final userProvider = FutureProvider<User>((ref) async {
  final response = await http.get(Uri.parse('https://api.example.com/user'));
  return User.fromJson(jsonDecode(response.body));
});

// StreamProvider - 流数据
final timerProvider = StreamProvider<int>((ref) {
  return Stream.periodic(Duration(seconds: 1), (count) => count);
});

// StateProvider - 简单状态
final counterProvider = StateProvider<int>((ref) => 0);

// NotifierProvider - 复杂状态
class TodosNotifier extends Notifier<List<Todo>> {
  @override
  List<Todo> build() => [];
  
  void add(Todo todo) => state = [...state, todo];
  void remove(String id) => state = state.where((t) => t.id != id).toList();
}

final todosProvider = NotifierProvider<TodosNotifier, List<Todo>>(TodosNotifier.new);
```

### 3. BLoC 高级用法
```dart
// BlocListener - 监听状态变化
BlocListener<CounterBloc, CounterState>(
  listener: (context, state) {
    if (state.count == 10) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('计数达到 10!')),
      );
    }
  },
  child: CounterWidget(),
)

// BlocConsumer - 监听和构建
BlocConsumer<CounterBloc, CounterState>(
  listener: (context, state) {
    if (state.count == 10) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('计数达到 10!')),
      );
    }
  },
  builder: (context, state) {
    return Text('计数: ${state.count}');
  },
)

// BlocBuilder - 条件构建
BlocBuilder<CounterBloc, CounterState>(
  buildWhen: (previous, current) => previous.count != current.count,
  builder: (context, state) {
    return Text('计数: ${state.count}');
  },
)
```

## 🎯 选择指南

### 按项目规模选择
| 项目规模 | 推荐方案 | 理由 |
|---------|---------|------|
| 小型项目 | Provider | 简单易用，快速开发 |
| 中型项目 | Riverpod | 类型安全，性能好 |
| 大型项目 | BLoC | 架构清晰，易于测试 |
| 快速原型 | GetX | 功能丰富，开发快速 |

### 按团队经验选择
| 团队经验 | 推荐方案 | 理由 |
|---------|---------|------|
| 新手团队 | Provider | 学习曲线平缓 |
| 有经验团队 | Riverpod | 更好的类型安全 |
| 专家团队 | BLoC | 架构清晰，易于维护 |

### 按功能需求选择
| 功能需求 | 推荐方案 | 理由 |
|---------|---------|------|
| 简单状态 | Provider | 简单易用 |
| 复杂状态 | Riverpod | 类型安全 |
| 企业级 | BLoC | 架构清晰 |
| 快速开发 | GetX | 功能丰富 |

## 🚀 最佳实践

### 1. Provider 最佳实践
```dart
// 使用 Consumer 优化性能
Consumer<CounterProvider>(
  builder: (context, counter, child) {
    // 只有当 counter 变化时才重建
    return Text('计数: ${counter.count}');
  },
)

// 使用 Selector 选择性重建
Selector<CounterProvider, int>(
  selector: (context, provider) => provider.count,
  builder: (context, count, child) {
    return Text('计数: $count');
  },
)

// 避免在 build 中创建 Provider
// ❌ 不好的实践
Widget build(BuildContext context) {
  return ChangeNotifierProvider(
    create: (context) => CounterProvider(), // 每次重建都创建新实例
    child: CounterWidget(),
  );
}

// ✅ 好的实践
void main() {
  runApp(
    ChangeNotifierProvider(
      create: (context) => CounterProvider(), // 只创建一次
      child: MyApp(),
    ),
  );
}
```

### 2. Riverpod 最佳实践
```dart
// 使用 ref.watch 监听变化
final count = ref.watch(counterProvider);

// 使用 ref.read 读取一次
final notifier = ref.read(counterProvider.notifier);

// 使用 ref.listen 监听变化
ref.listen(counterProvider, (previous, current) {
  if (current == 10) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(content: Text('计数达到 10!')),
    );
  }
});

// 使用 family 传递参数
final userProvider = FutureProvider.family<User, String>((ref, userId) async {
  final response = await http.get(
    Uri.parse('https://api.example.com/users/$userId'),
  );
  return User.fromJson(jsonDecode(response.body));
});
```

### 3. BLoC 最佳实践
```dart
// 使用 Equatable 优化性能
class CounterState extends Equatable {
  final int count;
  
  const CounterState({this.count = 0});
  
  @override
  List<Object> get props => [count];
}

// 使用 copyWith 创建新状态
CounterState copyWith({int? count}) {
  return CounterState(count: count ?? this.count);
}

// 使用 BlocObserver 监控状态变化
class AppBlocObserver extends BlocObserver {
  @override
  void onChange(BlocBase bloc, Change change) {
    super.onChange(bloc, change);
    print('${bloc.runtimeType} $change');
  }
  
  @override
  void onError(BlocBase bloc, Object error, StackTrace stackTrace) {
    print('${bloc.runtimeType} $error');
    super.onError(bloc, error, stackTrace);
  }
}

void main() {
  Bloc.observer = AppBlocObserver();
  runApp(MyApp());
}
```

## 🔧 性能优化

### 1. 减少重建
```dart
// Provider - 使用 Selector
Selector<CounterProvider, int>(
  selector: (context, provider) => provider.count,
  builder: (context, count, child) {
    return Text('计数: $count');
  },
)

// Riverpod - 使用 select
final count = ref.watch(counterProvider.select((state) => state.count));

// BLoC - 使用 buildWhen
BlocBuilder<CounterBloc, CounterState>(
  buildWhen: (previous, current) => previous.count != current.count,
  builder: (context, state) {
    return Text('计数: ${state.count}');
  },
)
```

### 2. 缓存优化
```dart
// Provider - 使用 lazy
ChangeNotifierProvider(
  lazy: true, // 延迟创建
  create: (context) => CounterProvider(),
  child: MyApp(),
)

// Riverpod - 使用 autoDispose
final counterProvider = StateNotifierProvider.autoDispose<CounterNotifier, int>((ref) {
  return CounterNotifier();
});

// BLoC - 使用 close
@override
Future<void> close() {
  _subscription?.cancel();
  return super.close();
}
```
