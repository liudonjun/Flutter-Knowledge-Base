```dart
class BasicCanvasExample extends StatelessWidget {

      ..strokeWidth = 5;

      Rect.fromLTWH(50, 50, 100, 100),
      paint,

    canvas.drawCircle(
      Offset(200, 100),
      50,
      paint,

    paint.color = Colors.green;
    paint.style = PaintingStyle.stroke;
    canvas.drawLine(
      Offset(50, 200),
      Offset(250, 200),
      paint,
    );
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
```

```dart
      ..style = PaintingStyle.stroke
      ..strokeWidth = 5;
    
    // 创建路径
    final path = Path();
    
    // 移动到起点
    path.moveTo(50, 50);
    
    // 直线到
    path.lineTo(150, 50);
    path.lineTo(150, 150);
    path.lineTo(50, 150);
    
    // 闭合路径
    path.close();
    
    // 绘制路径
    canvas.drawPath(path, paint);
    
    // 绘制圆形路径
    final circlePath = Path();
    circlePath.addOval(Rect.fromCircle(center: Offset(200, 100), radius: 50));
    
    paint.color = Colors.red;
    canvas.drawPath(circlePath, paint);
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
```

```dart
// 图像绘制基础
class ImageDrawingExample extends StatefulWidget {
  @override
  _ImageDrawingExampleState createState() => _ImageDrawingExampleState();
}

class _ImageDrawingExampleState extends State<ImageDrawingExample> {
  Image? _image;
  
  @override
  void initState() {
    super.initState();
    _loadImage();
  }
  
  Future<void> _loadImage() async {
    final image = await _loadImageFromNetwork('https://picsum.photos/200/200');
    setState(() {
      _image = image;
    });
  }
  
  Future<Image> _loadImageFromNetwork(String url) async {
    final completer = Completer<Image>();
    final imageStream = NetworkImage(url).resolve(ImageConfiguration());
    
    imageStream.addListener(ImageStreamListener(
      (imageInfo, _) {
        completer.complete(imageInfo.image);
      },
      onError: (error, stackTrace) {
        completer.completeError(error);
      },
    ));
    
    return completer.future;
  }
  
  @override
  Widget build(BuildContext context) {
    return CustomPaint(
      painter: ImageDrawingPainter(_image),
      size: Size(300, 300),
    );
  }
}

class ImageDrawingPainter extends CustomPainter {
  final Image? image;
  
  ImageDrawingPainter(this.image);
  
  @override
  void paint(Canvas canvas, Size size) {
    if (image == null) return;
    
    // 绘制图像
    canvas.drawImage(
      image!,
      Offset(50, 50),
      Paint(),
    );
    
    // 绘制图像的一部分
    canvas.drawImageRect(
      image!,
      Rect.fromLTWH(0, 0, 100, 100), // 源矩形
      Rect.fromLTWH(150, 50, 100, 100), // 目标矩形
      Paint(),
    );
    
    // 绘制圆角图像
    final rrect = RRect.fromRectAndRadius(
      Rect.fromLTWH(50, 150, 100, 100),
      Radius.circular(20),
    );
    
    canvas.save();
    canvas.clipRRect(rrect);
    canvas.drawImageRect(
      image!,
      Rect.fromLTWH(0, 0, image!.width.toDouble(), image!.height.toDouble()),
      Rect.fromLTWH(50, 150, 100, 100),
      Paint(),
    );
    canvas.restore();
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => true;
}
```

```dart
// 绘制动画
class DrawingAnimationExample extends StatefulWidget {
  @override
  _DrawingAnimationExampleState createState() => _DrawingAnimationExampleState();
}

class _DrawingAnimationExampleState extends State<DrawingAnimationExample>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<double> _animation;
  
  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      duration: Duration(seconds: 2),
      vsync: this,
    );
    
    _animation = Tween<double>(begin: 0, end: 1).animate(
      CurvedAnimation(parent: _controller, curve: Curves.easeInOut),
    );
    
    _controller.repeat(reverse: true);
  }
  
  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }
  
  @override
  Widget build(BuildContext context) {
    return AnimatedBuilder(
      animation: _animation,
      builder: (context, child) {
        return CustomPaint(
          painter: DrawingAnimationPainter(_animation.value),
          size: Size(300, 300),
        );
      },
    );
  }
}

class DrawingAnimationPainter extends CustomPainter {
  final double progress;
  
  DrawingAnimationPainter(this.progress);
  
  @override
  void paint(Canvas canvas, Size size) {
    final paint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.stroke
      ..strokeWidth = 5;
    
    // 绘制动画路径
    final path = Path();
    path.moveTo(50, 150);
    
    // 根据进度绘制路径
    final controlPoint = Offset(150, 50 + 200 * progress);
    final endPoint = Offset(250, 150);
    
    path.quadraticBezierTo(
      controlPoint.dx,
      controlPoint.dy,
      endPoint.dx,
      endPoint.dy,
    );
    
    canvas.drawPath(path, paint);
    
    // 绘制控制点
    paint.color = Colors.red;
    paint.style = PaintingStyle.fill;
    canvas.drawCircle(controlPoint, 10, paint);
    
    // 绘制起点和终点
    paint.color = Colors.green;
    canvas.drawCircle(Offset(50, 150), 10, paint);
    canvas.drawCircle(Offset(250, 150), 10, paint);
  }
  
  @override
  bool shouldRepaint(covariant DrawingAnimationPainter oldDelegate) {
    return oldDelegate.progress != progress;
  }
}
```

```dart
// 粒子动画
class ParticleAnimationExample extends StatefulWidget {
  @override
  _ParticleAnimationExampleState createState() => _ParticleAnimationExampleState();
}

class _ParticleAnimationExampleState extends State<ParticleAnimationExample>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  final List<Particle> _particles = [];
  
  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      duration: Duration(seconds: 10),
      vsync: this,
    );
    
    _controller.addListener(_updateParticles);
    _controller.repeat();
  }
  
  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }
  
  void _updateParticles() {
    setState(() {
      // 添加新粒子
      if (_particles.length < 50) {
        _particles.add(Particle(
          position: Offset(150, 150),
          velocity: Offset(
            (Random().nextDouble() - 0.5) * 10,
            (Random().nextDouble() - 0.5) * 10,
          ),
          color: Colors.primaries[Random().nextInt(Colors.primaries.length)],
          size: Random().nextDouble() * 10 + 5,
        ));
      }
      
      // 更新粒子
      for (final particle in _particles) {
        particle.update();
      }
      
      // 移除超出边界的粒子
      _particles.removeWhere((particle) {
        return particle.position.dx < 0 ||
            particle.position.dx > 300 ||
            particle.position.dy < 0 ||
            particle.position.dy > 300;
      });
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return CustomPaint(
      painter: ParticleAnimationPainter(_particles),
      size: Size(300, 300),
    );
  }
}

class Particle {
  Offset position;
  final Offset velocity;
  final Color color;
  final double size;
  
  Particle({
    required this.position,
    required this.velocity,
    required this.color,
    required this.size,
  });
  
  void update() {
    position = position + velocity;
  }
}

class ParticleAnimationPainter extends CustomPainter {
  final List<Particle> particles;
  
  ParticleAnimationPainter(this.particles);
  
  @override
  void paint(Canvas canvas, Size size) {
    for (final particle in particles) {
      final paint = Paint()
        ..color = particle.color
        ..style = PaintingStyle.fill;
      
      canvas.drawCircle(
        particle.position,
        particle.size,
        paint,
      );
    }
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => true;
}
```

```dart
class OptimizedDrawingPainter extends CustomPainter {
  @override
  void paint(Canvas canvas, Size size) {
    // 使用缓存
    final paint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.fill;
    
    // 批量绘制
    final List<Rect> rects = [];
    for (int i = 0; i < 100; i++) {
      rects.add(Rect.fromLTWH(
        (i % 10) * 30.0,
        (i ~/ 10) * 30.0,
        20,
        20,
      ));
    }
    
    // 使用 drawDRRect 批量绘制
    final rrects = rects.map((rect) {
      return RRect.fromRectAndRadius(rect, Radius.circular(5));
    }).toList();
    
    canvas.drawDRRect(
      RRect.fromRectAndRadius(
        Rect.fromLTWH(0, 0, size.width, size.height),
        Radius.circular(10),
      ),
      RRect.fromRectAndRadius(
        Rect.fromLTWH(5, 5, size.width - 10, size.height - 10),
        Radius.circular(8),
      ),
      paint,
    );
    
    // 使用离屏渲染
    final pictureRecorder = PictureRecorder();
    final offscreenCanvas = Canvas(pictureRecorder);
    
    // 在离屏画布上绘制
    offscreenCanvas.drawRect(
      Rect.fromLTWH(0, 0, 100, 100),
      paint,
    );
    
    // 将离屏画布绘制到主画布
    final picture = pictureRecorder.endRecording();
    canvas.drawPicture(picture);
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
```

```dart
class CachedDrawingExample extends StatefulWidget {
  @override
  _CachedDrawingExampleState createState() => _CachedDrawingExampleState();
}

class _CachedDrawingExampleState extends State<CachedDrawingExample> {
  Picture? _cachedPicture;
  
  @override
  void initState() {
    super.initState();
    _createCachedPicture();
  }
  
  void _createCachedPicture() {
    final pictureRecorder = PictureRecorder();
    final canvas = Canvas(pictureRecorder);
    
    // 绘制复杂内容
    final paint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.fill;
    
    for (int i = 0; i < 1000; i++) {
      canvas.drawCircle(
        Offset(
          (i % 30) * 10.0,
          (i ~/ 30) * 10.0,
        ),
        5,
        paint,
      );
    }
    
    _cachedPicture = pictureRecorder.endRecording();
  }
  
  @override
  Widget build(BuildContext context) {
    return CustomPaint(
      painter: CachedDrawingPainter(_cachedPicture),
      size: Size(300, 300),
    );
  }
}

class CachedDrawingPainter extends CustomPainter {
  final Picture? cachedPicture;
  
  CachedDrawingPainter(this.cachedPicture);
  
  @override
  void paint(Canvas canvas, Size size) {
    if (cachedPicture != null) {
      canvas.drawPicture(cachedPicture!);
    }
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
}
```
