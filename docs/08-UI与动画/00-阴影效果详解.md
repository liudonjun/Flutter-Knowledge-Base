### 2. 阴影效果实现

```dart
// 阴影效果实现
class ShadowEffectsImplementation {
  void explain() {
    print('''
    阴影效果实现：
    
    // 1. 基本阴影
    Container(
      decoration: BoxDecoration(
        color: Colors.white,
        boxShadow: [
          BoxShadow(
            color: Colors.black12,
            blurRadius: 4,
            offset: Offset(0, 2),
          ),
        ],
      ),
      child: Text('Hello'),
    )
    
    // 2. 多重阴影
    Container(
      decoration: BoxDecoration(
        color: Colors.white,
        boxShadow: [
          BoxShadow(
            color: Colors.black12,
            blurRadius: 4,
            offset: Offset(0, 2),
          ),
          BoxShadow(
            color: Colors.black12,
            blurRadius: 8,
            offset: Offset(0, 4),
          ),
        ],
      ),
      child: Text('Hello'),
    )
    
    // 3. 自定义阴影
    Container(
      decoration: BoxDecoration(
        color: Colors.white,
        boxShadow: [
          BoxShadow(
            color: Colors.blue.withOpacity(0.3),
            blurRadius: 10,
            spreadRadius: 2,
            offset: Offset(0, 5),
          ),
        ],
      ),
      child: Text('Hello'),
    )
    
    // 4. 阴影效果最佳实践
    // - 使用适当阴影
    // - 考虑性能
    // - 优化阴影效果
    // - 测试不同设备
    ''');
  }
}
```
