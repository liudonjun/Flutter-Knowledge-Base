### 2. 路径绘制实现

```dart
// 路径绘制实现
class PathDrawingImplementation {
  void explain() {
    print('''
    路径绘制实现：
    
    // 1. 创建路径
    final path = Path();
    
    // 2. 绘制直线
    path.moveTo(0, 0);
    path.lineTo(100, 100);
    path.lineTo(200, 0);
    path.close();
    
    // 3. 绘制曲线
    path.moveTo(0, 0);
    path.quadraticBezierTo(100, 100, 200, 0);
    
    // 4. 绘制圆弧
    path.addArc(
      Rect.fromCircle(center: Offset(100, 100), radius: 50),
      0,
      3.14159,
    );
    
    // 5. 绘制圆形
    path.addOval(Rect.fromCircle(center: Offset(100, 100), radius: 50));
    
    // 6. 绘制矩形
    path.addRect(Rect.fromLTWH(0, 0, 100, 100));
    
    // 7. 组合路径
    final path1 = Path();
    final path2 = Path();
    final combinedPath = Path.combine(PathOperation.union, path1, path2);
    
    // 8. 绘制路径
    canvas.drawPath(path, paint);

    // - 使用适当方法
    // - 考虑性能
    // - 优化路径复杂度
    // - 测试绘制效果
    ''');
  }
}
```
