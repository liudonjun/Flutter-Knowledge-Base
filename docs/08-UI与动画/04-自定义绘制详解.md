# 自定义绘制详解

> 掌握 Flutter 中的 CustomPainter 自定义绘制。

## 📖 CustomPainter 基础

### 1. 基本用法

```dart
// CustomPainter 基本用法
class CustomPainterBasics {
  void explain() {
    print('''
    CustomPainter 基本用法：
    
    class MyPainter extends CustomPainter {
      @override
      void paint(Canvas canvas, Size size) {
        final paint = Paint()
          ..color = Colors.blue
          ..strokeWidth = 2
          ..style = PaintingStyle.stroke;
        
        // 绘制圆形
        canvas.drawCircle(
          Offset(size.width / 2, size.height / 2),
          50,
          paint,
        );
      }
      
      @override
      bool shouldRepaint(covariant CustomPainter oldDelegate) => false;
    }
    
    // 使用
    CustomPaint(
      painter: MyPainter(),
      size: Size(200, 200),
    )
    ''');
  }
}
```

### 2. Canvas 绘制方法

```dart
// Canvas 绘制方法
class CanvasDrawing {
  void explain() {
    print('''
    Canvas 绘制方法：
    
    // 1. 绘制直线
    canvas.drawLine(
      Offset(0, 0),
      Offset(100, 100),
      paint,
    );
    
    // 2. 绘制矩形
    canvas.drawRect(
      Rect.fromLTWH(0, 0, 100, 50),
      paint,
    );
    
    // 3. 绘制圆角矩形
    canvas.drawRRect(
      RRect.fromRectAndRadius(
        Rect.fromLTWH(0, 0, 100, 50),
        Radius.circular(10),
      ),
      paint,
    );
    
    // 4. 绘制圆形
    canvas.drawCircle(
      Offset(50, 50),
      30,
      paint,
    );
    
    // 5. 绘制椭圆
    canvas.drawOval(
      Rect.fromLTWH(0, 0, 100, 50),
      paint,
    );
    
    // 6. 绘制弧线
    canvas.drawArc(
      Rect.fromLTWH(0, 0, 100, 100),
      0,
      3.14 / 2,
      false,
      paint,
    );
    
    // 7. 绘制路径
    final path = Path()
      ..moveTo(0, 50)
      ..lineTo(50, 0)
      ..lineTo(100, 50)
      ..lineTo(50, 100)
      ..close();
    canvas.drawPath(path, paint);
    ''');
  }
}
```

## 📖 实战示例

### 1. 绘制进度条

```dart
// 自定义进度条
class ProgressPainter extends CustomPainter {
  final double progress;
  
  ProgressPainter({required this.progress});
  
  @override
  void paint(Canvas canvas, Size size) {
    // 背景
    final bgPaint = Paint()
      ..color = Colors.grey[300]!
      ..style = PaintingStyle.fill;
    
    canvas.drawRRect(
      RRect.fromRectAndRadius(
        Rect.fromLTWH(0, 0, size.width, size.height),
        Radius.circular(10),
      ),
      bgPaint,
    );
    
    // 进度
    final progressPaint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.fill;
    
    canvas.drawRRect(
      RRect.fromRectAndRadius(
        Rect.fromLTWH(0, 0, size.width * progress, size.height),
        Radius.circular(10),
      ),
      progressPaint,
    );
  }
  
  @override
  bool shouldRepaint(covariant ProgressPainter oldDelegate) {
    return oldDelegate.progress != progress;
  }
}
''');
  }
}
```
