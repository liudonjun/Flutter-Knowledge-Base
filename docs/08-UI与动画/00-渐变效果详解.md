### 2. 渐变效果实现

```dart
// 渐变效果实现
class GradientEffectsImplementation {
  void explain() {
    print('''
    渐变效果实现：
    
    // 1. 线性渐变
    Container(
      decoration: BoxDecoration(
        gradient: LinearGradient(
          begin: Alignment.topLeft,
          end: Alignment.bottomRight,
          colors: [Colors.blue, Colors.green],
        ),
      ),
      child: Text('Hello'),
    )
    
    // 2. 径向渐变
    Container(
      decoration: BoxDecoration(
        gradient: RadialGradient(
          center: Alignment.center,
          radius: 0.5,
          colors: [Colors.blue, Colors.green],
        ),
      ),
      child: Text('Hello'),
    )
    
    // 3. 扫描渐变
    Container(
      decoration: BoxDecoration(
        gradient: SweepGradient(
          center: Alignment.center,
          startAngle: 0.0,
          endAngle: 3.14159 * 2,
          colors: [Colors.blue, Colors.green, Colors.blue],
        ),
      ),
      child: Text('Hello'),
    )
    
    // 4. 使用画笔渐变
    final paint = Paint()
      ..shader = LinearGradient(
        colors: [Colors.blue, Colors.green],
      ).createShader(Rect.fromLTWH(0, 0, 100, 100));
    
    // 5. 渐变效果最佳实践
    // - 使用适当渐变
    // - 考虑性能
    // - 优化渐变效果
    // - 测试不同设备
    ''');
  }
}
```
