# 手势处理详解

> 掌握 Flutter 中的各种手势识别和处理。

## 📖 GestureDetector

### 1. 基本手势

```dart
// GestureDetector 基本手势
class GestureDetectorBasics {
  void explain() {
    print('''
    GestureDetector 手势：
    
    GestureDetector(
      // 点击
      onTap: () => print('点击'),
      onDoubleTap: () => print('双击'),
      onLongPress: () => print('长按'),
      
      // 拖动
      onPanStart: (details) => print('开始拖动'),
      onPanUpdate: (details) => print('拖动中: \${details.delta}'),
      onPanEnd: (details) => print('结束拖动'),
      
      // 缩放
      onScaleStart: (details) => print('开始缩放'),
      onScaleUpdate: (details) => print('缩放: \${details.scale}'),
      onScaleEnd: (details) => print('结束缩放'),
      
      // 滑动
      onHorizontalDragStart: (details) {},
      onHorizontalDragUpdate: (details) {},
      onVerticalDragStart: (details) {},
      onVerticalDragUpdate: (details) {},
      
      child: Container(
        width: 200,
        height: 200,
        color: Colors.blue,
        child: Center(child: Text('手势区域')),
      ),
    )
    ''');
  }
}
```

### 2. 手势冲突

```dart
// 手势冲突处理
class GestureConflict {
  void explain() {
    print('''
    手势冲突处理：
    
    // 1. 使用 Listener 监听原始指针
    Listener(
      onPointerDown: (event) => print('按下'),
      onPointerMove: (event) => print('移动'),
      onPointerUp: (event) => print('抬起'),
      child: Container(...),
    )
    
    // 2. 使用 RawGestureDetector
    RawGestureDetector(
      gestures: {
        TapGestureRecognizer: GestureRecognizerFactoryWithHandlers<TapGestureRecognizer>(
          () => TapGestureRecognizer(),
          (instance) {
            instance.onTap = () => print('点击');
          },
        ),
      },
      child: Container(...),
    )
    
    // 3. 手势竞争
    GestureDetector(
      onTap: () => print('被其他手势竞争掉了'),  // 可能不触发
      child: GestureDetector(
        onDoubleTap: () => print('双击优先'),
        child: Container(...),
      ),
    )
    ''');
  }
}
```

## 📖 可拖动组件

### 1. Draggable

```dart
// Draggable
class DraggableExample {
  void explain() {
    print('''
    Draggable 和 DragTarget：
    
    // 可拖动组件
    Draggable<int>(
      data: 42,  // 传递的数据
      child: Container(
        width: 100,
        height: 100,
        color: Colors.blue,
        child: Center(child: Text('拖动我')),
      ),
      feedback: Container(  // 拖动时显示的组件
        width: 100,
        height: 100,
        color: Colors.blue.withOpacity(0.5),
        child: Center(child: Text('拖动中')),
      ),
      childWhenDragging: Container(  // 原位置显示的组件
        width: 100,
        height: 100,
        color: Colors.grey,
      ),
      onDragCompleted: () => print('拖动完成'),
    )
    
    // 接收组件
    DragTarget<int>(
      builder: (context, candidateData, rejectedData) {
        return Container(
          width: 200,
          height: 200,
          color: candidateData.isNotEmpty ? Colors.green : Colors.red,
          child: Center(child: Text('放置区域')),
        );
      },
      onWillAccept: (data) => data != null,
      onAccept: (data) => print('接收到: \$data'),
      onLeave: (data) => print('离开了'),
    )
    ''');
  }
}
```

### 2. 滑动删除

```dart
// Dismissible - 滑动删除
class DismissibleExample {
  void explain() {
    print('''
    Dismissible 滑动删除：
    
    ListView.builder(
      itemCount: items.length,
      itemBuilder: (context, index) {
        return Dismissible(
          key: Key(items[index].id.toString()),
          
          // 滑动方向
          direction: DismissDirection.endToStart,
          
          // 背景
          background: Container(
            color: Colors.red,
            alignment: Alignment.centerRight,
            padding: EdgeInsets.only(right: 20),
            child: Icon(Icons.delete, color: Colors.white),
          ),
          
          // 确认删除
          confirmDismiss: (direction) async {
            return await showDialog(
              context: context,
              builder: (context) => AlertDialog(
                title: Text('确认删除'),
                content: Text('确定要删除吗？'),
                actions: [
                  TextButton(onPressed: () => Navigator.pop(context, false), child: Text('取消')),
                  TextButton(onPressed: () => Navigator.pop(context, true), child: Text('删除')),
                ],
              ),
            );
          },
          
          // 删除回调
          onDismissed: (direction) {
            setState(() {
              items.removeAt(index);
            });
            ScaffoldMessenger.of(context).showSnackBar(
              SnackBar(content: Text('已删除')),
            );
          },
          
          child: ListTile(title: Text(items[index].title)),
        );
      },
    )
    ''');
  }
}
```
