# 动画系统详解

> 掌握 Flutter 动画系统的核心概念和实现方法。

## 📖 动画基础

### 1. Animation 和 AnimationController

```dart
// 动画基础
class AnimationBasics {
  void explain() {
    print('''
    Animation 基础：
    
    class MyAnimation extends StatefulWidget {
      @override
      _MyAnimationState createState() => _MyAnimationState();
    }
    
    class _MyAnimationState extends State<MyAnimation>
        with SingleTickerProviderStateMixin {
      late AnimationController _controller;
      late Animation<double> _animation;
      
      @override
      void initState() {
        super.initState();
        
        // 创建动画控制器
        _controller = AnimationController(
          duration: Duration(seconds: 2),
          vsync: this,
        );
        
        // 创建动画
        _animation = Tween<double>(
          begin: 0,
          end: 1,
        ).animate(CurvedAnimation(
          parent: _controller,
          curve: Curves.easeInOut,
        ));
        
        // 开始动画
        _controller.forward();
      }
      
      @override
      void dispose() {
        _controller.dispose();
        super.dispose();
      }
      
      @override
      Widget build(BuildContext context) {
        return AnimatedBuilder(
          animation: _animation,
          builder: (context, child) {
            return Opacity(
              opacity: _animation.value,
              child: Container(width: 100, height: 100, color: Colors.blue),
            );
          },
        );
      }
    }
    ''');
  }
}
```

### 2. Tween 动画

```dart
// Tween 动画
class TweenAnimation {
  void explain() {
    print('''
    Tween 动画类型：
    
    // 1. 数值动画
    Tween<double>(begin: 0, end: 100)
    
    // 2. 颜色动画
    ColorTween(begin: Colors.red, end: Colors.blue)
    
    // 3. 尺寸动画
    SizeTween(begin: Size(0, 0), end: Size(200, 200))
    
    // 4. 位置动画
    Tween<Offset>(begin: Offset.zero, end: Offset(100, 100))
    
    // 5. 矩阵动画
    Matrix4Tween(begin: Matrix4.identity(), end: Matrix4.rotationZ(0.5))
    
    // 使用示例
    Animation<Color> colorAnimation = ColorTween(
      begin: Colors.red,
      end: Colors.blue,
    ).animate(_controller);
    ''');
  }
}
```

## 📖 隐式动画

### 1. AnimatedContainer

```dart
// AnimatedContainer
class AnimatedContainerExample {
  void explain() {
    print('''
    AnimatedContainer：
    
    // 自动动画化的 Container
    AnimatedContainer(
      duration: Duration(seconds: 1),
      curve: Curves.easeInOut,
      width: _isExpanded ? 200 : 100,
      height: _isExpanded ? 200 : 100,
      color: _isExpanded ? Colors.blue : Colors.red,
      child: Text('动画容器'),
    )
    
    // 点击触发动画
    GestureDetector(
      onTap: () {
        setState(() {
          _isExpanded = !_isExpanded;
        });
      },
      child: AnimatedContainer(...),
    )
    ''');
  }
}
```

### 2. 其他隐式动画

```dart
// 隐式动画 Widget
class ImplicitAnimations {
  void explain() {
    print('''
    隐式动画 Widget：
    
    // AnimatedOpacity - 透明度动画
    AnimatedOpacity(
      opacity: _visible ? 1.0 : 0.0,
      duration: Duration(milliseconds: 500),
      child: Container(...),
    )
    
    // AnimatedPositioned - 位置动画
    Stack(
      children: [
        AnimatedPositioned(
          duration: Duration(seconds: 1),
          left: _x,
          top: _y,
          child: Container(...),
        ),
      ],
    )
    
    // AnimatedPadding - 内边距动画
    AnimatedPadding(
      padding: EdgeInsets.all(_padding),
      duration: Duration(milliseconds: 300),
      child: Container(...),
    )
    
    // AnimatedAlign - 对齐动画
    AnimatedAlign(
      alignment: _alignment,
      duration: Duration(seconds: 1),
      child: Container(...),
    )
    
    // AnimatedTextStyle - 文本样式动画
    AnimatedDefaultTextStyle(
      style: TextStyle(fontSize: _fontSize, color: _color),
      duration: Duration(milliseconds: 300),
      child: Text('Hello'),
    )
    ''');
  }
}
```
