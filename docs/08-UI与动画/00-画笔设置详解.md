### 2. 画笔设置实现

```dart
// 画笔设置实现
class PaintSettingsImplementation {
  void explain() {
    print('''
    画笔设置实现：
    
    // 1. 基本画笔设置
    final paint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.fill
      ..strokeWidth = 2.0;
    
    // 2. 画笔样式
    // PaintingStyle.fill：填充
    // PaintingStyle.stroke：描边
    
    // 3. 线条设置
    final paint = Paint()
      ..color = Colors.blue
      ..style = PaintingStyle.stroke
      ..strokeWidth = 4.0
      ..strokeCap = StrokeCap.round
      ..strokeJoin = StrokeJoin.round;
    
    // 4. 渐变设置
    final paint = Paint()
      ..shader = LinearGradient(
        colors: [Colors.blue, Colors.green],
      ).createShader(Rect.fromLTWH(0, 0, 100, 100));
    
    // 5. 混合模式
    final paint = Paint()
      ..color = Colors.blue
      ..blendMode = BlendMode.srcOver;
    
    // 6. 画笔设置最佳实践
    // - 使用适当设置
    // - 考虑性能
    // - 优化绘制效果
    // - 测试不同设备
    ''');
  }
}
```
