# 变量与数据类型

> 掌握 Dart 语言的变量声明和数据类型，为 Flutter 开发打下坚实基础。

## 📖 变量声明

### 1. 变量声明方式

```dart
// Dart 变量声明方式
class VariableDeclaration {
  /*
  Dart 变量声明方式：
  
  1. var - 类型推断
  2. dynamic - 动态类型
  3. Object - 所有类型的基类
  4. 显式类型 - 具体类型声明
  5. final - 运行时常量
  6. const - 编译时常量
  */
  
  void explain() {
    print('''
    Dart 变量声明方式：
    
    1. var - 类型推断
       var name = 'Flutter';        // 推断为 String
       var age = 25;                // 推断为 int
       var price = 99.99;           // 推断为 double
       var isActive = true;         // 推断为 bool
       
       // 一旦赋值，类型就确定了
       var value = 'Hello';
       value = 123;  // 错误！类型不匹配
    
    2. dynamic - 动态类型
       dynamic variable = 'Hello';
       variable = 123;      // 可以
       variable = true;     // 可以
       variable = [1, 2];   // 可以
       
       // 编译时不检查类型，运行时检查
    
    3. Object - 对象类型
       Object obj = 'Hello';
       obj = 123;           // 可以
       obj = true;          // 可以
       
       // Object 是所有类型的基类
       // 使用时需要类型转换
    
    4. 显式类型
       String name = 'Flutter';
       int age = 25;
       double price = 99.99;
       bool isActive = true;
       
       // 明确指定类型
       // 编译时检查类型安全
    
    5. final - 运行时常量
       final name = 'Flutter';        // 类型推断
       final String city = 'Beijing'; // 显式类型
       
       // 只能赋值一次
       // 运行时确定值
    
    6. const - 编译时常量
       const PI = 3.14159;
       const String APP_NAME = 'MyApp';
       const List<int> numbers = [1, 2, 3];
       
       // 编译时确定值
       // 必须是编译时常量
    ''');
  }
}
```

### 2. 变量命名规范

```dart
// 变量命名规范
class NamingConventions {
  /*
  命名规范：
  
  1. 小驼峰命名法 (camelCase)
     - 变量、方法、参数
  
  2. 大驼峰命名法 (PascalCase)
     - 类、枚举、类型定义
  
  3. 下划线命名法 (snake_case)
     - 文件名
  
  4. 常量命名
     - const 使用大写下划线
     - final 使用小驼峰
  */
  
  void explain() {
    print('''
    命名规范：
    
    1. 变量命名 - 小驼峰 (camelCase)
       String userName = 'Alice';
       int userAge = 25;
       double accountBalance = 1000.50;
       bool isLoggedIn = true;
       
       // 私有变量以 _ 开头
       String _privateVariable = 'private';
       int _counter = 0;
    
    2. 常量命名
       // const - 大写下划线
       const double PI = 3.14159;
       const String API_BASE_URL = 'https://api.example.com';
       const int MAX_RETRY_COUNT = 3;
       
       // final - 小驼峰
       final DateTime now = DateTime.now();
       final String appName = 'MyApp';
    
    3. 类命名 - 大驼峰 (PascalCase)
       class UserProfile { }
       class HttpClient { }
       enum Color { red, green, blue }
    
    4. 文件命名 - 下划线 (snake_case)
       // user_profile.dart
       // http_client.dart
       // date_utils.dart
    
    命名建议：
    - 使用有意义的名称
    - 避免缩写（除非是通用的）
    - 布尔变量使用 is/has/can 前缀
    - 避免使用单字母变量（循环变量除外）
    ''');
  }
}
```

## 📖 数据类型

### 1. 数值类型

```dart
// 数值类型
class NumberTypes {
  /*
  数值类型：
  
  1. int - 整数
     - 64位有符号整数
     - 范围：-2^63 到 2^63-1
  
  2. double - 浮点数
     - 64位双精度浮点数
     - IEEE 754 标准
  
  3. num - 数值基类
     - int 和 double 的父类
     - 可以存储整数或浮点数
  */
  
  void explain() {
    print('''
    数值类型：
    
    1. int - 整数
       int age = 25;
       int count = 100;
       int negative = -10;
       
       // 不同进制
       int hex = 0xFF;           // 十六进制
       int oct = 077;            // 八进制
       int bin = 0b1010;         // 二进制
       
       // 大整数
       int bigNumber = 9007199254740991;
       
       // 常用属性和方法
       int value = 42;
       value.isEven;             // 是否偶数
       value.isOdd;              // 是否奇数
       value.abs();              // 绝对值
       value.toRadixString(16);  // 转换为十六进制字符串
    
    2. double - 浮点数
       double price = 99.99;
       double pi = 3.14159;
       double scientific = 1.5e10;  // 科学计数法
       
       // 特殊值
       double infinity = double.infinity;
       double negativeInfinity = double.negativeInfinity;
       double nan = double.nan;
       
       // 常用属性和方法
       double value = 3.14159;
       value.isFinite;          // 是否有限
       value.isInfinite;        // 是否无限
       value.isNaN;             // 是否 NaN
       value.round();           // 四舍五入
       value.floor();           // 向下取整
       value.ceil();            // 向上取整
       value.toStringAsFixed(2); // 保留2位小数
    
    3. num - 数值基类
       num value1 = 42;         // int
       num value2 = 3.14;       // double
       
       // 类型转换
       int intValue = value1.toInt();
       double doubleValue = value2.toDouble();
       
       // 数学运算
       num a = 10;
       num b = 3;
       a + b;   // 13
       a - b;   // 7
       a * b;   // 30
       a / b;   // 3.333...
       a ~/ b;  // 3 (整除)
       a % b;   // 1 (取余)
    ''');
  }
}
```

### 2. 字符串类型

```dart
// 字符串类型
class StringTypes {
  /*
  字符串类型：
  
  1. String - 字符串
     - UTF-16 编码
     - 不可变对象
     - 支持字符串插值
  
  2. 字符串创建方式
     - 单引号：'Hello'
     - 双引号："Hello"
     - 三引号：'''多行'''
  
  3. 字符串操作
     - 拼接、查找、替换
     - 分割、连接
     - 大小写转换
  */
  
  void explain() {
    print('''
    字符串类型：
    
    1. 字符串创建
       // 单引号
       String s1 = 'Hello';
       
       // 双引号
       String s2 = "World";
       
       // 三引号（多行字符串）
       String s3 = '''
       这是
       多行
       字符串
       ''';
       
       // 原始字符串（不转义）
       String s4 = r'这是原始字符串\n不会转义';
    
    2. 字符串插值
       String name = 'Alice';
       int age = 25;
       
       // 使用 $ 插值
       String message1 = 'Hello, $name!';
       
       // 使用 ${} 插值表达式
       String message2 = '明年你${age + 1}岁';
       
       // 插值特殊字符
       String message3 = '你的名字有${name.length}个字符';
    
    3. 字符串操作
       String str = 'Hello, World!';
       
       // 长度
       str.length;                 // 13
       
       // 查找
       str.contains('World');      // true
       str.startsWith('Hello');    // true
       str.endsWith('!');          // true
       str.indexOf('World');       // 7
       str.lastIndexOf('l');       // 10
       
       // 截取
       str.substring(7);           // 'World!'
       str.substring(0, 5);        // 'Hello'
       
       // 替换
       str.replaceAll('l', 'L');   // 'HeLLo, WorLd!'
       str.replaceFirst('l', 'L'); // 'HeLlo, World!'
       
       // 分割和连接
       str.split(', ');            // ['Hello', 'World!']
       ['a', 'b', 'c'].join('-');  // 'a-b-c'
       
       // 大小写转换
       str.toUpperCase();          // 'HELLO, WORLD!'
       str.toLowerCase();          // 'hello, world!'
       
       // 去除空格
       '  hello  '.trim();         // 'hello'
       '  hello  '.trimLeft();     // 'hello  '
       '  hello  '.trimRight();    // '  hello'
       
       // 填充
       '5'.padLeft(3, '0');        // '005'
       '5'.padRight(3, '0');       // '500'
       
       // 重复
       'ab' * 3;                   // 'ababab'
    ''');
  }
}
```

### 3. 布尔类型

```dart
// 布尔类型
class BooleanType {
  /*
  布尔类型：
  
  1. bool - 布尔值
     - true 或 false
     - 用于条件判断
  
  2. 布尔运算
     - 与：&&
     - 或：||
     - 非：!
  
  3. 布尔转换
     - 只有 true 是 true
     - 其他值都是 false
  */
  
  void explain() {
    print('''
    布尔类型：
    
    1. 基本使用
       bool isActive = true;
       bool isDeleted = false;
       bool hasPermission = true;
       
       // 条件判断
       if (isActive) {
         print('账户已激活');
       }
       
       if (!isDeleted) {
         print('记录未删除');
       }
    
    2. 布尔运算
       bool a = true;
       bool b = false;
       
       // 与运算
       a && b;                     // false
       true && true;               // true
       
       // 或运算
       a || b;                     // true
       false || false;             // false
       
       // 非运算
       !a;                         // false
       !b;                         // true
       
       // 短路运算
       true || print('不会执行');   // 短路，不执行 print
       false && print('不会执行');  // 短路，不执行 print
    
    3. 布尔转换
       // 非布尔值转换为 bool
       bool result1 = 1 == 1;      // true
       bool result2 = 1 > 2;       // false
       bool result3 = 'hello'.isNotEmpty; // true
       
       // 不能直接将其他类型转为 bool
       // bool b = 1;              // 错误
       // bool b = 'true';         // 错误
       
       // 需要显式比较
       bool b1 = 1 != 0;          // true
       bool b2 = 'hello'.length > 0; // true
    ''');
  }
}
```

### 4. 集合类型

```dart
// 集合类型
class CollectionTypes {
  /*
  集合类型：
  
  1. List - 列表
     - 有序集合
     - 可重复元素
     - 索引访问
  
  2. Set - 集合
     - 无序集合
     - 不重复元素
     - 集合运算
  
  3. Map - 映射
     - 键值对集合
     - 键不重复
     - 键值关联
  */
  
  void explain() {
    print('''
    集合类型：
    
    1. List - 列表
       // 创建
       List<int> numbers = [1, 2, 3, 4, 5];
       List<String> names = ['Alice', 'Bob', 'Charlie'];
       var mixed = [1, 'hello', true]; // List<dynamic>
       
       // 访问
       numbers[0];                 // 1
       numbers.first;              // 1
       numbers.last;               // 5
       numbers.length;             // 5
       
       // 修改
       numbers[0] = 10;
       numbers.add(6);
       numbers.addAll([7, 8]);
       numbers.insert(0, 0);
       numbers.remove(5);
       numbers.removeAt(0);
       
       // 查找
       numbers.contains(3);        // true
       numbers.indexOf(3);         // 2
       numbers.where((n) => n > 3); // (4, 5)
    
    2. Set - 集合
       // 创建
       Set<int> numbers = {1, 2, 3, 4, 5};
       Set<String> names = {'Alice', 'Bob'};
       
       // 添加和删除
       numbers.add(6);
       numbers.remove(1);
       numbers.contains(3);        // true
       
       // 集合运算
       Set<int> a = {1, 2, 3};
       Set<int> b = {3, 4, 5};
       
       a.union(b);                 // {1, 2, 3, 4, 5}
       a.intersection(b);          // {3}
       a.difference(b);            // {1, 2}
    
    3. Map - 映射
       // 创建
       Map<String, int> scores = {
         'Alice': 90,
         'Bob': 85,
         'Charlie': 95,
       };
       
       // 访问
       scores['Alice'];            // 90
       scores.keys;                // (Alice, Bob, Charlie)
       scores.values;              // (90, 85, 95)
       scores.length;              // 3
       
       // 修改
       scores['David'] = 88;
       scores.remove('Bob');
       scores.containsKey('Alice'); // true
       scores.containsValue(90);   // true
       
       // 遍历
       scores.forEach((key, value) {
         print('$key: $value');
       });
    ''');
  }
}
```

```dart
// 数值转换
class NumberConversion {
  /*
  数值转换：
  
  1. String 转数值
     - int.parse()
     - double.parse()
     - tryParse()
  
  2. 数值转 String
     - toString()
     - toStringAsFixed()
     - toStringAsPrecision()
  
  3. int 和 double 互转
     - toInt()
     - toDouble()
  */

    数值转换：
    
    1. String 转数值
       // int.parse
       int n1 = int.parse('42');           // 42
       int n2 = int.parse('0xFF', radix: 16); // 255
       
       // double.parse
       double d1 = double.parse('3.14');   // 3.14
       double d2 = double.parse('1.5e2');  // 150.0
       
       // tryParse - 安全转换
       int? n3 = int.tryParse('42');       // 42
       int? n4 = int.tryParse('abc');      // null
       double? d3 = double.tryParse('3.14'); // 3.14

       double d = 3.14159;

       d.toString();                 // '3.14159'

       d.toStringAsPrecision(3);     // '3.14'
       d.toStringAsExponential(2);   // '3.14e+0'
       
       // 不同进制
       255.toRadixString(16);        // 'ff'
       255.toRadixString(2);         // '11111111'
    
    3. int 和 double 互转
       int n = 42;
       double d = 3.14;
       
       // int 转 double
       double d1 = n.toDouble();     // 42.0
       
       // double 转 int（截断）
       int n1 = d.toInt();           // 3
       
       // 四舍五入
       int n2 = d.round();           // 3
       int n3 = d.ceil();            // 4
       int n4 = d.floor();           // 3
       int n5 = d.truncate();        // 3
    ''');
```

### 2. 类型检查

```dart
// 类型检查
class TypeChecking {
  /*
  类型检查：
  
  1. is - 类型检查
     - 检查对象是否是某种类型
  
  2. is! - 否定类型检查
     - 检查对象是否不是某种类型
  
  3. as - 类型转换
     - 将对象转换为指定类型
  */
  
  void explain() {
    print('''
    类型检查：
    
    1. is - 类型检查
       dynamic value = 'Hello';
       
       if (value is String) {
         print('是字符串');
         // 在这个作用域内，value 被当作 String
         print(value.length);
       }
       
       // 常用类型检查
       value is int;               // false
       value is double;            // false
       value is bool;              // false
       value is String;            // true
       value is List;              // false
       value is Map;               // false
    
    2. is! - 否定检查
       dynamic value = 42;
       
       if (value is! String) {
         print('不是字符串');
       }
    
    3. as - 类型转换
       dynamic value = 'Hello';
       
       // 强制转换
       String str = value as String;
       print(str.length);          // 5
       
       // 注意：转换失败会抛出异常
       // int n = value as int;    // 类型转换错误
       
       // 安全的做法
       if (value is String) {
         print(value.length);
       }
       
       // 使用 try-catch
       try {
         int n = value as int;
       } catch (e) {
         print('转换失败');
       }
    ''');
  }
}
```

## 📖 空安全

### 1. 空安全基础

```dart
// 空安全基础
class NullSafety {
  /*
  空安全（Null Safety）：
  
  1. 非空类型
     - 默认不可为 null
     - String、int、bool 等
  
  2. 可空类型
     - 使用 ? 标记
     - String?、int?、bool? 等
  
  3. 空安全操作符
     - ?.
     - ??
     - ??=
  */
  
  void explain() {
    print('''
    空安全基础：
    
    1. 非空类型
       String name = 'Alice';      // 不能为 null
       int age = 25;               // 不能为 null
       bool isActive = true;       // 不能为 null
       
       // name = null;             // 错误！不能赋值为 null
    
    2. 可空类型
       String? name = null;        // 可以为 null
       int? age = null;            // 可以为 null
       bool? isActive = null;      // 可以为 null
       
       // 使用前需要检查
       if (name != null) {
         print(name.length);
       }
    
    3. 安全操作符
       String? name = null;
       
       // ?. - 安全调用
       int? length = name?.length; // null
       
       // ?? - 默认值
       String displayName = name ?? '匿名';
       
       // ??= - 条件赋值
       name ??= '默认名称';
       
       // ! - 非空断言（谨慎使用）
       // String s = name!;        // 如果 name 是 null，会抛出异常
    
    4. late 关键字
       late String name;
       
       // 延迟初始化
       void init() {
         name = 'Alice';
       }
       
       // 使用前必须初始化
       // 否则运行时会抛出异常
       
       // late final - 延迟初始化，只能赋值一次
       late final String id;
       id = '123';
       // id = '456';  // 错误！
    ''');
  }
}
```

### 数据类型速查表

| 类型 | 说明 | 示例 |
|------|------|------|
| **int** | 整数 | `int age = 25;` |
| **double** | 浮点数 | `double price = 99.99;` |
| **num** | 数值基类 | `num value = 42;` |
| **String** | 字符串 | `String name = 'Alice';` |
| **bool** | 布尔值 | `bool isActive = true;` |
| **List** | 列表 | `List<int> numbers = [1, 2, 3];` |
| **Set** | 集合 | `Set<int> unique = {1, 2, 3};` |
| **Map** | 映射 | `Map<String, int> scores = {};` |

### 变量声明方式

| 关键字 | 说明 | 示例 |
|--------|------|------|
| **var** | 类型推断 | `var name = 'Alice';` |
| **dynamic** | 动态类型 | `dynamic value = 42;` |

| **const** | 编译时常量 | `const PI = 3.14;` |
| **late** | 延迟初始化 | `late String name;` |
