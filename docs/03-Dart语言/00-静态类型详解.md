### 2. 类型系统

```dart
// 类型系统
class TypeSystem {
  void explain() {
    print('''
    类型系统：
    
    // 1. 基本类型
    int integer = 42;
    double decimal = 3.14;
    String text = 'Hello';
    bool flag = true;
    
    // 2. 集合类型
    List<int> numbers = [1, 2, 3];
    Map<String, int> scores = {'math': 90};
    Set<String> names = {'John', 'Jane'};
    
    // 3. 函数类型
    int Function(int, int) add = (a, b) => a + b;

      String name;
      int age;
      
      Person(this.name, this.age);
    }
    
    Person person = Person('John', 30);
    
    // 5. 类型层次
    // Object
    // ├── num
    // │   ├── int
    // │   └── double
    // ├── String
    // ├── bool
    // ├── List<E>
    // ├── Map<K, V>
    // └── ...
    
    // 6. 类型检查
    void checkType(Object obj) {
      if (obj is String) {
        print('String: $obj');
      } else if (obj is int) {
        print('int: $obj');
      } else {
        print('Other: $obj');
      }
    }
    ''');
  }
}
```

## 🔧 类型操作

### 1. 类型转换

```dart
// 类型转换
class TypeCasting {
  void explain() {
    print('''
    类型转换：
    
    // 1. 显式转换
    num value = 42;
    int integer = value as int;
    
    // 2. 类型检查
    void process(Object obj) {
      if (obj is String) {
        print(obj.length);  // 自动提升为 String
      }
    }
    
    // 3. 安全转换
    num value = 42.5;
    int? integer = value is int ? value : null;
    
    // 4. 转换失败
    try {
      num value = 42.5;
      int integer = value as int;  // 运行时错误
    } catch (e) {
      print('转换失败: $e');
    }
    
    // 5. 类型转换最佳实践
    // - 使用 is 检查类型
    // - 使用 as 进行转换
    // - 处理转换失败的情况
    // - 避免不必要的转换
    ''');
  }
}
```

### 2. 泛型

```dart
// 泛型
class Generics {
  void explain() {
    print('''
    泛型：
    
    // 1. 泛型类
    class Box<T> {
      T value;
      
      Box(this.value);
      
      T getValue() => value;
    }
    
    Box<int> intBox = Box(42);
    Box<String> stringBox = Box('Hello');
    
    // 2. 泛型函数
    T first<T>(List<T> list) {
      return list.first;
    }
    
    int firstInt = first([1, 2, 3]);
    String firstString = first(['a', 'b', 'c']);
    
    // 3. 泛型约束
    class NumberBox<T extends num> {
      T value;
      
      NumberBox(this.value);
      
      double toDouble() => value.toDouble();
    }
    
    NumberBox<int> intBox = NumberBox(42);
    NumberBox<double> doubleBox = NumberBox(3.14);
    
    // 4. 泛型最佳实践
    // - 使用泛型提高代码复用
    // - 使用约束限制泛型类型
    // - 保持泛型参数有意义
    // - 避免过度使用泛型
    ''');
  }
}
```
