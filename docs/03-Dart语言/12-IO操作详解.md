# IO 操作详解

> 掌握 Dart 的文件和 IO 操作，包括文件读写、目录操作和网络 IO。

## 📖 文件操作

### 1. 读取文件

```dart
// 读取文件
class FileReading {
  void explain() {
    print('''
    文件读取：
    
    import 'dart:io';
    
    // 1. 读取全部内容
    Future<String> readAll() async {
      var file = File('path/to/file.txt');
      return await file.readAsString();
    }
    
    // 2. 按行读取
    Future<List<String>> readLines() async {
      var file = File('path/to/file.txt');
      return await file.readAsLines();
    }
    
    // 3. 读取字节
    Future<List<int>> readBytes() async {
      var file = File('path/to/file.txt');
      return await file.readAsBytes();
    }
    
    // 4. 流式读取（大文件）
    Stream<String> readAsStream() {
      var file = File('path/to/file.txt');
      return file.openRead()
          .transform(utf8.decoder)
          .transform(LineSplitter());
    }
    
    // 5. 同步读取
    String readSync() {
      var file = File('path/to/file.txt');
      return file.readAsStringSync();
    }
    ''');
  }
}
```

### 2. 写入文件

```dart
// 写入文件
class FileWriting {
  void explain() {
    print('''
    文件写入：
    
    import 'dart:io';
    
    // 1. 写入字符串
    Future<void> writeString() async {
      var file = File('path/to/file.txt');
      await file.writeAsString('Hello, Dart!');
    }
    
    // 2. 追加内容
    Future<void> append() async {
      var file = File('path/to/file.txt');
      await file.writeAsString(
        '追加的内容',
        mode: FileMode.append,
      );
    }
    
    // 3. 写入字节
    Future<void> writeBytes() async {
      var file = File('path/to/file.txt');
      await file.writeAsBytes([72, 101, 108, 108, 111]);
    }
    
    // 4. 流式写入
    Future<void> writeStream() async {
      var file = File('path/to/file.txt');
      var sink = file.openWrite();
      sink.write('第一行\n');
      sink.write('第二行\n');
      await sink.close();
    }
    ''');
  }
}
```

## 📖 目录操作

```dart
// 目录操作
class DirectoryOperations {
  void explain() {
    print('''
    目录操作：
    
    import 'dart:io';
    
    // 1. 创建目录
    Future<void> createDir() async {
      var dir = Directory('path/to/dir');
      await dir.create(recursive: true);
    }
    
    // 2. 列出文件
    Future<void> listFiles() async {
      var dir = Directory('path/to/dir');
      await for (var entity in dir.list()) {
        print(entity.path);
      }
    }
    
    // 3. 递归列出
    Future<void> listRecursive() async {
      var dir = Directory('path/to/dir');
      await for (var entity in dir.list(recursive: true)) {
        print(entity.path);
      }
    }
    
    // 4. 删除目录
    Future<void> deleteDir() async {
      var dir = Directory('path/to/dir');
      await dir.delete(recursive: true);
    }
    
    // 5. 检查是否存在
    Future<bool> checkExists() async {
      var dir = Directory('path/to/dir');
      return await dir.exists();
    }
    ''');
  }
}
```

## 📖 路径操作

```dart
// 路径操作
class PathOperations {
  void explain() {
    print('''
    路径操作：
    
    import 'package:path/path.dart' as path;
    
    // 1. 路径拼接
    var fullPath = path.join('dir', 'subdir', 'file.txt');
    
    // 2. 获取文件名
    var fileName = path.basename('/path/to/file.txt');  // file.txt
    
    // 3. 获取扩展名
    var extension = path.extension('file.txt');  // .txt
    
    // 4. 获取目录
    var dir = path.dirname('/path/to/file.txt');  // /path/to
    
    // 5. 获取绝对路径
    var absolute = path.absolute('relative/path');
    
    // 6. 规范化路径
    var normalized = path.normalize('/path/../other/./file');
    ''');
  }
}
```
