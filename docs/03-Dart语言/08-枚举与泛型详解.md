# 枚举与泛型详解

> 掌握 Dart 语言的枚举类型和泛型编程。

## 📖 枚举

### 1. 基本枚举

```dart
// 基本枚举
class EnumBasics {
  void explain() {
    print('''
    基本枚举：
    
    enum Color {
      red,
      green,
      blue,
    }
    
    // 使用枚举
    Color favoriteColor = Color.blue;
    
    // 枚举属性
    print(Color.red.name);      // 'red'
    print(Color.red.index);     // 0
    print(Color.values);        // [Color.red, Color.green, Color.blue]
    
    // switch 使用枚举
    switch (favoriteColor) {
      case Color.red:
        print('红色');
        break;
      case Color.green:
        print('绿色');
        break;
      case Color.blue:
        print('蓝色');
        break;
    }
    ''');
  }
}
```

### 2. 增强枚举（Dart 2.17+）

```dart
// 增强枚举
class EnhancedEnum {
  void explain() {
    print('''
    增强枚举（Dart 2.17+）：
    
    enum HttpStatus {
      ok(200, 'OK'),
      notFound(404, 'Not Found'),
      serverError(500, 'Internal Server Error');
      
      final int code;
      final String message;
      
      const HttpStatus(this.code, this.message);
      
      bool get isSuccess => code >= 200 && code < 300;
      bool get isError => code >= 400;
    }
    
    // 使用
    var status = HttpStatus.ok;
    print(status.code);     // 200
    print(status.message);  // 'OK'
    print(status.isSuccess); // true
    ''');
  }
}
```

## 📖 泛型

### 1. 泛型基础

```dart
// 泛型基础
class GenericBasics {
  void explain() {
    print('''
    泛型基础：
    
    // 泛型类
    class Box<T> {
      T value;
      
      Box(this.value);
      
      T getValue() => value;
      void setValue(T newValue) => value = newValue;
    }
    
    // 使用泛型类
    var intBox = Box<int>(42);
    var stringBox = Box<String>('Hello');
    var listBox = Box<List<int>>([1, 2, 3]);
    
    print(intBox.getValue());     // 42
    print(stringBox.getValue());  // 'Hello'
    
    // 泛型函数
    T first<T>(List<T> list) {
      if (list.isEmpty) throw Exception('列表为空');
      return list.first;
    }
    
    var numbers = [1, 2, 3];
    var strings = ['a', 'b', 'c'];
    
    print(first(numbers));  // 1
    print(first(strings));  // 'a'
    ''');
  }
}
```

### 2. 泛型约束

```dart
// 泛型约束
class GenericConstraints {
  void explain() {
    print('''
    泛型约束：
    
    // 限制泛型类型
    class NumberBox<T extends num> {
      T value;
      
      NumberBox(this.value);
      
      double toDouble() => value.toDouble();
    }
    
    var intBox = NumberBox<int>(42);
    var doubleBox = NumberBox<double>(3.14);
    // var stringBox = NumberBox<String>('hello');  // 错误！
    
    // 多重约束
    class Repository<T extends Comparable<T>> {
      List<T> items = [];
      
      void add(T item) => items.add(item);
      
      T? findMax() {
        if (items.isEmpty) return null;
        return items.reduce((a, b) => a.compareTo(b) > 0 ? a : b);
      }
    }
    ''');
  }
}
```

### 3. 泛型与集合

```dart
// 泛型与集合
class GenericCollections {
  void explain() {
    print('''
    泛型与集合：
    
    // List 泛型
    List<int> numbers = [1, 2, 3];
    List<String> names = ['Alice', 'Bob'];
    
    // Map 泛型
    Map<String, int> scores = {
      'Alice': 90,
      'Bob': 85,
    };
    
    // Set 泛型
    Set<String> uniqueNames = {'Alice', 'Bob', 'Charlie'};
    
    // 泛型方法
    List<T> createList<T>(T value, int count) {
      return List.filled(count, value);
    }
    
    var intList = createList<int>(0, 5);      // [0, 0, 0, 0, 0]
    var stringList = createList<String>('', 3); // ['', '', '']
    
    // 泛型转换
    List<Object> objects = [1, 'hello', 3.14];
    List<int> ints = objects.whereType<int>().toList();
    print(ints);  // [1]
    ''');
  }
}
```
