# 运算符详解

> 掌握 Dart 语言的各种运算符，包括算术、比较、逻辑、位运算等。

## 📖 算术运算符

### 1. 基本算术运算

```dart
// 算术运算符
class ArithmeticOperators {
  /*
  算术运算符：
  
  +   加法
  -   减法
  *   乘法
  /   除法（返回 double）
  ~/  整除（返回 int）
  %   取余
  */
  
  void explain() {
    print('''
    算术运算符：
    
    int a = 10;
    int b = 3;
    
    1. 基本运算
       a + b   // 13  - 加法
       a - b   // 7   - 减法
       a * b   // 30  - 乘法
       a / b   // 3.333... - 除法（返回 double）
       a ~/ b  // 3   - 整除（返回 int）
       a % b   // 1   - 取余
    
    2. 一元运算符
       -a      // -10 - 负号
       +a      // 10  - 正号
       a++     // 先返回 a，再加 1
       ++a     // 先加 1，再返回
       a--     // 先返回 a，再减 1
       --a     // 先减 1，再返回
    
    3. 复合赋值运算符
       a += b  // a = a + b
       a -= b  // a = a - b
       a *= b  // a = a * b
       a /= b  // a = a / b
       a ~/= b // a = a ~/ b
       a %= b  // a = a % b
    ''');
  }
}
```

## 📖 比较运算符

```dart
// 比较运算符
class ComparisonOperators {
  /*
  比较运算符：
  
  ==  等于
  !=  不等于
  >   大于
  <   小于
  >=  大于等于
  <=  小于等于
  */
  
  void explain() {
    print('''
    比较运算符：
    
    int a = 10;
    int b = 20;
    
    a == b   // false - 等于
    a != b   // true  - 不等于
    a > b    // false - 大于
    a < b    // true  - 小于
    a >= b   // false - 大于等于
    a <= b   // true  - 小于等于
    
    // 比较运算符返回 bool 类型
    bool result = a < b;  // true
    ''');
  }
}
```

## 📖 逻辑运算符

```dart
// 逻辑运算符
class LogicalOperators {
  /*
  逻辑运算符：
  
  &&  逻辑与（AND）
  ||  逻辑或（OR）
  !   逻辑非（NOT）
  */
  
  void explain() {
    print('''
    逻辑运算符：
    
    bool a = true;
    bool b = false;
    
    // 逻辑与
    a && b   // false - 两个都为 true 才为 true
    
    // 逻辑或
    a || b   // true  - 有一个为 true 就为 true
    
    // 逻辑非
    !a       // false - 取反
    !b       // true
    
    // 短路求值
    true || print('不会执行')   // 短路，不执行 print
    false && print('不会执行')  // 短路，不执行 print
    ''');
  }
}
```

## 📖 赋值运算符

```dart
// 赋值运算符
class AssignmentOperators {
  /*
  赋值运算符：
  
  =     赋值
  ??=   空赋值（仅当变量为 null 时赋值）
  */
  
  void explain() {
    print('''
    赋值运算符：
    
    // 基本赋值
    int a = 10;
    
    // 复合赋值
    a += 5;   // a = a + 5 = 15
    a -= 3;   // a = a - 3 = 12
    a *= 2;   // a = a * 2 = 24
    a ~/= 4;  // a = a ~/ 4 = 6
    a %= 4;   // a = a % 4 = 2
    
    // 空赋值
    String? name;
    name ??= '默认值';  // name 为 null，所以赋值
    name ??= '其他值';  // name 不为 null，不赋值
    
    print(name);  // '默认值'
    ''');
  }
}
```

## 📖 条件运算符

```dart
// 条件运算符
class ConditionalOperators {
  /*
  条件运算符：
  
  condition ? expr1 : expr2   三元运算符
  expr1 ?? expr2              空合并运算符
  */
  
  void explain() {
    print('''
    条件运算符：
    
    // 三元运算符
    int age = 20;
    String status = age >= 18 ? '成年' : '未成年';
    
    // 空合并运算符
    String? name;
    String displayName = name ?? '匿名';
    
    // 链式空合并
    String? first;
    String? second;
    String result = first ?? second ?? '默认值';
    ''');
  }
}
```

## 📖 类型运算符

```dart
// 类型运算符
class TypeOperators {
  /*
  类型运算符：
  
  as    类型转换
  is    类型检查（是）
  is!   类型检查（不是）
  */
  
  void explain() {
    print('''
    类型运算符：
    
    dynamic value = 'Hello';
    
    // is - 类型检查
    if (value is String) {
      print(value.length);  // value 被当作 String
    }
    
    // is! - 否定类型检查
    if (value is! int) {
      print('不是整数');
    }
    
    // as - 类型转换
    String str = value as String;
    print(str.length);
    
    // 注意：错误的类型转换会抛出异常
    // int n = value as int;  // 运行时错误
    ''');
  }
}
```

## 📖 级联运算符

```dart
// 级联运算符
class CascadeNotation {
  /*
  级联运算符：..
  
  允许对同一个对象进行一系列操作
  */
  
  void explain() {
    print('''
    级联运算符：..
    
    // 不使用级联
    var list = [1, 2, 3];
    list.add(4);
    list.remove(1);
    list.sort();
    
    // 使用级联
    var list2 = [1, 2, 3]
      ..add(4)
      ..remove(1)
      ..sort();
    
    // 级联可以链式调用
    class Person {
      String name = '';
      int age = 0;
      
      void setName(String name) => this.name = name;
      void setAge(int age) => this.age = age;
    }
    
    var person = Person()
      ..setName('Alice')
      ..setAge(25);
    
    // 等价于
    var person2 = Person();
    person2.setName('Alice');
    person2.setAge(25);
    ''');
  }
}
```
