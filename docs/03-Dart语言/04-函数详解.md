# 函数详解

> 掌握 Dart 语言的函数定义、参数传递、匿名函数、闭包等核心概念。

## 📖 函数基础

### 1. 函数定义

```dart
// 函数定义方式
class FunctionBasics {
  void explain() {
    print('''
    函数定义：
    
    // 基本函数
    返回类型 函数名(参数列表) {
      函数体
      return 返回值;
    }
    
    // 示例
    int add(int a, int b) {
      return a + b;
    }
    
    // 箭头函数（单表达式）
    int multiply(int a, int b) => a * b;
    
    // 无返回值函数
    void printMessage(String message) {
      print(message);
    }
    
    // 箭头函数无返回值
    void greet(String name) => print('Hello, $name');
    ''');
  }
}
```

### 2. 参数类型

```dart
// 参数类型
class FunctionParameters {
  void explain() {
    print('''
    参数类型：
    
    // 位置参数（必需）
    String greet(String name, int age) {
      return 'Hello, $name. You are $age years old.';
    }
    
    // 可选位置参数
    String greetOptional(String name, [int? age]) {
      if (age != null) {
        return 'Hello, $name. You are $age years old.';
      }
      return 'Hello, $name.';
    }
    
    // 命名参数
    String greetNamed({required String name, int age = 18}) {
      return 'Hello, $name. You are $age years old.';
    }
    
    // 使用
    greet('Alice', 25);
    greetOptional('Bob');
    greetOptional('Charlie', 30);
    greetNamed(name: 'David');
    greetNamed(name: 'Eve', age: 28);
    ''');
  }
}
```

## 📖 匿名函数

### 1. Lambda 表达式

```dart
// Lambda 表达式
class LambdaExpressions {
  void explain() {
    print('''
    Lambda 表达式：
    
    // 基本 lambda
    var add = (int a, int b) => a + b;
    
    // 等价写法
    int add(int a, int b) {
      return a + b;
    }
    
    // 无参数 lambda
    var greet = () => print('Hello');
    
    // 带类型的 lambda
    var multiply = (int a, int b) {
      return a * b;
    };
    
    // 使用 lambda
    var numbers = [1, 2, 3, 4, 5];
    var doubled = numbers.map((n) => n * 2).toList();
    print(doubled);  // [2, 4, 6, 8, 10]
    
    // forEach 使用 lambda
    numbers.forEach((n) {
      print(n);
    });
    ''');
  }
}
```

### 2. 高阶函数

```dart
// 高阶函数
class HigherOrderFunctions {
  void explain() {
    print('''
    高阶函数：
    
    // 函数作为参数
    void executeFunction(Function function) {
      function();
    }
    
    // 函数作为返回值
    Function createMultiplier(int factor) {
      return (int number) => number * factor;
    }
    
    // 示例
    var double = createMultiplier(2);
    var triple = createMultiplier(3);
    
    print(double(5));  // 10
    print(triple(5));  // 15
    
    // 内置高阶函数
    var numbers = [1, 2, 3, 4, 5];
    
    // map: 转换每个元素
    var doubled = numbers.map((n) => n * 2).toList();
    
    // where: 过滤元素
    var evens = numbers.where((n) => n % 2 == 0).toList();
    
    // reduce: 合并元素
    var sum = numbers.reduce((a, b) => a + b);
    
    // fold: 带初始值的合并
    var sumWithInitial = numbers.fold(0, (sum, n) => sum + n);
    ''');
  }
}
```

## 📖 闭包

### 1. 闭包概念

```dart
// 闭包
class Closures {
  void explain() {
    print('''
    闭包：
    
    // 闭包是可以访问外部变量的函数
    Function createCounter() {
      int count = 0;
      return () {
        count++;
        return count;
      };
    }
    
    var counter = createCounter();
    print(counter());  // 1
    print(counter());  // 2
    print(counter());  // 3
    
    // 多个闭包共享变量
    Function createAdder(int addBy) {
      return (int number) => number + addBy;
    }
    
    var add2 = createAdder(2);
    var add5 = createAdder(5);
    
    print(add2(10));  // 12
    print(add5(10));  // 15
    ''');
  }
}
```

## 📖 递归函数

### 1. 递归

```dart
// 递归函数
class RecursiveFunctions {
  void explain() {
    print('''
    递归函数：
    
    // 阶乘
    int factorial(int n) {
      if (n <= 1) return 1;
      return n * factorial(n - 1);
    }
    
    print(factorial(5));  // 120
    
    // 斐波那契数列
    int fibonacci(int n) {
      if (n <= 1) return n;
      return fibonacci(n - 1) + fibonacci(n - 2);
    }
    
    print(fibonacci(10));  // 55
    
    // 尾递归优化（使用迭代）
    int factorialTailRecursive(int n, [int accumulator = 1]) {
      if (n <= 1) return accumulator;
      return factorialTailRecursive(n - 1, n * accumulator);
    }
    ''');
  }
}
```
