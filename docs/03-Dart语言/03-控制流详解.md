# 控制流详解

> 掌握 Dart 语言的控制流语句，包括条件语句、循环语句和跳转语句。

## 📖 条件语句

### 1. if-else 语句

```dart
// if-else 语句
class IfElseStatement {
  void explain() {
    print('''
    if-else 语句：
    
    // 基本 if 语句
    if (condition) {
      // 条件为 true 时执行
    }
    
    // if-else 语句
    if (condition) {
      // 条件为 true 时执行
    } else {
      // 条件为 false 时执行
    }
    
    // if-else if-else 语句
    if (score >= 90) {
      grade = 'A';
    } else if (score >= 80) {
      grade = 'B';
    } else if (score >= 70) {
      grade = 'C';
    } else {
      grade = 'D';
    }
    ''');
  }
}
```

### 2. switch-case 语句

```dart
// switch-case 语句
class SwitchCaseStatement {
  void explain() {
    print('''
    switch-case 语句：
    
    switch (expression) {
      case value1:
        // 代码
        break;
      case value2:
        // 代码
        break;
      default:
        // 默认代码
    }
    
    // 示例
    String command = 'OPEN';
    switch (command) {
      case 'CLOSED':
        executeClosed();
        break;
      case 'PENDING':
        executePending();
        break;
      case 'APPROVED':
        executeApproved();
        break;
      case 'DENIED':
        executeDenied();
        break;
      case 'OPEN':
        executeOpen();
        break;
      default:
        executeUnknown();
    }
    ''');
  }
}
```

## 📖 循环语句

### 1. for 循环

```dart
// for 循环
class ForLoop {
  void explain() {
    print('''
    for 循环：
    
    // 标准 for 循环
    for (int i = 0; i < 5; i++) {
      print(i);
    }
    
    // for-in 循环
    var numbers = [1, 2, 3, 4, 5];
    for (var number in numbers) {
      print(number);
    }
    
    // forEach 方法
    numbers.forEach((number) {
      print(number);
    });
    ''');
  }
}
```

### 2. while 和 do-while 循环

```dart
// while 和 do-while 循环
class WhileLoop {
  void explain() {
    print('''
    while 和 do-while 循环：
    
    // while 循环
    int count = 0;
    while (count < 5) {
      print(count);
      count++;
    }
    
    // do-while 循环
    int number = 0;
    do {
      print(number);
      number++;
    } while (number < 5);
    
    // 区别：do-while 至少执行一次
    ''');
  }
}
```

## 📖 跳转语句

### 1. break 和 continue

```dart
// break 和 continue
class JumpStatements {
  void explain() {
    print('''
    跳转语句：
    
    // break - 跳出循环
    for (int i = 0; i < 10; i++) {
      if (i == 5) {
        break;  // 当 i 等于 5 时跳出循环
      }
      print(i);
    }
    // 输出: 0, 1, 2, 3, 4
    
    // continue - 跳过当前迭代
    for (int i = 0; i < 5; i++) {
      if (i == 2) {
        continue;  // 跳过 i 等于 2 的迭代
      }
      print(i);
    }
    // 输出: 0, 1, 3, 4
    
    // 标签跳转
    outerLoop:
    for (int i = 0; i < 3; i++) {
      for (int j = 0; j < 3; j++) {
        if (i == 1 && j == 1) {
          break outerLoop;  // 跳出外层循环
        }
        print('i=$i, j=$j');
      }
    }
    ''');
  }
}
```

## 📖 断言

### 1. assert 语句

```dart
// assert 语句
class AssertStatement {
  void explain() {
    print('''
    assert 语句：
    
    // 基本断言
    assert(condition);
    
    // 带消息的断言
    assert(condition, '错误消息');
    
    // 示例
    void setAge(int age) {
      assert(age >= 0, '年龄不能为负数');
      assert(age <= 150, '年龄不能超过150');
      this.age = age;
    }
    
    // 断言仅在调试模式下生效
    // 发布模式下会被移除
    ''');
  }
}
```
