### 2. 断点调试

```dart
// 断点调试
class BreakpointDebugging {
  void explain() {
    print('''
    断点调试：
    
    // 1. 设置断点
    // - 在 IDE 中点击行号设置断点
    // - 条件断点：满足条件时暂停
    // - 日志断点：输出日志不暂停
    // - 异常断点：捕获异常时暂停
    
    // 2. 调试控制
    // - 继续执行：F8 / F5
    // - 单步执行：F10
    // - 步入函数：F11
    // - 步出函数：Shift+F11
    
    // 3. 变量查看
    // - 局部变量：查看当前作用域
    // - 监视表达式：监视特定表达式
    // - 调用堆栈：查看函数调用链
    
    // 4. 断点调试示例
    void processList(List<int> list) {
      for (int i = 0; i < list.length; i++) {
        // 设置断点：查看 list[i] 的值
        int value = list[i];
        
        if (value > 100) {
          // 条件断点：value > 100 时暂停
          print('大值: $value');
        }
      }
    }
    ''');
  }
}
```

## 🔧 调试技巧

### 1. 常见问题调试

```dart
// 常见问题调试
class CommonIssueDebugging {
  void explain() {
    print('''
    常见问题调试：
    
    // 1. 空指针调试
    void debugNullPointer() {
      String? name = null;
      
      // 安全调用
      int? length = name?.length;
      
      // 空检查
      if (name != null) {
        print(name.length);
      }
      
      // 空合并
      String displayName = name ?? 'Unknown';
    }
    
    // 2. 类型错误调试
    void debugTypeError() {
      Object obj = 'Hello';
      
      // 类型检查
      if (obj is String) {
        print(obj.length);
      }
      
      // 类型转换
      try {
        String str = obj as String;
        print(str.length);
      } catch (e) {
        print('类型转换错误: $e');
      }
    }
    
    // 3. 越界错误调试
    void debugIndexError() {
      List<int> list = [1, 2, 3];
      
      // 边界检查
      if (index >= 0 && index < list.length) {
        print(list[index]);
      } else {
        print('索引越界: $index');
      }
    }
    
    // 4. 异步错误调试
    void debugAsyncError() async {
      try {
        final result = await asyncOperation();
        print(result);
      } catch (e, stackTrace) {
        print('异步错误: $e');
        print('堆栈: $stackTrace');
      }
    }
    ''');
  }
}
```
