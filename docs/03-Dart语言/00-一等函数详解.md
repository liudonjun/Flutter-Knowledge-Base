## 📖 一等函数基础

### 1. 什么是一等函数

```dart
// 一等函数概念
class FirstClassFunctionsConcept {
  void explain() {
    print('''

    // - 可测试性：更容易测试
    ''');
```

### 2. 函数类型

```dart
// 函数类型
class FunctionTypes {
  void explain() {
    print('''
    函数类型：
    
    // 1. 函数类型定义
    // 函数类型语法：ReturnType Function(ParameterTypes)
    typedef IntTransformer = int Function(int);
    
    // 2. 函数作为参数
    void applyOperation(int a, int b, int Function(int, int) operation) {
      final result = operation(a, b);
      print('Result: $result');
    }
    
    // 使用
    applyOperation(5, 3, (a, b) => a + b);
    applyOperation(5, 3, (a, b) => a * b);
    
    // 3. 函数作为返回值
    Function(int) createMultiplier(int factor) {
      return (int value) => value * factor;
    }
    
    // 使用
    final doubler = createMultiplier(2);
    final tripler = createMultiplier(3);
    print(doubler(5)); // 10
    print(tripler(5)); // 15
    
    // 4. 函数赋值给变量
    int add(int a, int b) => a + b;
    
    final operation = add;
    print(operation(5, 3)); // 8
    
    // 匿名函数赋值
    final multiply = (int a, int b) => a * b;
    print(multiply(5, 3)); // 15
    ''');
  }
}
```

## 🔧 高阶函数

### 1. 常用高阶函数

```dart
// 常用高阶函数
class HigherOrderFunctions {
  void explain() {
    print('''
    常用高阶函数：
    
    // 1. map - 转换每个元素
    final numbers = [1, 2, 3, 4, 5];
    final doubled = numbers.map((n) => n * 2).toList();
    print(doubled); // [2, 4, 6, 8, 10]
    
    // 2. where - 过滤元素
    final evenNumbers = numbers.where((n) => n % 2 == 0).toList();
    print(evenNumbers); // [2, 4]
    
    // 3. reduce - 聚合元素
    final sum = numbers.reduce((a, b) => a + b);
    print(sum); // 15
    
    // 4. fold - 带初始值的聚合
    final sumWithInitial = numbers.fold(10, (a, b) => a + b);
    print(sumWithInitial); // 25
    
    // 5. forEach - 遍历元素
    numbers.forEach((n) => print(n));
    
    // 6. any - 检查是否有满足条件的元素
    final hasEven = numbers.any((n) => n % 2 == 0);
    print(hasEven); // true
    
    // 7. every - 检查是否所有元素满足条件
    final allPositive = numbers.every((n) => n > 0);
    print(allPositive); // true
    
    // 8. firstWhere - 查找第一个满足条件的元素
    final firstEven = numbers.firstWhere((n) => n % 2 == 0);
    print(firstEven); // 2
    ''');
  }
}
```

### 2. 函数组合

```dart
// 函数组合
class FunctionComposition {
  void explain() {
    print('''
    函数组合：
    
    // 1. 函数组合概念
    // 将多个函数组合成一个函数
    
    // 2. 手动组合
    int addOne(int x) => x + 1;
    int double(int x) => x * 2;
    
    int addOneThenDouble(int x) => double(addOne(x));
    print(addOneThenDouble(5)); // 12
    
    // 3. 组合函数
    T Function(T) compose<T>(T Function(T) f, T Function(T) g) {
      return (T x) => f(g(x));
    }
    
    final addOneThenDouble = compose(double, addOne);
    print(addOneThenDouble(5)); // 12
    
    // 4. 管道操作
    T pipe<T>(T value, List<T Function(T)> functions) {
      return functions.fold(value, (v, f) => f(v));
    }
    
    final result = pipe(5, [
      addOne,
      double,
      (x) => x - 3,
    ]);
    print(result); // 9
    
    // 5. 函数组合最佳实践
    // - 保持函数纯粹
    // - 避免副作用
    // - 使用有意义的命名
    // - 保持函数简短
    ''');
  }
}
```
