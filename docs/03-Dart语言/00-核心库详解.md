### 2. 数字处理

```dart
// 数字处理
class NumberProcessing {
  void explain() {
    print('''
    数字处理：
    
    // 1. 数字类型
    int integer = 42;
    double decimal = 3.14;
    num number = 42.5;
    
    // 2. 数字转换
    int fromDouble = 3.14.toInt();  // 3
    double fromInt = 42.toDouble();  // 42.0
    int fromString = int.parse('42');  // 42
    double fromString = double.parse('3.14');  // 3.14
    
    // 3. 数字格式化
    String formatted = 3.14159.toStringAsFixed(2);  // 3.14
    String scientific = 123456789.toStringAsExponential(2);  // 1.23e+8
    
    // 4. 数学运算
    double abs = (-42).abs();  // 42
    double floor = 3.7.floorToDouble();  // 3.0
    double ceil = 3.2.ceilToDouble();  // 4.0
    double round = 3.5.roundToDouble();  // 4.0
    
    // 5. 数字比较
    bool isEqual = 42 == 42;  // true
    int comparison = 42.compareTo(100);  // -1
    int max = 42.clamp(0, 100);  // 42
    ''');
  }
}
```

## 🔧 字符串处理

### 1. 字符串操作

```dart
// 字符串操作
class StringOperations {
  void explain() {
    print('''
    字符串操作：
    
    // 1. 字符串创建
    String single = 'Hello';
    String double = "World";
    String multi = '''
    多行
    字符串
    ''';
    
    // 2. 字符串拼接
    String concat = 'Hello' + ' ' + 'World';
    String interpolate = 'Hello $name';
    String expression = 'Result: ${1 + 2}';
    
    // 3. 字符串方法
    String upper = 'hello'.toUpperCase();  // HELLO
    String lower = 'HELLO'.toLowerCase();  // hello
    String trim = '  Hello  '.trim();  // Hello
    String pad = '42'.padLeft(5, '0');  // 00042
    
    // 4. 字符串查找
    bool contains = 'Hello World'.contains('World');  // true
    int index = 'Hello World'.indexOf('World');  // 6
    bool starts = 'Hello World'.startsWith('Hello');  // true
    bool ends = 'Hello World'.endsWith('World');  // true
    
    // 5. 字符串分割
    List<String> split = 'a,b,c'.split(',');  // ['a', 'b', 'c']
    String substring = 'Hello World'.substring(0, 5);  // Hello
    String replace = 'Hello World'.replaceFirst('World', 'Dart');  // Hello Dart
    ''');
  }
}
```

### 2. 正则表达式

```dart
// 正则表达式
class RegularExpressions {
  void explain() {
    print('''
    正则表达式：
    
    // 1. 创建正则表达式
    final emailRegex = RegExp(r'^[\w-\.]+@[\w-]+\.[a-zA-Z]{2,}$');
    final phoneRegex = RegExp(r'^\d{3}-\d{3}-\d{4}$');
    
    // 2. 匹配检查
    bool hasMatch = emailRegex.hasMatch('test@example.com');  // true
    Match? match = emailRegex.firstMatch('test@example.com');
    
    // 3. 查找所有匹配
    Iterable<Match> matches = RegExp(r'\d+').allMatches('123 abc 456');
    for (final match in matches) {
      print(match.group(0));  // 123, 456
    }
    
    // 4. 替换
    String replaced = 'Hello 123 World 456'.replaceAll(RegExp(r'\d+'), '#');
    // Hello # World #
    
    // 5. 分割
    List<String> parts = 'a1b2c3'.split(RegExp(r'\d+'));
    // ['a', 'b', 'c']
    
    // 6. 常用正则表达式
    // 邮箱: ^[\w-\.]+@[\w-]+\.[a-zA-Z]{2,}$
    // 手机号: ^1[3-9]\d{9}$
    // 身份证: ^\d{17}[\dXx]$
    // URL: ^https?://[\w\-]+(\.[\w\-]+)+[/#?]?.*$
    ''');
  }
}
```
