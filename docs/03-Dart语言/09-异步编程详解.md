# 异步编程详解

> 掌握 Dart 异步编程的核心概念：Future、async/await、Stream。

## 📖 Future 基础

### 1. Future 概念

```dart
// Future 基础
class FutureBasics {
  void explain() {
    print('''
    Future 基础：
    
    // Future 表示一个异步操作的结果
    Future<String> fetchData() async {
      // 模拟网络请求
      await Future.delayed(Duration(seconds: 2));
      return '数据加载完成';
    }
    
    // 使用 Future
    void loadData() {
      print('开始加载...');
      
      fetchData().then((data) {
        print(data);
      }).catchError((error) {
        print('错误: $error');
      }).whenComplete(() {
        print('完成');
      });
    }
    
    // Future 状态
    // - 未完成（Pending）
    // - 完成（Completed）- 成功或失败
    ''');
  }
}
```

### 2. 创建 Future

```dart
// 创建 Future
class CreatingFutures {
  void explain() {
    print('''
    创建 Future 的方式：
    
    // 1. Future.value - 立即返回值
    Future<int> getValue() {
      return Future.value(42);
    }
    
    // 2. Future.error - 返回错误
    Future<int> getError() {
      return Future.error('发生错误');
    }
    
    // 3. Future.delayed - 延迟执行
    Future<String> delayed() {
      return Future.delayed(
        Duration(seconds: 1),
        () => '延迟完成',
      );
    }
    
    // 4. Future.sync - 同步执行
    Future<int> sync() {
      return Future.sync(() => 100);
    }
    
    // 5. Future.microtask - 微任务
    Future<String> microtask() {
      return Future.microtask(() => '微任务完成');
    }
    ''');
  }
}
```

## 📖 async/await

### 1. 基本用法

```dart
// async/await
class AsyncAwait {
  void explain() {
    print('''
    async/await：
    
    // async 标记函数为异步
    // await 等待 Future 完成
    
    Future<String> fetchUser() async {
      await Future.delayed(Duration(seconds: 1));
      return '用户数据';
    }
    
    Future<String> fetchPosts() async {
      await Future.delayed(Duration(seconds: 1));
      return '文章数据';
    }
    
    // 使用 async/await
    Future<void> loadData() async {
      try {
        print('开始加载...');
        
        String user = await fetchUser();
        print(user);
        
        String posts = await fetchPosts();
        print(posts);
        
        print('加载完成');
      } catch (e) {
        print('错误: $e');
      } finally {
        print('清理资源');
      }
    }
    ''');
  }
}
```

### 2. 并行执行

```dart
// 并行执行
class ParallelExecution {
  void explain() {
    print('''
    并行执行：
    
    Future<int> task1() async {
      await Future.delayed(Duration(seconds: 2));
      return 1;
    }
    
    Future<int> task2() async {
      await Future.delayed(Duration(seconds: 1));
      return 2;
    }
    
    Future<int> task3() async {
      await Future.delayed(Duration(seconds: 3));
      return 3;
    }
    
    // 串行执行（总时间 6 秒）
    Future<void> sequential() async {
      int r1 = await task1();
      int r2 = await task2();
      int r3 = await task3();
      print('结果: $r1, $r2, $r3');
    }
    
    // 并行执行（总时间 3 秒）
    Future<void> parallel() async {
      List<int> results = await Future.wait([
        task1(),
        task2(),
        task3(),
      ]);
      print('结果: $results');
    }
    
    // 任意一个完成
    Future<void> any() async {
      int result = await Future.any([
        task1(),
        task2(),
        task3(),
      ]);
      print('第一个完成: $result');
    }
    ''');
  }
}
```

## 📖 错误处理

### 1. 异步错误处理

```dart
// 错误处理
class AsyncErrorHandling {
  void explain() {
    print('''
    异步错误处理：
    
    Future<int> riskyOperation() async {
      await Future.delayed(Duration(seconds: 1));
      throw Exception('操作失败');
    }
    
    // 方式 1: try-catch
    Future<void> method1() async {
      try {
        int result = await riskyOperation();
        print(result);
      } catch (e) {
        print('捕获错误: $e');
      }
    }
    
    // 方式 2: then-catchError
    Future<void> method2() {
      return riskyOperation()
          .then((value) => print(value))
          .catchError((e) => print('捕获错误: $e'));
    }
    
    // 自定义异常
    class NetworkException implements Exception {
      final String message;
      NetworkException(this.message);
    }
    
    Future<void> networkRequest() async {
      try {
        throw NetworkException('网络连接失败');
      } on NetworkException catch (e) {
        print('网络错误: ${e.message}');
      } catch (e) {
        print('其他错误: $e');
      }
    }
    ''');
  }
}
```

## 📖 Stream 基础

### 1. Stream 概念

```dart
// Stream 基础
class StreamBasics {
  void explain() {
    print('''
    Stream 基础：
    
    // Stream 是一系列异步事件
    // 可以持续发出多个值
    
    Stream<int> countStream() async* {
      for (int i = 1; i <= 5; i++) {
        await Future.delayed(Duration(seconds: 1));
        yield i;  // 发出值
      }
    }
    
    // 监听 Stream
    void listen() {
      countStream().listen(
        (data) => print('收到: $data'),
        onError: (e) => print('错误: $e'),
        onDone: () => print('完成'),
      );
    }
    
    // Stream 可以被多次监听
    // 每次监听都会从头开始
    ''');
  }
}
```

### 2. 创建 Stream

```dart
// 创建 Stream
class CreatingStreams {
  void explain() {
    print('''
    创建 Stream 的方式：
    
    // 1. Stream.fromIterable
    Stream<int> fromList() {
      return Stream.fromIterable([1, 2, 3, 4, 5]);
    }
    
    // 2. Stream.periodic
    Stream<int> periodic() {
      return Stream.periodic(
        Duration(seconds: 1),
        (count) => count,
      ).take(5);
    }
    
    // 3. StreamController
    StreamController<int> controller = StreamController<int>();
    
    void addData() {
      controller.add(1);
      controller.add(2);
      controller.add(3);
      controller.close();
    }
    
    // 4. async* 生成器
    Stream<int> generate() async* {
      int i = 0;
      while (i < 5) {
        yield i++;
      }
    }
    ''');
  }
}
```

### 3. Stream 转换

```dart
// Stream 转换
class StreamTransformations {
  void explain() {
    print('''
    Stream 转换：
    
    Stream<int> numbers = Stream.fromIterable([1, 2, 3, 4, 5]);
    
    // map - 转换每个值
    Stream<int> doubled = numbers.map((n) => n * 2);
    
    // where - 过滤
    Stream<int> evens = numbers.where((n) => n % 2 == 0);
    
    // take - 取前几个
    Stream<int> first3 = numbers.take(3);
    
    // skip - 跳过前几个
    Stream<int> skip2 = numbers.skip(2);
    
    // expand - 展开
    Stream<int> expanded = numbers.expand((n) => [n, n * 10]);
    
    // reduce - 合并
    Future<int> sum = numbers.reduce((a, b) => a + b);
    
    // toList - 转换为 List
    Future<List<int>> list = numbers.toList();
    ''');
  }
}
```
