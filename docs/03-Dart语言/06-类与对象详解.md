# 类与对象详解

> 掌握 Dart 面向对象编程的核心概念：类、对象、构造函数、属性和方法。

## 📖 类的定义

### 1. 基本类定义

```dart
// 类的定义
class ClassDefinition {
  void explain() {
    print('''
    类的定义：
    
    class Person {
      // 实例变量
      String name;
      int age;
      
      // 构造函数
      Person(this.name, this.age);
      
      // 方法
      void introduce() {
        print('我叫$name，今年$age岁');
      }
    }
    
    // 创建对象
    var person = Person('张三', 25);
    person.introduce();  // 我叫张三，今年25岁
    ''');
  }
}
```

### 2. 构造函数

```dart
// 构造函数类型
class Constructors {
  void explain() {
    print('''
    构造函数类型：
    
    class User {
      String name;
      int age;
      String email;
      
      // 默认构造函数
      User(this.name, this.age, this.email);
      
      // 命名构造函数
      User.guest()
          : name = '游客',
            age = 0,
            email = '';
      
      User.fromJson(Map<String, dynamic> json)
          : name = json['name'],
            age = json['age'],
            email = json['email'];
      
      // 重定向构造函数
      User.defaultUser() : this('默认用户', 18, 'default@example.com');
      
      // 常量构造函数
      const User.constant(this.name, this.age, this.email);
      
      // 工厂构造函数
      factory User.create(String type) {
        if (type == 'admin') {
          return User('管理员', 30, 'admin@example.com');
        }
        return User('普通用户', 18, 'user@example.com');
      }
    }
    ''');
  }
}
```

## 📖 Getter 和 Setter

### 1. 属性访问器

```dart
// Getter 和 Setter
class Accessors {
  void explain() {
    print('''
    Getter 和 Setter：
    
    class Rectangle {
      double width;
      double height;
      
      Rectangle(this.width, this.height);
      
      // Getter
      double get area => width * height;
      
      double get perimeter => 2 * (width + height);
      
      bool get isSquare => width == height;
      
      // Setter
      set setWidth(double value) {
        if (value > 0) {
          width = value;
        }
      }
      
      set setHeight(double value) {
        if (value > 0) {
          height = value;
        }
      }
    }
    
    // 使用
    var rect = Rectangle(10, 20);
    print(rect.area);       // 200
    print(rect.perimeter);  // 60
    print(rect.isSquare);   // false
    
    rect.setWidth = 15;
    print(rect.area);       // 300
    ''');
  }
}
```

## 📖 方法

### 1. 实例方法

```dart
// 实例方法
class Methods {
  void explain() {
    print('''
    实例方法：
    
    class Calculator {
      double _result = 0;
      
      // 普通方法
      void add(double value) {
        _result += value;
      }
      
      void subtract(double value) {
        _result -= value;
      }
      
      // 返回值方法
      double getResult() {
        return _result;
      }
      
      // 方法链
      Calculator chain() {
        return this;
      }
      
      // 链式调用示例
      Calculator addChain(double value) {
        _result += value;
        return this;
      }
      
      void reset() {
        _result = 0;
      }
    }
    
    // 链式调用
    var calc = Calculator();
    calc.addChain(10).addChain(20).addChain(30);
    print(calc.getResult());  // 60
    ''');
  }
}
```

}

## 📖 私有成员

### 1. 封装

```dart
// 私有成员和封装
class Encapsulation {
  void explain() {
    print('''
    私有成员：
    
    // 使用 _ 前缀定义私有成员
    class BankAccount {
      String _accountNumber;
      double _balance;
      
      BankAccount(this._accountNumber, this._balance);
      
      // 公共 Getter
      String get accountNumber => _accountNumber;
      double get balance => _balance;
      
      // 公共方法
      void deposit(double amount) {
        if (amount > 0) {
          _balance += amount;
        }
      }
      
      bool withdraw(double amount) {
        if (amount > 0 && amount <= _balance) {
          _balance -= amount;
          return true;
        }
        return false;
      }
      
      // 私有方法
      void _logTransaction(String type, double amount) {
        print('$type: $amount, 余额: $_balance');
      }
    }
    
    // 使用
    var account = BankAccount('123456', 1000);
    print(account.balance);  // 1000
    account.deposit(500);
    account.withdraw(200);
    // account._balance  // 无法访问私有成员
    ''');
  }
}
```

## 📖 toString 方法

### 1. 对象字符串表示

```dart
// toString 方法
class ToStringMethod {
  void explain() {
    print('''
    toString 方法：
    
    class Product {
      String name;
      double price;
      
      Product(this.name, this.price);
      
      @override
      String toString() {
        return 'Product{name: $name, price: $price}';
      }
      
      // 更详细的表示
      String toDetailedString() {
        return '''
        Product Details:
          Name: $name
          Price: \$${price.toStringAsFixed(2)}
        ''';
      }

    // 使用
    var product = Product('iPhone', 999.99);
    print(product);  // Product{name: iPhone, price: 999.99}
    print(product.toDetailedString());
    ''');
  }
}
```

## 📖 相等性比较

### 1. == 和 hashCode

```dart
// 相等性比较
class Equality {
  void explain() {
    print('''
    相等性比较：
    
    class Point {
      final int x;
      final int y;
      
      const Point(this.x, this.y);
      
      @override
      bool operator ==(Object other) {
        if (identical(this, other)) return true;
        return other is Point && other.x == x && other.y == y;
      }
      
      @override
      int get hashCode => x.hashCode ^ y.hashCode;
      
      @override
      String toString() => 'Point($x, $y)';
    }
    
    // 使用
    var p1 = Point(1, 2);
    var p2 = Point(1, 2);
    var p3 = Point(2, 3);
    
    print(p1 == p2);  // true
    print(p1 == p3);  // false
    print(p1.hashCode == p2.hashCode);  // true
    ''');
  }
}
```
