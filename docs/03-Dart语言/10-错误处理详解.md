# 错误处理详解

> 掌握 Dart 语言的异常处理机制，包括异常捕获、自定义异常和错误恢复。

## 📖 异常基础

### 1. 内置异常类型

```dart
// 内置异常类型
class BuiltinExceptions {
  void explain() {
    print('''
    Dart 内置异常类型：
    
    1. Exception - 通用异常
       throw Exception('发生错误');
    
    2. FormatException - 格式错误
       int.parse('abc');  // 抛出 FormatException
    
    3. IOException - IO 错误
       文件读写、网络操作相关
    
    4. TimeoutException - 超时错误
       Future.delayed(Duration(seconds: 1))
           .timeout(Duration(milliseconds: 100));
    
    5. ArgumentError - 参数错误
       void test(int x) {
         if (x < 0) throw ArgumentError('x 不能为负数');
       }
    
    6. StateError - 状态错误
       [].first;  // 抛出 StateError
    
    7. UnsupportedError - 不支持的操作
       throw UnsupportedError('此操作不支持');
    
    8. TypeError - 类型错误
       强制类型转换失败时抛出
    ''');
  }
}
```

## 📖 异常捕获

### 1. try-catch-finally

```dart
// try-catch-finally
class ExceptionHandling {
  void explain() {
    print('''
    try-catch-finally：
    
    // 基本用法
    try {
      int result = 10 ~/ 0;
    } catch (e) {
      print('错误: $e');
    }
    
    // 捕获特定异常
    try {
      int.parse('abc');
    } on FormatException catch (e) {
      print('格式错误: ${e.message}');
    } catch (e) {
      print('其他错误: $e');
    }
    
    // 获取堆栈信息
    try {
      throw Exception('测试错误');
    } catch (e, stackTrace) {
      print('错误: $e');
      print('堆栈: $stackTrace');
    }
    
    // finally 块
    try {
      print('执行操作');
      throw Exception('错误');
    } catch (e) {
      print('捕获错误');
    } finally {
      print('清理资源');  // 总是执行
    }
    
    // 多个异常类型
    try {
      // 可能抛出多种异常
    } on FormatException catch (e) {
      print('格式错误');
    } on IOException catch (e) {
      print('IO 错误');
    } catch (e) {
      print('未知错误');
    }
    ''');
  }
}
```

## 📖 自定义异常

### 1. 创建自定义异常

```dart
// 自定义异常
class CustomExceptions {
  void explain() {
    print('''
    自定义异常：
    
    // 基本自定义异常
    class AppException implements Exception {
      final String message;
      final int? code;
      
      AppException(this.message, {this.code});
      
      @override
      String toString() => 'AppException: $message (code: $code)';
    }
    
    // 特定异常类
    class NetworkException extends AppException {
      NetworkException(String message) : super(message, code: 1001);
    }
    
    class ValidationException extends AppException {
      final String field;
      
      ValidationException(this.field, String message)
          : super(message, code: 1002);
    }
    
    // 使用
    void validateAge(int age) {
      if (age < 0) {
        throw ValidationException('age', '年龄不能为负数');
      }
      if (age > 150) {
        throw ValidationException('age', '年龄不能超过150');
      }
    }
    
    // 捕获自定义异常
    try {
      validateAge(-1);
    } on ValidationException catch (e) {
      print('${e.field}: ${e.message}');
    }
    ''');
  }
}
```

## 📖 错误恢复

### 1. 重试机制

```dart
// 错误恢复
class ErrorRecovery {
  void explain() {
    print('''
    错误恢复策略：
    
    // 1. 重试机制
    Future<T> retry<T>(
      Future<T> Function() operation, {
      int maxRetries = 3,
      Duration delay = const Duration(seconds: 1),
    }) async {
      for (int i = 0; i < maxRetries; i++) {
        try {
          return await operation();
        } catch (e) {
          if (i == maxRetries - 1) rethrow;
          await Future.delayed(delay);
        }
      }
      throw Exception('重试失败');
    }
    
    // 使用
    Future<String> fetchData() async {
      return retry(() async {
        // 可能失败的操作
        return await http.get(url);
      });
    }
    
    // 2. 降级策略
    Future<String> getDataWithFallback() async {
      try {
        return await fetchFromNetwork();
      } catch (e) {
        return await fetchFromCache();
      }
    }
    
    // 3. 默认值
    Future<int> getCount() async {
      try {
        return await fetchCount();
      } catch (e) {
        return 0;  // 默认值
      }
    }
    ''');
  }
}
```

## 📖 Error vs Exception

### 1. 区别

```dart
// Error vs Exception
class ErrorVsException {
  void explain() {
    print('''
    Error vs Exception：
    
    // Exception - 可恢复的异常
    // 程序应该捕获并处理
    throw Exception('可恢复的错误');
    
    // Error - 不可恢复的错误
    // 程序不应该捕获
    throw StateError('状态错误');
    
    // 区别：
    // 1. Exception 用于预期的错误情况
    // 2. Error 用于程序错误（bug）
    
    // 示例
    void divide(int a, int b) {
      if (b == 0) {
        throw Exception('除数不能为零');  // 预期错误
      }
      return a ~/ b;
    }
    
    void accessList(List list, int index) {
      if (index < 0 || index >= list.length) {
        throw RangeError('索引超出范围');  // 程序错误
      }
      return list[index];
    }
    
    // 最佳实践：
    // - 捕获 Exception，不捕获 Error
    // - 使用自定义 Exception 类
    // - 提供有意义的错误信息
    ''');
  }
}
```
