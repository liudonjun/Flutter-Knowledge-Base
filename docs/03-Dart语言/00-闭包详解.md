# 闭包详解

> 深入理解 Dart 中的闭包概念。

## 📖 闭包基础

### 1. 什么是闭包

```dart
// 闭包概念
class ClosureConcept {
  void explain() {
    print('''
    闭包概念：
    
    // 1. 什么是闭包
    // - 可以访问外部变量的函数
    // - 即使外部函数已经返回
    // - 捕获并记住外部变量
    // - 形成一个封闭的环境
    
    // 2. 闭包的特点
    // - 访问外部变量：可以访问外部作用域的变量
    // - 持久化：即使外部函数返回，变量仍然存在
    // - 私有：变量对外不可见
    // - 灵活：可以动态创建
    
    // 3. 闭包的使用场景
    // - 回调函数：事件处理
    // - 函数工厂：动态创建函数
    // - 数据封装：封装私有数据
    // - 延迟执行：延迟执行代码
    
    // 4. 闭包的优势
    // - 数据封装：隐藏实现细节
    // - 灵活性：动态创建函数
    // - 代码复用：减少重复代码
    // - 函数式编程：支持函数式风格
    ''');
  }
}
```

### 2. 闭包示例

```dart
// 闭包示例
class ClosureExample {
  void explain() {
    print('''
    闭包示例：
    
    // 1. 基本闭包
    Function createCounter() {
      int count = 0;
      return () {
        count++;
        return count;
      };
    }
    
    final counter = createCounter();
    print(counter()); // 1
    print(counter()); // 2
    print(counter()); // 3
    
    // 2. 闭包捕获变量
    Function createAdder(int x) {
      return (int y) => x + y;
    }
    
    final add5 = createAdder(5);
    print(add5(3)); // 8
    print(add5(7)); // 12
    
    // 3. 多个闭包共享变量
    List<Function> createCounters() {
      int count = 0;
      return [
        () => ++count,
        () => --count,
        () => count,
      ];
    }
    
    final counters = createCounters();
    print(counters[0]()); // 1
    print(counters[0]()); // 2
    print(counters[1]()); // 1
    print(counters[2]()); // 1
    
    // 4. 闭包作为参数
    void processNumbers(List<int> numbers, int Function(int) transformer) {
      for (final number in numbers) {
        print(transformer(number));
      }
    }
    
    final multiplier = 3;
    processNumbers([1, 2, 3], (n) => n * multiplier);
    ''');
  }
}
```

## 🔧 闭包应用

### 1. 数据封装

```dart
// 数据封装
class DataEncapsulation {
  void explain() {
    print('''
    数据封装：
    
    // 1. 私有数据封装
    Function createPerson(String name, int age) {
      // 私有变量
      String _name = name;
      int _age = age;
      
      // 返回访问器
      return (String action) {
        switch (action) {
          case 'getName':
            return _name;
          case 'getAge':
            return _age;
          case 'setName':
            return (String newName) => _name = newName;
          case 'setAge':
            return (int newAge) => _age = newAge;
          default:
            return null;
        }
      };
    }
    
    final person = createPerson('John', 30);
    print(person('getName')); // John
    print(person('getAge')); // 30
    
    // 2. 模块模式
    Function createModule() {
      // 私有变量
      Map<String, dynamic> _data = {};
      
      // 返回公共接口
      return {
        'set': (String key, dynamic value) => _data[key] = value,
        'get': (String key) => _data[key],
        'remove': (String key) => _data.remove(key),
        'getAll': () => Map<String, dynamic>.from(_data),
      };
    }
    
    final module = createModule();
    module['set']('name', 'John');
    print(module['get']('name')); // John
    ''');
  }
}
```

### 2. 函数工厂

```dart
// 函数工厂
class FunctionFactory {
  void explain() {
    print('''
    函数工厂：
    
    // 1. 创建乘法器
    Function createMultiplier(int factor) {
      return (int value) => value * factor;
    }
    
    final double = createMultiplier(2);
    final triple = createMultiplier(3);
    print(double(5)); // 10
    print(triple(5)); // 15
    
    // 2. 创建验证器
    Function createValidator({
      int? minLength,
      int? maxLength,
      Pattern? pattern,
    }) {
      return (String value) {
        if (minLength != null && value.length < minLength) {
          return false;
        }
        if (maxLength != null && value.length > maxLength) {
          return false;
        }
        if (pattern != null && !RegExp(pattern.toString()).hasMatch(value)) {
          return false;
        }
        return true;
      };
    }
    
    final emailValidator = createValidator(
      pattern: r'^[a-zA-Z0-9.]+@[a-zA-Z0-9]+\.[a-zA-Z]+',
    );
    print(emailValidator('test@example.com')); // true
    print(emailValidator('invalid')); // false
    
    // 3. 创建格式化器
    Function createFormatter(String template) {
      return (Map<String, dynamic> data) {
        String result = template;
        data.forEach((key, value) {
          result = result.replaceAll('{$key}', value.toString());
        });
        return result;
      };
    }
    
    final greetingFormatter = createFormatter('Hello, {name}! You are {age} years old.');
    print(greetingFormatter({'name': 'John', 'age': 30}));
    ''');
  }
}
```
