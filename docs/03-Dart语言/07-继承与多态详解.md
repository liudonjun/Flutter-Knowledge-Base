## 📖 继承基础

### 1. 类的继承

```dart
// 继承基础
class InheritanceBasics {
  void explain() {
    print('''
    类的继承：
    
    // 父类（基类）
    class Animal {
      String name;
      int age;
      
      Animal(this.name, this.age);
      
      void eat() {
        print('$name 正在吃东西');
      }
      
      void sleep() {
        print('$name 正在睡觉');
      }
    }
    
    // 子类（派生类）
    class Dog extends Animal {
      String breed;
      
      Dog(String name, int age, this.breed) : super(name, age);
      
      void bark() {
        print('$name 正在汪汪叫');
      }
    }
    
    class Cat extends Animal {
      String color;
      
      Cat(String name, int age, this.color) : super(name, age);
      
      void meow() {
        print('$name 正在喵喵叫');
      }
    }
    
    // 使用继承
    var dog = Dog('旺财', 3, '哈士奇');
    dog.eat();   // 继承自父类
    dog.bark();  // 子类特有方法
    
    var cat = Cat('咪咪', 2, '橘色');
    cat.eat();   // 继承自父类
    cat.meow();  // 子类特有方法
    ''');
  }
}
```

```dart
// super 关键字
class SuperKeyword {
  void explain() {
    print('''
    super 关键字：
    
    class Vehicle {
      String brand;
      int year;
      
      Vehicle(this.brand, this.year);
      
      void start() {
        print('$brand $year 启动');
      }
      
      void stop() {
        print('$brand $year 停止');
      }
    }

      int doors;
      
      Car(String brand, int year, this.doors) : super(brand, year);
      
      @override
      void start() {
        super.start();  // 调用父类方法
        print('汽车启动完成');
      }
      
      void openDoor() {
        print('打开 $doors 门');
      }
    }
    
    // 使用
    var car = Car('Toyota', 2023, 4);
    car.start();  // 调用重写的方法
    car.stop();   // 继承父类方法
    ''');
  }
}
```

## 📖 方法重写

### 1. @override 注解

```dart
// 方法重写
class MethodOverriding {
  void explain() {
    print('''
    方法重写：
    
    class Shape {
      String color;
      
      Shape(this.color);
      
      double get area => 0;
      
      void draw() {
        print('绘制 $color 形状');
      }
      
      @override
      String toString() => 'Shape(color: $color)';
    }
    
    class Circle extends Shape {
      double radius;
      
      Circle(String color, this.radius) : super(color);
      
      @override
      double get area => 3.14159 * radius * radius;
      
      @override
      void draw() {
        super.draw();  // 可选：调用父类实现
        print('绘制圆形，半径: $radius');
      }
      
      @override
      String toString() => 'Circle(color: $color, radius: $radius)';
    }
    
    class Rectangle extends Shape {
      double width;
      double height;
      
      Rectangle(String color, this.width, this.height) : super(color);
      
      @override
      double get area => width * height;
      
      @override
      void draw() {
        print('绘制矩形，${width}x$height');
      }
      
      @override
      String toString() => 'Rectangle(color: $color, ${width}x$height)';
    }
    ''');
  }
}
```

## 📖 多态

### 1. 多态概念

```dart
// 多态
class Polymorphism {
  void explain() {
    print('''
    多态：
    
    // 多态：同一个方法在不同对象上有不同的表现
    class Animal {
      String name;
      
      Animal(this.name);
      
      void makeSound() {
        print('$name 发出声音');
      }
    }
    
    class Dog extends Animal {
      Dog(String name) : super(name);
      
      @override
      void makeSound() {
        print('$name 汪汪汪');
      }
    }
    
    class Cat extends Animal {
      Cat(String name) : super(name);
      
      @override
      void makeSound() {
        print('$name 喵喵喵');
      }
    }
    
    class Duck extends Animal {
      Duck(String name) : super(name);
      
      @override
      void makeSound() {
        print('$name 嘎嘎嘎');
      }
    }
    
    // 多态使用
    List<Animal> animals = [
      Dog('旺财'),
      Cat('咪咪'),
      Duck('唐老鸭'),
    ];
    
    for (var animal in animals) {
      animal.makeSound();  // 每个动物发出不同的声音
    }
    
    // 输出：
    // 旺财 汪汪汪
    // 咪咪 喵喵喵
    // 唐老鸭 嘎嘎嘎
    ''');
  }
}
```

### 2. 类型检查和转换

```dart
// 类型检查和转换
class TypeCheckAndCast {
  void explain() {
    print('''
    类型检查和转换：
    
    class Animal {
      String name;
      Animal(this.name);
    }
    
    class Dog extends Animal {
      Dog(String name) : super(name);
      void bark() => print('汪汪');
    }
    
    class Cat extends Animal {
      Cat(String name) : super(name);
      void meow() => print('喵喵');
    }
    
    void handleAnimal(Animal animal) {
      // 类型检查
      if (animal is Dog) {
        // 自动类型转换
        animal.bark();
      } else if (animal is Cat) {
        animal.meow();
      }
      
      // 显式类型转换
      if (animal is Dog) {
        Dog dog = animal as Dog;
        dog.bark();
      }
    }
    
    // 使用
    var dog = Dog('旺财');
    var cat = Cat('咪咪');
    
    handleAnimal(dog);  // 汪汪
    handleAnimal(cat);  // 喵喵
    ''');
  }
}
```

## 📖 抽象类

### 1. 抽象类和抽象方法

```dart
// 抽象类
class AbstractClasses {
  void explain() {
    print('''
    抽象类：
    
    // 抽象类不能实例化
    abstract class Shape {
      String color;
      
      Shape(this.color);
      
      // 抽象方法：没有实现
      double get area;
      double get perimeter;
      
      // 普通方法：有实现
      void describe() {
        print('这是一个$color的形状');
        print('面积: $area');
        print('周长: $perimeter');
      }
    }
    
    // 子类必须实现抽象方法
    class Circle extends Shape {
      double radius;
      
      Circle(String color, this.radius) : super(color);
      
      @override
      double get area => 3.14159 * radius * radius;
      
      @override
      double get perimeter => 2 * 3.14159 * radius;
    }
    
    class Rectangle extends Shape {
      double width;
      double height;
      
      Rectangle(String color, this.width, this.height) : super(color);
      
      @override
      double get area => width * height;
      
      @override
      double get perimeter => 2 * (width + height);
    }
    
    // 使用
    // var shape = Shape('红色');  // 错误：不能实例化抽象类
    var circle = Circle('红色', 5);
    circle.describe();
    ''');
  }
}
```

## 📖 Mixin

### 1. Mixin 基础

```dart
// Mixin
class MixinBasics {
  void explain() {
    print('''
    Mixin：
    
    // Mixin 定义
    mixin Logger {
      void log(String message) {
        print('[LOG] $message');
      }
    }
    
    mixin Validator {
      bool isEmail(String email) {
        return email.contains('@');
      }
      
      bool isPhone(String phone) {
        return phone.length == 11;
      }
    }
    
    // 使用 mixin
    class UserService with Logger, Validator {
      void createUser(String email) {
        if (!isEmail(email)) {
          log('邮箱格式错误');
          return;
        }
        log('创建用户: $email');
      }
    }
    
    // 多个 mixin
    class AuthService with Logger, Validator {
      void login(String phone) {
        if (!isPhone(phone)) {
          log('手机号格式错误');
          return;
        }
        log('用户登录: $phone');
      }
    }
    
    // 使用
    var userService = UserService();
    userService.createUser('test@example.com');
    ''');
  }
}
```

```dart
// Mixin 限制
class MixinConstraints {
  void explain() {
    print('''
    Mixin 限制：
    
    // 使用 on 限制 mixin 的使用
    abstract class Animal {
      String name;
      Animal(this.name);
    }
    
    mixin Flyable on Animal {
      void fly() {
        print('$name 正在飞');
      }
    }
    
    mixin Swimmable on Animal {
      void swim() {
        print('$name 正在游');
      }
    }
    
    // 必须继承 Animal 才能使用这些 mixin
    class Bird extends Animal with Flyable {
      Bird(String name) : super(name);
    }
    
    class Fish extends Animal with Swimmable {
      Fish(String name) : super(name);
    }
    
    class Duck extends Animal with Flyable, Swimmable {
      Duck(String name) : super(name);
    }
    
    // 使用
    var duck = Duck('唐老鸭');
    duck.fly();
    duck.swim();
    ''');
  }
}
```
