```dart
// 手动依赖注入
class ManualDependencyInjection {
  /*
  手动依赖注入：
  
  1. 构造函数注入
     - 最常用方式
     - 不可变依赖
     - 清晰的依赖关系
  
  2. 工厂模式
     - 集中创建对象
     - 配置管理
     - 生命周期控制
  
  3. 服务定位器
     - 全局访问
     - 延迟加载
     - 简单易用
  */
  
  void explain() {
    print('''
    手动依赖注入：
    
    1. 构造函数注入
       class UserRepository {
         final ApiClient api;
         final Database db;
         
         UserRepository({required this.api, required this.db});
         
         Future<User> getUser(int id) async {
           final data = await api.get('/users/$id');
           return User.fromJson(data);
         }
       }
       
       class UserService {
         final UserRepository repository;
         
         UserService({required this.repository});
         
         Future<User> getUser(int id) {
           return repository.getUser(id);
         }
       }
    
    2. 工厂模式
       class ServiceLocator {
         static final ServiceLocator _instance = ServiceLocator._internal();
         factory ServiceLocator() => _instance;
         ServiceLocator._internal();
         
         final Map<String, dynamic> _services = {};
         
         void register<T>(T service) {
           _services[T.toString()] = service;
         }
         
         T get<T>() {
           return _services[T.toString()] as T;
         }
       }
       
       // 使用
       final locator = ServiceLocator();
       locator.register(ApiClient());
       locator.register(UserRepository(api: locator.get<ApiClient>()));
    
    3. 依赖注入容器
       class DIContainer {
         final Map<Type, dynamic Function()> _factories = {};
         final Map<Type, dynamic> _instances = {};
         
         void register<T>(T Function() factory, {bool singleton = true}) {
           _factories[T] = factory;
         }
         
         T get<T>() {
           // 检查单例
           if (_instances.containsKey(T)) {
             return _instances[T] as T;
           }
           
           // 创建实例
           final factory = _factories[T];
           if (factory == null) {
             throw Exception('No factory registered for $T');
           }
           
           final instance = factory() as T;
           _instances[T] = instance;
           
           return instance;
         }
       }
    ''');
  }
}
```

## 📖 状态提升

### 1. 状态提升概述

```dart
// 状态提升概述
class StateLiftingOverview {
  /*
  状态提升 (State Lifting) 是 Flutter 的核心概念：
  
  1. 核心原则
     - 单一数据源：状态存储在共同父级
     - 向下传递：通过构造函数传递状态
     - 向上回调：通过回调函数更新状态
  
  2. 优势
     - 可预测性：状态变化可追踪
     - 可测试性：易于单元测试
     - 可复用性：组件更通用
     - 数据一致性：避免状态不一致
  
  3. 使用场景
     - 兄弟组件通信
     - 表单状态管理
     - 全局状态共享
  */
  
  void explain() {
    print('''
    状态提升概述：
    
    1. 核心原则
       - 单一数据源：状态存储在共同父级
       - 向下传递：通过构造函数传递状态
       - 向上回调：通过回调函数更新状态
       - 不可变状态：状态不可直接修改
    
    2. 优势
       - 可预测性：状态变化可追踪
       - 可测试性：易于单元测试
       - 可复用性：组件更通用
       - 数据一致性：避免状态不一致
       - 调试友好：易于调试
    
    3. 使用场景
       - 兄弟组件通信：共享状态
       - 表单状态管理：表单数据
       - 全局状态共享：用户信息
       - 列表选择：多选列表
    
    4. 示例
       // 没有状态提升
       class Counter extends StatefulWidget {
         int _count = 0; // 状态在子组件
         
         void increment() {
           setState(() => _count++);
         }
       }
       
       // 使用状态提升
       class ParentWidget extends StatefulWidget {
         @override
         _ParentWidgetState createState() => _ParentWidgetState();
       }
       
       class _ParentWidgetState extends State<ParentWidget> {
         int _count = 0; // 状态在父组件
         
         void _increment() {
           setState(() => _count++);
         }
         
         @override
         Widget build(BuildContext context) {
           return CounterDisplay(
             count: _count, // 向下传递
             onIncrement: _increment, // 向上回调
           );
         }
       }
    ''');
  }
}
```

### 3. 状态提升最佳实践

```dart
// 状态提升最佳实践
class StateLiftingBestPractices {
  /*
  状态提升最佳实践：
  
  1. 最小化状态
     - 只提升必要的状态
     - 避免过度提升
     - 保持组件简单
  
  2. 清晰的接口
     - 定义清晰的回调
     - 使用类型安全
     - 文档化接口
  
  3. 性能优化
     - 使用 const
     - 避免不必要的重建
     - 使用 Selector
  */
  
  void explain() {
    print('''
    状态提升最佳实践：
    
    1. 最小化状态
       // 好的做法：只提升必要状态
       class ParentWidget extends StatefulWidget {
         @override
         _ParentWidgetState createState() => _ParentWidgetState();
       }
       
       class _ParentWidgetState extends State<ParentWidget> {
         // 只提升共享状态
         int _sharedValue = 0;
         
         @override
         Widget build(BuildContext context) {
           return Row(
             children: [
               // 子组件有自己的内部状态
               CounterButton(
                 value: _sharedValue,
                 onChanged: (value) => setState(() => _sharedValue = value),
               ),
               // 显示组件
               ValueDisplay(value: _sharedValue),
             ],
           );
         }
       }
    
    2. 清晰的接口
       // 定义清晰的回调类型
       typedef ValueChanged<T> = void Function(T value);
       
       class CounterButton extends StatelessWidget {
         final int value;
         final ValueChanged<int> onChanged;
         
         const CounterButton({
           Key? key,
           required this.value,
           required this.onChanged,
         }) : super(key: key);
         
         @override
         Widget build(BuildContext context) {
           return ElevatedButton(
             onPressed: () => onChanged(value + 1),
             child: Text('$value'),
           );
         }
       }
    
    3. 性能优化
       class OptimizedParent extends StatefulWidget {
         @override
         _OptimizedParentState createState() => _OptimizedParentState();
       }
       
       class _OptimizedParentState extends State<OptimizedParent> {
         int _count = 0;
         
         @override
         Widget build(BuildContext context) {
           return Column(
             children: [
               // 使用 const 避免重建
               const TitleWidget(),
               
               // 只有需要重建的组件
               CounterDisplay(count: _count),
               
               // 使用 child 参数
               CounterButton(
                 value: _count,
                 onChanged: (value) => setState(() => _count = value),
```

## 📖 总结

### 核心概念总结

| 概念 | 描述 | 使用场景 |
|------|------|----------|
| **依赖注入** | 通过外部注入依赖 | 服务层、Repository |
| **状态提升** | 将状态提升到共同父级 | UI 状态、组件通信 |
| **控制反转** | 控制权交给外部 | 解耦、可测试性 |
| **单一数据源** | 状态存储在一个地方 | 状态一致性 |

### 最佳实践

1. **依赖注入**：用于服务层，提高可测试性
2. **状态提升**：用于 UI 状态，保持数据一致性
3. **结合使用**：根据场景选择合适的方案
4. **最小化状态**：只提升必要的状态
5. **清晰接口**：定义清晰的回调和参数
