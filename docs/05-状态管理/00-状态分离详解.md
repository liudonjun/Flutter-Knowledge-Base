### 2. 状态分离实现

```dart
// 状态分离实现
class StateSeparationImplementation {
  void explain() {
    print('''
    状态分离实现：
    
    // 1. UI 状态
    class UIState {
      final bool isLoading;
      final String? errorMessage;
      
      const UIState({
        this.isLoading = false,
        this.errorMessage,
      });
    }
    
    // 2. 业务状态
    class BusinessState {
      final List<Item> items;
      final int selectedIndex;
      
      const BusinessState({
        required this.items,
        required this.selectedIndex,
      });
    }
    
    // 3. 使用 BLoC 分离
    class MyBloc extends Bloc<MyEvent, MyState> {
      MyBloc() : super(MyState()) {
        on<LoadData>(_onLoadData);
        on<SelectItem>(_onSelectItem);
      }
    }
    
    // 4. 使用 MVVM 分离
    class MyViewModel extends ChangeNotifier {
      UIState _uiState = UIState();
      BusinessState _businessState = BusinessState(
        items: [],
        selectedIndex: 0,
      );
    }
    ''');
  }
}
```
