### 2. 初始化成本优化

```dart
// 初始化成本优化
class InitializationCostOptimization {
  void explain() {
    print('''
    初始化成本优化：
    
    // 1. 延迟初始化
    late final ApiService _apiService;
    
    void init() {
      _apiService = ApiService();
    }
    
    // 2. 按需加载
    class LazyProvider<T> {
      T? _value;
      final T Function() _factory;
      
      LazyProvider(this._factory);
      
      T get value {
        _value ??= _factory();
        return _value!;
      }
    }
    
    // 3. 简化配置
    // - 减少配置项
    // - 使用默认值
    // - 延迟配置
    // - 按需配置
    
    // 4. 初始化成本最佳实践
    // - 延迟初始化
    // - 按需加载
    // - 简化配置
    // - 监控启动时间
    ''');
  }
}
```
