### 2. 状态不可变性实现

```dart
// 状态不可变性实现
class StateImmutabilityImplementation {
  void explain() {
    print('''
    状态不可变性实现：
    
    // 1. 使用 final
    class AppState {
      final int count;
      final String name;
      
      const AppState({
        required this.count,
        required this.name,
      });
    }
    
    // 2. 使用 copyWith
    class AppState {
      final int count;
      final String name;
      
      const AppState({
        required this.count,
        required this.name,
      });
      
      AppState copyWith({
        int? count,
        String? name,
      }) {
        return AppState(
          count: count ?? this.count,
          name: name ?? this.name,
        );
      }
    }
    
    // 3. 更新状态
    final newState = state.copyWith(count: state.count + 1);
    
    // 4. 使用 immutable 包
    @immutable
    class AppState {
      final int count;
      final String name;
      
      const AppState({
        required this.count,
        required this.name,
      });
    }
    ''');
  }
}
```
