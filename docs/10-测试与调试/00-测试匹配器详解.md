### 2. 测试匹配器实现

```dart
// 测试匹配器实现
class TestMatchersImplementation {
  void explain() {
    print('''
    测试匹配器实现：
    
    // 1. 基本匹配器
    expect(1 + 1, equals(2));
    expect('hello', isNotEmpty);
    expect([1, 2, 3], contains(2));
    expect(value, isNotNull);
    
    // 2. 比较匹配器
    expect(10, greaterThan(5));
    expect(5, lessThan(10));
    expect(5, inInclusiveRange(1, 10));
    
    // 3. 类型匹配器
    expect(value, isA<int>());
    expect(value, isA<String>());
    expect(value, isA<List>());
    
    // 4. 集合匹配器
    expect([1, 2, 3], contains(2));
    expect([1, 2, 3], hasLength(3));
    expect([1, 2, 3], everyElement(greaterThan(0)));
    
    // 5. 自定义匹配器
    Matcher isEvenNumber = predicate<int>((n) => n % 2 == 0, 'is even');
    expect(4, isEvenNumber);
    
    // 6. 测试匹配器最佳实践
    // - 使用有意义的匹配器
    // - 组合匹配器
    // - 自定义匹配器
    // - 清晰的错误信息
    ''');
  }
}
```
