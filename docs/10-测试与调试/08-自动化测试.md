// Mock 测试
class MockTesting {
  /*
  Mock 测试：
  
  1. 创建 Mock
     - 使用 @GenerateMocks
     - 运行 build_runner
     - 导入生成的文件
  
  2. 使用 Mock
     - 设置期望
     - 调用方法
     - 验证调用
  
  3. 最佳实践
     - 只 Mock 直接依赖
     - 验证重要交互
     - 保持测试简单
  */
  
  void explain() {
    print('''
    Mock 测试：
    
    1. 定义接口
       abstract class UserRepository {
         Future<User> getUser(int id);
         Future<void> saveUser(User user);
       }
       
       class UserService {
         final UserRepository repository;
         
         UserService(this.repository);
         
         Future<String> getUserName(int id) async {
           final user = await repository.getUser(id);
           return user.name;
         }
       }
    
    2. 生成 Mock
       // user_service_test.dart
       import 'package:mockito/annotations.dart';
       import 'package:mockito/mockito.dart';
       
       @GenerateMocks([UserRepository])
       import 'user_service_test.mocks.dart';
       
       // 运行生成命令
       // flutter pub run build_runner build
    
    3. 编写测试
       void main() {
         late MockUserRepository mockRepository;
         late UserService userService;
         
         setUp(() {
           mockRepository = MockUserRepository();
           userService = UserService(mockRepository);
         });
         
         test('获取用户名', () async {
           // 设置 Mock 行为
           when(mockRepository.getUser(1))
             .thenAnswer((_) async => User(id: 1, name: 'Alice'));
           
           // 调用方法
           final name = await userService.getUserName(1);
           
           // 验证结果
           expect(name, 'Alice');
           
           // 验证调用
           verify(mockRepository.getUser(1)).called(1);
         });
         
         test('处理错误', () async {
           // 设置 Mock 抛出异常
           when(mockRepository.getUser(1))
             .thenThrow(Exception('Not found'));
           
           // 验证异常
           expect(
             () => userService.getUserName(1),
             throwsException,
           );
         });
       }
    ''');
  }
}

class TestFinders {
  /*
  测试查找器：
  
  1. 基本查找

     - find.byType：按类型查找
     - find.byKey：按 Key 查找
  
  2. 组合查找
     - find.descendant：查找后代
     - find.ancestor：查找祖先
     - find.byPredicate：自定义查找
  
  3. 匹配器
     - findsOneWidget：找到一个
     - findsWidgets：找到多个
     - findsNothing：没找到
  */

    print('''
    测试查找器：
    
    1. 基本查找
       // 按文本查找
       expect(find.text('Hello'), findsOneWidget);
       expect(find.textContaining('He'), findsOneWidget);
       
       // 按类型查找
       expect(find.byType(Text), findsOneWidget);
       expect(find.byType(ElevatedButton), findsOneWidget);
       
       // 按 Key 查找
       expect(find.byKey(Key('my_key')), findsOneWidget);
       
       // 按图标查找
       expect(find.byIcon(Icons.star), findsOneWidget);
    
    2. 组合查找
       // 查找后代
       expect(
         find.descendant(
           of: find.byType(Row),
           matching: find.byType(Text),
         ),
         findsOneWidget,
       );
       
       // 查找祖先
       expect(
         find.ancestor(
           of: find.text('Hello'),
           matching: find.byType(Container),
         ),
         findsOneWidget,
       );
       
       // 自定义查找
       expect(
         find.byPredicate((widget) {
           return widget is Text && widget.data!.startsWith('H');
         }),
         findsOneWidget,
       );
    
    3. 匹配器
       // 找到一个
       expect(find.text('Hello'), findsOneWidget);
       
       // 找到多个
       expect(find.byType(Text), findsNWidgets(3));
       
       // 没找到
       expect(find.text('World'), findsNothing);
       
       // 至少一个
       expect(find.byType(Text), findsWidgets);
    ''');
  }
}

// Page Object 模式
class PageObjectPattern {
  /*
  Page Object 模式：

     - 复用测试代码

       // integration_test/pages/login_page.dart
       import 'package:flutter_test/flutter_test.dart';
       
       class LoginPage {
         final WidgetTester tester;
         
         LoginPage(this.tester);
         
         // 查找器
         Finder get usernameField => find.byKey(Key('username'));
         Finder get passwordField => find.byKey(Key('password'));
         Finder get loginButton => find.text('Login');
         
         // 操作
         Future<void> enterUsername(String username) async {
           await tester.enterText(usernameField, username);
         }
         
         Future<void> enterPassword(String password) async {
           await tester.enterText(passwordField, password);
         }
         
         Future<void> tapLogin() async {
           await tester.tap(loginButton);
           await tester.pumpAndSettle();
         }
         
         // 组合操作
         Future<void> login(String username, String password) async {
           await enterUsername(username);
           await enterPassword(password);
           await tapLogin();
         }
       }
    
    2. 使用 Page Object
       // integration_test/app_test.dart
       import 'pages/login_page.dart';
       import 'pages/home_page.dart';
       
       void main() {
         testWidgets('登录流程', (tester) async {
           app.main();
           await tester.pumpAndSettle();
           
           final loginPage = LoginPage(tester);
           await loginPage.login('testuser', 'password');
           
           final homePage = HomePage(tester);
           expect(await homePage.isLoggedIn, true);
         });
       }
    
    3. 最佳实践
       - 每个页面一个 Page 类
       - 封装所有操作
       - 提供验证方法
       - 保持 Page 类简单
    ''');

    4. 测试独立性
       // 每个测试独立
       setUp(() {
         // 初始化测试环境
       });
       
       tearDown(() {
         // 清理测试环境
       });
       
       // 避免测试间依赖
       test('测试A', () {});
       test('测试B', () {}); // 不依赖测试A
    ''');
  }

## 📖 总结

### 测试核心概念

| 测试类型 | 数量 | 速度 | 成本 | 范围 |
|----------|------|------|------|------|
| **单元测试** | 70-80% | 毫秒级 | 低 | 函数、类 |
| **Widget 测试** | 15-20% | 秒级 | 中 | Widget |
| **集成测试** | 5-10% | 分钟级 | 高 | 应用流程 |

### 最佳实践总结

1. **测试金字塔**：大量单元测试，适量 Widget 测试，少量集成测试
2. **测试覆盖**：覆盖关键逻辑、边界条件、错误处理
3. **测试组织**：清晰分组、独立测试、良好命名
4. **测试维护**：定期运行、更新过时测试、重构测试代码
