// Dart 日志 API
class DartLogging {
  /*
  Dart 日志 API：
  
  1. print 函数
     - 基础输出
     - 调试信息
     - 简单日志
  
  2. developer.log
     - 结构化日志
     - 支持名称和错误
     - 开发者工具集成
  
  3. debugPrint
     - 限制输出长度
     - 避免日志截断
     - 适合大文本
  */
  
  void explain() {
    print('''
    Dart 日志 API：
    
    1. print 函数
       // 基础输出
       print('Hello, Flutter!');
       
       // 输出变量
       int count = 42;
       print('计数: $count');
       
       // 输出对象
       User user = User(name: 'Alice', age: 25);
       print('用户: $user');
    
    2. developer.log
       import 'dart:developer';
       
       // 结构化日志
       log('这是一条日志');
       
       // 带名称的日志
       log('网络请求开始', name: 'network');
       
       // 带错误的日志
       try {
         // 可能出错的代码
       } catch (e, stackTrace) {
         log('发生错误', error: e, stackTrace: stackTrace);
       }
    
    3. debugPrint
       // 限制输出长度
       debugPrint('很长的文本...');
       
       // 适合大对象输出
       debugPrint('复杂对象: $complexObject');
       
       // 在 debug 模式下输出
       assert(() {
         debugPrint('仅在 debug 模式输出');
         return true;
       }());
    ''');
  }
}

         };

       enum LogLevel { debug, info, warning, error }

         final String name;
         final LogLevel minLevel;
         
         Logger(this.name, {this.minLevel = LogLevel.debug});
         
         void log(LogLevel level, String message) {
           if (level.index >= minLevel.index) {
             final prefix = '[${level.name.toUpperCase()}] [$name]';
             print('$prefix $message');
           }
         }
         
         void debug(String message) => log(LogLevel.debug, message);
         void info(String message) => log(LogLevel.info, message);
         void warning(String message) => log(LogLevel.warning, message);
         void error(String message) => log(LogLevel.error, message);

## 📖 高级日志系统

### 1. 结构化日志

```dart
// 结构化日志
class StructuredLogging {
  /*
  结构化日志：
  
  1. 日志模型
     - 级别
     - 时间戳
     - 消息
     - 上下文
  
  2. 日志格式化
     - JSON 格式
     - 自定义格式
     - 可读格式
  
  3. 日志输出
     - 控制台
     - 文件
     - 网络
  */
  
  void explain() {
    print('''
    结构化日志：
    
    1. 日志模型
       class LogEntry {
         final LogLevel level;
         final String message;
         final DateTime timestamp;
         final String? name;
         final dynamic error;
         final StackTrace? stackTrace;
         final Map<String, dynamic>? context;
         
         LogEntry({
           required this.level,
           required this.message,
           this.name,
           this.error,
           this.stackTrace,
           this.context,
         }) : timestamp = DateTime.now();
         
         Map<String, dynamic> toJson() => {
           'level': level.name,
           'message': message,
           'timestamp': timestamp.toIso8601String(),
           if (name != null) 'name': name,
           if (error != null) 'error': error.toString(),
           if (context != null) 'context': context,
         };
       }
    
    2. 日志格式化
       class LogFormatter {
         static String format(LogEntry entry) {
           final buffer = StringBuffer();
           buffer.write('[${entry.timestamp.toIso8601String()}]');
           buffer.write('[${entry.level.name.toUpperCase()}]');
           if (entry.name != null) buffer.write('[${entry.name}]');
           buffer.write(' ${entry.message}');
           
           if (entry.error != null) {
             buffer.write('\nError: ${entry.error}');
           }
           
           return buffer.toString();
         }
         
         static String formatJson(LogEntry entry) {
           return jsonEncode(entry.toJson());
         }
       }
    
    3. 日志输出器
       abstract class LogOutput {
         void output(LogEntry entry);
       }
       
       class ConsoleOutput implements LogOutput {
         @override
         void output(LogEntry entry) {
           print(LogFormatter.format(entry));
         }
       }
       
       class FileOutput implements LogOutput {
         final File _file;
         
         FileOutput(this._file);
         
         @override
         void output(LogEntry entry) {
           _file.writeAsStringSync(
             '${LogFormatter.formatJson(entry)}\n',
             mode: FileMode.append,
           );
         }
       }
    ''');
  }
}
```

### 2. 日志管理器

```dart
// 日志管理器
class LogManager {
  /*
  日志管理器：
  
  1. 单例模式
     - 全局访问
     - 配置管理
  
  2. 多输出支持
     - 控制台
     - 文件
     - 网络
  
  3. 过滤和路由
     - 按级别过滤
     - 按名称路由
  */
  
  void explain() {
    print('''
    日志管理器：
    
    1. 日志管理器实现
       class Logger {
         static final Logger _instance = Logger._internal();
         factory Logger() => _instance;
         Logger._internal();
         
         final List<LogOutput> _outputs = [];
         LogLevel _minLevel = LogLevel.debug;
         
         void addOutput(LogOutput output) {
           _outputs.add(output);
         }
         
         void setMinLevel(LogLevel level) {
           _minLevel = level;
         }
         
         void log(
           LogLevel level,
           String message, {
           String? name,
           dynamic error,
           StackTrace? stackTrace,
           Map<String, dynamic>? context,
         }) {
           if (level.index < _minLevel.index) return;
           
           final entry = LogEntry(
             level: level,
             message: message,
             name: name,
             error: error,
             stackTrace: stackTrace,
             context: context,
           );
           
           for (final output in _outputs) {
             output.output(entry);
           }
         }
         
         void debug(String message, {String? name}) {
           log(LogLevel.debug, message, name: name);
         }
         
         void info(String message, {String? name}) {
           log(LogLevel.info, message, name: name);
         }
         
         void error(String message, {dynamic error, StackTrace? stackTrace}) {
           log(LogLevel.error, message, error: error, stackTrace: stackTrace);
         }
       }
    
    2. 使用示例
       // 初始化
       void setupLogger() {
         final logger = Logger();
         logger.addOutput(ConsoleOutput());
         logger.addOutput(FileOutput(File('app.log')));
         logger.setMinLevel(LogLevel.info);
       }
       
       // 使用
       Logger().info('应用启动', name: 'app');
       Logger().error('网络错误', error: e, stackTrace: s);
    ''');
  }
}
```

class FormattedOutput {
  /*
  格式化输出：
  
  1. 表格输出
     - 对齐
     - 边框
     - 分隔符
  
  2. 树形输出
     - 缩进
     - 层级
     - 连接符
  
  3. JSON 输出
     - 美化
     - 压缩
  */

    print('''
    格式化输出：
    
    1. 表格输出
       void printTable(List<List<String>> data) {
         // 计算列宽
         final columnWidths = <int>[];
         for (var i = 0; i < data[0].length; i++) {
           final maxWidth = data.map((row) => row[i].length).reduce(max);
           columnWidths.add(maxWidth);
         }
         
         // 打印表格
         for (final row in data) {
           final buffer = StringBuffer();
           for (var i = 0; i < row.length; i++) {
             buffer.write(row[i].padRight(columnWidths[i] + 2));
           }
           print(buffer.toString());
         }
       }
    
    2. 树形输出
       void printTree(Map<String, dynamic> data, [String indent = '']) {
         data.forEach((key, value) {
           if (value is Map<String, dynamic>) {
             print('$indent├── $key:');
             printTree(value, '$indent│   ');
           } else {
             print('$indent├── $key: $value');
           }
         });
       }
    
    3. JSON 美化
       String prettyJson(dynamic data) {
         final encoder = JsonEncoder.withIndent('  ');
         return encoder.convert(data);
       }
       
       void printJson(dynamic data) {
         print(prettyJson(data));
       }
    ''');
  }
}

## 📖 总结

### 日志级别

| 级别 | 用途 | 生产环境 |
|------|------|----------|
| **DEBUG** | 调试信息 | 关闭 |
| **INFO** | 一般信息 | 开启 |
| **WARNING** | 警告信息 | 开启 |
| **ERROR** | 错误信息 | 开启 |

### 最佳实践

1. **使用结构化日志**：便于分析和搜索
2. **控制日志级别**：生产环境关闭 DEBUG
3. **条件日志**：使用 kDebugMode 判断
4. **多输出支持**：控制台、文件、网络
