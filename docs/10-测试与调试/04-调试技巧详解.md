# 调试技巧详解

> 掌握 Flutter 开发中的各种调试技巧。

## 📖 调试工具

### 1. Flutter Inspector

```dart
// Flutter Inspector
class FlutterInspector {
  void explain() {
    print('''
    Flutter Inspector：
    
    // 打开方式
    // VS Code: Ctrl+Shift+P -> Flutter Inspector
    // Android Studio: View -> Tool Windows -> Flutter Inspector
    
    // 功能
    1. Widget 树查看
    2. 布局调试
    3. 选择 Widget
    4. 查看属性
    
    // 调试标志
    void main() {
      debugPaintSizeEnabled = true;  // 显示布局边界
      debugRepaintRainbowEnabled = true;  // 显示重绘区域
      
      runApp(MyApp());
    }
    ''');
  }
}
```

### 2. 日志调试

```dart
// 日志调试
class LoggingDebug {
  void explain() {
    print('''
    日志调试：
    
    // 1. print
    print('调试信息');
    
    // 2. debugPrint（限制输出）
    debugPrint('长文本...');
    
    // 3. developer.log
    import 'dart:developer';
    log('消息', name: 'my.app');
    
    // 4. 条件调试
    void debugLog(String message) {
      if (kDebugMode) {
        print('[DEBUG] $message');
      }
    }
    
    // 5. 格式化输出
    void prettyPrint(Map<String, dynamic> data) {
      final encoder = JsonEncoder.withIndent('  ');
      print(encoder.convert(data));
    }
    ''');
  }
}
```

### 3. 断点调试

```dart
// 断点调试
class BreakpointDebug {
  void explain() {
    print('''
    断点调试：
    
    // 1. 普通断点
    // 点击行号左侧
    
    // 2. 条件断点
    // 右键断点 -> 输入条件
    
    // 3. 日志断点
    // 右键断点 -> Log message
    
    // 4. 代码断点
    debugger();  // 需要 import 'dart:developer';
    
    // 5. 条件代码断点
    if (error != null) {
      debugger();
    }
    
    // 调试快捷键
    // F5: 继续
    // F10: 单步跳过
    // F11: 单步进入
    // Shift+F11: 单步跳出
    ''');
  }
}
```

## 📖 性能调试

### 1. 性能面板

```dart
// 性能调试
class PerformanceDebug {
  void explain() {
    print('''
    性能调试：
    
    // 1. Performance Overlay
    MaterialApp(
      showPerformanceOverlay: true,
      home: MyHomePage(),
    )
    
    // 2. 检查帧率
    void checkFrameRate() {
      SchedulerBinding.instance.addTimingsCallback((timings) {
        for (final timing in timings) {
          final duration = timing.totalSpan.inMilliseconds;
          if (duration > 16) {
            print('慢帧: ${duration}ms');
          }
        }
      });
    }
    
    // 3. 内存检查
    void checkMemory() {
      // 使用 DevTools Memory 面板
      // 查看内存使用情况
      // 检测内存泄漏
    }
    
    // 4. 重建检查
    class DebugWidget extends StatelessWidget {
      @override
      Widget build(BuildContext context) {
        print('Widget 重建');
        return Container();
      }
    }
    ''');
  }
}
```

### 2. 布局调试

```dart
// 布局调试
class LayoutDebug {
  void explain() {
    print('''
    布局调试：
    
    // 1. 显示布局边界
    debugPaintSizeEnabled = true;
    
    // 2. 显示重绘区域
    debugRepaintRainbowEnabled = true;
    
    // 3. 打印布局
    debugPrintLayouts = true;

    LayoutBuilder(
      builder: (context, constraints) {
        print('约束: $constraints');
        return Container();
      },
    )
    
    // 5. 检查溢出
    // 使用 SingleChildScrollView
    // 使用 Expanded
    // 使用 Flexible
    ''');
  }
}
```
