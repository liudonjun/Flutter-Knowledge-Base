# 单元测试详解

> 掌握 Flutter 单元测试的编写方法。

## 📖 测试基础

### 1. 基本测试

```dart
// 单元测试基础
import 'package:flutter_test/flutter_test.dart';

class UnitTestBasics {
  void explain() {
    print('''
    单元测试基础：
    
    // 测试文件命名：*_test.dart
    // 运行测试：flutter test
    
    void main() {
      test('基本测试', () {
        expect(1 + 1, equals(2));
      });
      
      test('字符串测试', () {
        expect('hello'.toUpperCase(), equals('HELLO'));
      });
      
      test('列表测试', () {
        expect([1, 2, 3].length, equals(3));
      });
    }
    ''');
  }
}
```

### 2. 测试组

```dart
// 测试组
class TestGroups {
  void explain() {
    print('''
    测试组（group）：
    
    void main() {
      group('字符串测试', () {
        test('大写转换', () {
          expect('hello'.toUpperCase(), equals('HELLO'));
        });
        
        test('小写转换', () {
          expect('HELLO'.toLowerCase(), equals('hello'));
        });
        
        test('包含子串', () {
          expect('hello world'.contains('world'), isTrue);
        });
      });
      
      group('列表测试', () {
        test('添加元素', () {
          final list = [1, 2, 3];
          list.add(4);
          expect(list, equals([1, 2, 3, 4]));
        });
        
        test('删除元素', () {
          final list = [1, 2, 3];
          list.remove(2);
          expect(list, equals([1, 3]));
        });
      });
    }
    ''');
  }
}
```

### 3. 匹配器

```dart
// 匹配器
class Matchers {
  void explain() {
    print('''
    常用匹配器：
    
    // 相等
    expect(value, equals(expected));
    expect(value, expected);  // 简写
    
    // 布尔
    expect(value, isTrue);
    expect(value, isFalse);
    expect(value, isNull);
    expect(value, isNotNull);
    
    // 数值
    expect(value, isZero);
    expect(value, isPositive);
    expect(value, isNegative);
    expect(value, greaterThan(5));
    expect(value, lessThanOrEqualTo(10));
    expect(value, inInclusiveRange(1, 10));
    
    // 字符串
    expect('hello', contains('ell'));
    expect('hello', startsWith('hel'));
    expect('hello', endsWith('llo'));
    expect('hello', hasLength(5));
    
    // 集合
    expect(list, hasLength(3));
    expect(list, contains(2));
    expect(list, containsAll([1, 2, 3]));
    expect(list, isEmpty);
    expect(list, isNotEmpty);
    
    // 类型
    expect(value, isA<int>());
    expect(value, isA<String>());
    expect(value, isA<List>());
    
    // 异常
    expect(() => throw Exception(), throwsException);
    expect(() => throw FormatException(), throwsFormatException);
    expect(() => throw StateError(''), throwsStateError);
    ''');
  }
}
```

## 📖 测试类

### 1. Model 测试

```dart
// Model 测试
class ModelTest {
  void explain() {
    print('''
    Model 测试：
    
    class User {
      final String name;
      final int age;
      
      User({required this.name, required this.age});
      
      factory User.fromJson(Map<String, dynamic> json) {
        return User(
          name: json['name'] as String,
          age: json['age'] as int,
        );
      }
      
      Map<String, dynamic> toJson() {
        return {'name': name, 'age': age};
      }
    }
    
    void main() {
      group('User', () {
        test('fromJson', () {
          final user = User.fromJson({'name': 'Alice', 'age': 25});
          expect(user.name, equals('Alice'));
          expect(user.age, equals(25));
        });
        
        test('toJson', () {
          final user = User(name: 'Bob', age: 30);
          final json = user.toJson();
          expect(json['name'], equals('Bob'));
          expect(json['age'], equals(30));
        });
        
        test('JSON 序列化和反序列化', () {
          final user = User(name: 'Charlie', age: 35);
          final json = user.toJson();
          final restored = User.fromJson(json);
          
          expect(restored.name, equals(user.name));
          expect(restored.age, equals(user.age));
        });
      });
    }
    ''');
  }
}
```

### 2. 异步测试

```dart
// 异步测试
class AsyncTest {
  void explain() {
    print('''
    异步测试：
    
    void main() {
      test('Future 测试', () async {
        final result = await Future.delayed(
          Duration(milliseconds: 100),
          () => 'completed',
        );
        expect(result, equals('completed'));
      });
      
      test('Stream 测试', () async {
        final stream = Stream.fromIterable([1, 2, 3]);
        final list = await stream.toList();
        expect(list, equals([1, 2, 3]));
      });
      
      test('异步异常', () async {
        expect(
          () async => await Future.error(Exception('error')),
          throwsException,
        );
      });
      
      test('超时测试', () async {
        expect(
          () => Future.delayed(Duration(seconds: 10)),
          throwsA(isA<TimeoutException>()),
        );
      }, timeout: Timeout(Duration(seconds: 1)));
    }
    ''');
  }
}
```
