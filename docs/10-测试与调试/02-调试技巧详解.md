## 📊 日志调试

### 1. 基本日志
```dart
// 使用 print
print('调试信息: $variable');

// 使用 debugPrint
debugPrint('调试信息: $variable');

// 使用 log
import 'dart:developer';
log('日志信息', name: 'my_app');

// 使用 Logger
import 'package:logger/logger.dart';
final logger = Logger();
logger.d('调试信息');
logger.i('信息');
logger.w('警告');
logger.e('错误');
```

### 2. 条件日志
```dart
// 使用 kDebugMode
import 'package:flutter/foundation.dart';

void debugLog(String message) {
  if (kDebugMode) {
    print(message);
  }
}

// 使用 assert
assert(() {
  print('调试信息: $variable');
  return true;
}());

// 使用自定义日志类
class AppLogger {
  static void debug(String message) {
    if (kDebugMode) {
      print('DEBUG: $message');
    }
  }
  
  static void info(String message) {
    print('INFO: $message');
  }
  
  static void warning(String message) {
    print('WARNING: $message');
  }
  
  static void error(String message) {
    print('ERROR: $message');
  }
}
```

### 3. 日志格式化
```dart
// 格式化日志
class LogFormatter {
  static String format(String level, String message, {String? tag}) {
    final timestamp = DateTime.now().toIso8601String();
    final tagStr = tag != null ? '[$tag] ' : '';
    return '[$timestamp] [$level] $tagStr$message';
  }
  
  static void debug(String message, {String? tag}) {
    print(format('DEBUG', message, tag: tag));
  }
  
  static void info(String message, {String? tag}) {
    print(format('INFO', message, tag: tag));
  }
  
  static void warning(String message, {String? tag}) {
    print(format('WARNING', message, tag: tag));
  }
  
  static void error(String message, {String? tag}) {
    print(format('ERROR', message, tag: tag));
  }
}

// 使用示例
LogFormatter.debug('用户点击按钮', tag: 'HomePage');
LogFormatter.info('数据加载完成', tag: 'DataService');
LogFormatter.warning('网络连接不稳定', tag: 'Network');
LogFormatter.error('登录失败', tag: 'Auth');
```

## 🔧 性能调试

### 1. 帧率调试
```dart
// 监控帧率
class FrameRateMonitor {
  static int _frameCount = 0;
  static int _lastTimestamp = 0;
  
  static void startMonitoring() {
    WidgetsBinding.instance.addTimingsCallback((timings) {
      for (final timing in timings) {
        final duration = timing.totalSpan.inMilliseconds;
        if (duration > 16) { // 超过 60fps
          print('帧耗时: $duration ms');
        }
      }
    });
  }
  
  static void recordFrame() {
    _frameCount++;
    final now = DateTime.now().millisecondsSinceEpoch;
    if (now - _lastTimestamp >= 1000) {
      print('FPS: $_frameCount');
      _frameCount = 0;
      _lastTimestamp = now;
    }
  }
}

// 在 build 方法中调用
@override
Widget build(BuildContext context) {
  FrameRateMonitor.recordFrame();
  return Container();
}
```

### 3. 网络调试
```dart
// 网络请求监控
class NetworkMonitor {
  static final List<NetworkRequest> _requests = [];
  
  static void logRequest(NetworkRequest request) {
    _requests.add(request);
    print('网络请求: ${request.method} ${request.url}');
  }
  
  static void logResponse(NetworkResponse response) {
    print('网络响应: ${response.statusCode} ${response.url}');
  }
  
  static void printSummary() {
    print('网络请求统计:');
    print('  总请求数: ${_requests.length}');
    print('  成功请求: ${_requests.where((r) => r.success).length}');
    print('  失败请求: ${_requests.where((r) => !r.success).length}');
  }
}
```
