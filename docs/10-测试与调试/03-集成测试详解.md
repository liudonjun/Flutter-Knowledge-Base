# 集成测试详解

> 掌握 Flutter 集成测试的编写方法。

## 📖 集成测试基础

### 1. 基本设置

```dart
// 集成测试设置
class IntegrationTestSetup {
  void explain() {
    print('''
    集成测试设置：
    
    // 1. 添加依赖
    // pubspec.yaml
    dev_dependencies:
      integration_test:
        sdk: flutter
    
    // 2. 创建测试文件
    // integration_test/app_test.dart
    
    import 'package:integration_test/integration_test.dart';
    import 'package:flutter_test/flutter_test.dart';
    import 'package:my_app/main.dart' as app;
    
    void main() {
      IntegrationTestWidgetsFlutterBinding.ensureInitialized();
      
      testWidgets('完整流程测试', (WidgetTester tester) async {
        app.main();
        await tester.pumpAndSettle();
        
        // 测试流程
        expect(find.text('首页'), findsOneWidget);
        
        await tester.tap(find.byIcon(Icons.add));
        await tester.pumpAndSettle();
        
        expect(find.text('计数: 1'), findsOneWidget);
      });
    }
    ''');
  }
}
```

### 2. 页面对象模式

```dart
// 页面对象模式
class PageObjectPattern {
  void explain() {
    print('''
    页面对象模式：
    
    // 定义页面对象
    class LoginPage {
      final WidgetTester tester;
      
      LoginPage(this.tester);
      
      // 查找器
      Finder get usernameField => find.byKey(Key('username'));
      Finder get passwordField => find.byKey(Key('password'));
      Finder get loginButton => find.text('登录');
      
      // 操作
      Future<void> enterUsername(String username) async {
        await tester.enterText(usernameField, username);
      }
      
      Future<void> enterPassword(String password) async {
        await tester.enterText(passwordField, password);
      }
      
      Future<void> tapLogin() async {
        await tester.tap(loginButton);
        await tester.pumpAndSettle();
      }
      
      Future<void> login(String username, String password) async {
        await enterUsername(username);
        await enterPassword(password);
        await tapLogin();
      }
    }
    
    // 使用页面对象
    testWidgets('登录流程', (WidgetTester tester) async {
      app.main();
      await tester.pumpAndSettle();
      
      final loginPage = LoginPage(tester);
      await loginPage.login('user', 'password');
      
      expect(find.text('欢迎'), findsOneWidget);
    });
    ''');
  }
}
```

## 📖 性能测试

### 1. 帧率测试

```dart
// 帧率测试
class PerformanceTest {
  void explain() {
    print('''
    性能测试：
    
    testWidgets('滚动性能测试', (WidgetTester tester) async {
      await tester.pumpWidget(
        MaterialApp(
          home: Scaffold(
            body: ListView.builder(
              itemCount: 1000,
              itemBuilder: (context, index) => ListTile(
                title: Text('Item $index'),
              ),
            ),
          ),
        ),
      );
      
      // 测量滚动性能
      final watch = Stopwatch()..start();
      
      for (int i = 0; i < 10; i++) {
        await tester.fling(find.byType(ListView), Offset(0, -500), 1000);
        await tester.pumpAndSettle();
      }
      
      watch.stop();
      
      // 验证性能
      expect(watch.elapsedMilliseconds, lessThan(5000));
    });
    
    testWidgets('重建性能测试', (WidgetTester tester) async {
      int buildCount = 0;
      
      await tester.pumpWidget(
        MaterialApp(
          home: StatefulBuilder(
            builder: (context, setState) {
              buildCount++;
              return Text('$buildCount');
            },
          ),
        ),
      );
      
      expect(buildCount, equals(1));
      
      // 触发重建
      tester.state(find.byType(StatefulBuilder)).setState(() {});
      await tester.pump();
      
      expect(buildCount, equals(2));
    });
    ''');
  }
}
```
