```dart
            email TEXT NOT NULL,
            phone TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
          )
        ''');

          CREATE TABLE posts(
            id INTEGER PRIMARY KEY AUTOINCREMENT,
```

```dart
// 缓存管理器
class CacheManager {
  static final Map<String, CacheItem> _cache = {};
  static const Duration _defaultMaxAge = Duration(minutes: 5);
  
  // 获取缓存
  static dynamic get(String key) {
    final item = _cache[key];
    if (item != null && !item.isExpired) {
      return item.value;
    }
    _cache.remove(key);
    return null;
  }
  
  // 设置缓存
  static void set(String key, dynamic value, {Duration? maxAge}) {
    _cache[key] = CacheItem(
      value: value,
      expiry: DateTime.now().add(maxAge ?? _defaultMaxAge),
    );
  }
  
  // 删除缓存
  static void remove(String key) {
    _cache.remove(key);
  }
  
  // 清空缓存
  static void clear() {
    _cache.clear();
  }
  
  // 检查是否存在
  static bool has(String key) {
    final item = _cache[key];
    return item != null && !item.isExpired;
  }
}

class CacheItem {
  final dynamic value;
  final DateTime expiry;
  
  CacheItem({required this.value, required this.expiry});
  
  bool get isExpired => DateTime.now().isAfter(expiry);
}

// 网络缓存服务
class NetworkCacheService {
  // 带缓存的网络请求
  static Future<T> getWithCache<T>(
    String cacheKey,
    Future<T> Function() request, {
    Duration maxAge = const Duration(minutes: 5),
  }) async {
    // 检查缓存
    final cached = CacheManager.get(cacheKey);
    if (cached != null) {
      return cached as T;
    }
    
    // 发起网络请求
    final result = await request();
    
    // 保存到缓存
    CacheManager.set(cacheKey, result, maxAge: maxAge);
    
    return result;
  }
  
  // 清除缓存
  static void clearCache(String cacheKey) {
    CacheManager.remove(cacheKey);
  }
  
  // 清除所有缓存
  static void clearAllCache() {
    CacheManager.clear();
  }
}
```

### 2. 数据同步策略
```dart
// 数据同步管理器
class DataSyncManager {
  static final Map<String, DateTime> _lastSyncTime = {};
  static const Duration _syncInterval = Duration(hours: 1);
  
  // 检查是否需要同步
  static bool shouldSync(String key) {
    final lastSync = _lastSyncTime[key];
    if (lastSync == null) return true;
    return DateTime.now().difference(lastSync) > _syncInterval;
  }
  
  // 标记同步完成
  static void markSynced(String key) {
    _lastSyncTime[key] = DateTime.now();
  }
  
  // 强制同步
  static Future<void> forceSync(String key, Future<void> Function() syncAction) async {
    await syncAction();
    markSynced(key);
  }
  
  // 智能同步
  static Future<void> smartSync(String key, Future<void> Function() syncAction) async {
    if (shouldSync(key)) {
      await syncAction();
      markSynced(key);
    }
  }
}

// 使用示例
class UserSyncService {
  static const String _syncKey = 'users';
  
  // 同步用户数据
  static Future<void> syncUsers() async {
    await DataSyncManager.smartSync(_syncKey, () async {
      // 从服务器获取数据
      final response = await http.get(
        Uri.parse('https://api.example.com/users'),
      );
      
      if (response.statusCode == 200) {
        List<dynamic> data = jsonDecode(response.body);
        List<User> users = data.map((json) => User.fromJson(json)).toList();

          await UserRepository.insertUser(user);
        }
      }
    });
```

### 1. 存储选择建议
```dart
// 1. 简单配置: SharedPreferences
// 2. 复杂数据: SQLite 或 Hive
// 3. 安全数据: flutter_secure_storage
// 4. 大量数据: 数据库
// 5. 临时数据: 文件存储

// 存储封装
class StorageService {
  static Future<void> save(String key, dynamic value) async {
    if (value is String) {
      await SharedPreferencesService.saveString(key, value);
    } else if (value is int) {
      await SharedPreferencesService.saveInt(key, value);
    } else if (value is bool) {
      await SharedPreferencesService.saveBool(key, value);
    } else if (value is Map || value is List) {
      await SharedPreferencesService.saveString(key, jsonEncode(value));
    }
  }
  
  static Future<dynamic> read(String key) async {
    return await SharedPreferencesService.getString(key);
  }
}
```

```dart
// 1. 批量操作
class BatchOperations {
  static Future<void> batchInsert(List<User> users) async {
    final db = await DatabaseService.database;
    final batch = db.batch();
    
    for (var user in users) {
      batch.insert('users', user.toJson());
    }
    
    await batch.commit();
  }
  
  // 2. 索引优化
  static Future<void> createIndexes() async {
    final db = await DatabaseService.database;
```
