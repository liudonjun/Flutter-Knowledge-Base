class _SwipeExampleState extends State<SwipeExample> {
  String _direction = '等待滑动';
  
  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onPanEnd: (details) {
        final velocity = details.velocity.pixelsPerSecond;
        final dx = velocity.dx;
        final dy = velocity.dy;
        
        if (dx.abs() > dy.abs()) {
          if (dx > 0) {
            setState(() {
              _direction = '向右滑动';
            });
          } else {
            setState(() {
              _direction = '向左滑动';
            });
          }
        } else {
          if (dy > 0) {
            setState(() {
              _direction = '向下滑动';
            });
          } else {
            setState(() {
              _direction = '向上滑动';
            });
          }
        }
      },
      child: Container(
        width: 200,

        child: Center(

class MultiTouchPainter extends CustomPainter {
  final List<Offset> points;
  
  MultiTouchPainter(this.points);
  
  @override
  void paint(Canvas canvas, Size size) {
    final paint = Paint()
      ..color = Colors.blue
      ..strokeWidth = 5
      ..strokeCap = StrokeCap.round;
    
    for (int i = 0; i < points.length - 1; i++) {
      canvas.drawLine(points[i], points[i + 1], paint);
    }
  }
  
  @override
  bool shouldRepaint(covariant CustomPainter oldDelegate) => true;
}

class CustomGestureRecognizer extends OneSequenceGestureRecognizer {
  final Function(Offset) onCustomGesture;
  
  CustomGestureRecognizer({required this.onCustomGesture});
  
  @override
  void addPointer(PointerEvent event) {
    startTrackingPointer(event.pointer);
  }
  
  @override
  void handleEvent(PointerEvent event) {
    if (event is PointerMoveEvent) {
      // 自定义手势逻辑
      if (event.delta.dx.abs() > event.delta.dy.abs() * 2) {
        // 水平移动为主
        onCustomGesture(event.position);
        resolve(GestureDisposition.accepted);
      }
    }
  }
  
  @override
  String get debugDescription => 'custom';
  
  @override
  void didStopTrackingLastPointer(int pointer) {}
}

class GestureAnimationExample extends StatefulWidget {

  _GestureAnimationExampleState createState() => _GestureAnimationExampleState();

class _GestureAnimationExampleState extends State<GestureAnimationExample>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<double> _animation;
  double _dragExtent = 0;
  
  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      duration: Duration(milliseconds: 300),
      vsync: this,
    );
    
    _animation = Tween<double>(begin: 0, end: 1).animate(
      CurvedAnimation(parent: _controller, curve: Curves.easeOut),
    );
  }
  
  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }
  
  void _handleDragUpdate(DragUpdateDetails details) {
    setState(() {
      _dragExtent += details.delta.dx;
    });
  }
  
  void _handleDragEnd(DragEndDetails details) {
    _controller.forward(from: 0).then((_) {
      setState(() {
        _dragExtent = 0;
      });
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onHorizontalDragUpdate: _handleDragUpdate,
      onHorizontalDragEnd: _handleDragEnd,
      child: AnimatedBuilder(
        animation: _animation,
        builder: (context, child) {
          return Transform.translate(
            offset: Offset(_dragExtent * (1 - _animation.value), 0),
            child: child,
          );
        },
        child: Container(
          width: 200,
          height: 200,

            child: Text(

// 手势测试
void main() {
  group('Gesture', () {
    testWidgets('should detect tap', (tester) async {
      bool tapped = false;
      
      await tester.pumpWidget(
        GestureDetector(
          onTap: () {
            tapped = true;
          },
          child: Container(
            width: 100,
            height: 100,
            color: Colors.blue,
          ),
        ),
      );
      
      await tester.tap(find.byType(Container));
      expect(tapped, true);
    });
    
    testWidgets('should detect drag', (tester) async {
      Offset? dragDelta;
      
      await tester.pumpWidget(
        GestureDetector(
          onPanUpdate: (details) {
            dragDelta = details.delta;
          },
          child: Container(
            width: 100,
            height: 100,
            color: Colors.blue,
          ),
        ),
      );
      
      await tester.drag(find.byType(Container), Offset(50, 0));
      expect(dragDelta, Offset(50, 0));
    });
  });
}
