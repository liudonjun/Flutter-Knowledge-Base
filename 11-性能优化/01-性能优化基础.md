  static Future<void> preloadImage(BuildContext context, String imageUrl) async {
    if (_cache.containsKey(imageUrl)) return;

    _cache[imageUrl] = imageProvider;
    
    await precacheImage(imageProvider, context);
  }
  
  // 批量预加载

  static void clearCache() {

// 内存泄漏检测
class MemoryLeakDetector {
  static final Map<String, WeakReference<Object>> _references = {};
  static final Map<String, int> _counts = {};
  
  // 跟踪对象
  static void track(Object object, String name) {
    _references[name] = WeakReference(object);
    _counts[name] = (_counts[name] ?? 0) + 1;
  }
  
  // 检查内存泄漏
  static void checkLeaks() {
    for (final entry in _references.entries) {
      if (entry.value.target == null) {
        print('内存泄漏: ${entry.key} (${_counts[entry.key]} 次)');
      }
    }
  }
  
  // 打印内存使用情况
  static void printMemoryUsage() {
    print('内存使用情况:');
    for (final entry in _counts.entries) {
      final ref = _references[entry.key];
      final status = ref?.target != null ? '活跃' : '已释放';
      print('  ${entry.key}: ${entry.value} 次 ($status)');
    }
  }
}

// 使用示例
class MyWidget extends StatefulWidget {
  @override
  _MyWidgetState createState() => _MyWidgetState();
}

class _MyWidgetState extends State<MyWidget> {
  @override
  void initState() {
    super.initState();
    MemoryLeakDetector.track(this, 'MyWidget');
  }
  
  @override
  void dispose() {
    MemoryLeakDetector.checkLeaks();
    super.dispose();
  }
  
  @override
  Widget build(BuildContext context) {
    return Container();
  }
}

```dart
// 对象池模式
class ObjectPool<T> {
  final List<T> _pool = [];
  final T Function() _factory;
  final int _maxSize;
  
  ObjectPool(this._factory, {int maxSize = 100}) : _maxSize = maxSize;
  
  // 获取对象
  T acquire() {
    if (_pool.isNotEmpty) {
      return _pool.removeLast();
    }
    return _factory();
  }
  
  // 释放对象
  void release(T object) {
    if (_pool.length < _maxSize) {
      _pool.add(object);
    }
  }
  
  // 清空池
  void clear() {
    _pool.clear();
  }
  
  // 获取池大小
  int get size => _pool.length;
}

// 使用示例
class MyObject {
  final String id;
  MyObject(this.id);
}

final objectPool = ObjectPool<MyObject>(() => MyObject(DateTime.now().toString()));

void useObjectPool() {
  // 获取对象
  final obj = objectPool.acquire();
  
  // 使用对象
  print('使用对象: ${obj.id}');
  
  // 释放对象
  objectPool.release(obj);
}

// 问题：网络请求慢
// 解决：优化请求策略

// 1. 请求合并
class RequestBatcher {
  static final Map<String, List<Completer>> _pendingRequests = {};
  
  static Future<T> batchRequest<T>(
    String key,
    Future<T> Function() request,
  ) async {
    if (_pendingRequests.containsKey(key)) {
      final completer = Completer<T>();
      _pendingRequests[key]!.add(completer);
      return completer.future;
    }
    
    _pendingRequests[key] = [];
    
    try {
      final result = await request();
      for (final completer in _pendingRequests[key]!) {
        completer.complete(result);
      }
      return result;
    } catch (error) {
      for (final completer in _pendingRequests[key]!) {
        completer.completeError(error);
      }
      rethrow;
    } finally {
      _pendingRequests.remove(key);
    }
  }
}

// 2. 请求缓存
class RequestCache {
  static final Map<String, CacheItem> _cache = {};
  static const Duration _defaultMaxAge = Duration(minutes: 5);
  
  static Future<T> getWithCache<T>(
    String key,
    Future<T> Function() request, {
    Duration maxAge = _defaultMaxAge,
  }) async {
    final item = _cache[key];
    if (item != null && !item.isExpired) {
      return item.value as T;
    }
    
    final result = await request();
    _cache[key] = CacheItem(result, DateTime.now().add(maxAge));
    return result;
  }
  
  static void clearCache() {
    _cache.clear();
  }
}

class CacheItem {
  final dynamic value;
  final DateTime expiry;
  
  CacheItem(this.value, this.expiry);
  
  bool get isExpired => DateTime.now().isAfter(expiry);
}

// 3. 请求重试
class RetryHelper {
  static Future<T> retry<T>(
    Future<T> Function() request, {
    int maxRetries = 3,
    Duration delay = const Duration(seconds: 1),
  }) async {
    int attempts = 0;
    
    while (true) {
      try {
        return await request();
      } catch (e) {
        attempts++;
        if (attempts >= maxRetries) {
          rethrow;
        }
        await Future.delayed(delay * attempts);
      }
    }
  }
}

// 性能监控
class PerformanceMonitor {
  static final Map<String, Stopwatch> _timers = {};
  static final List<PerformanceMetric> _metrics = [];
  
  // 开始计时
  static void startTimer(String name) {
    _timers[name] = Stopwatch()..start();
  }
  
  // 结束计时
  static void endTimer(String name) {
    final timer = _timers[name];
    if (timer != null) {
      timer.stop();
      _metrics.add(PerformanceMetric(name, timer.elapsedMilliseconds));
      _timers.remove(name);
    }
  }
  
  // 测量函数执行时间
  static Future<T> measureAsync<T>(
    String name,
    Future<T> Function() operation,
  ) async {
    startTimer(name);
    try {
      return await operation();
    } finally {
      endTimer(name);
    }
  }
  
  // 打印性能报告
  static void printReport() {
    print('性能报告:');
    for (final metric in _metrics) {
      print('  ${metric.name}: ${metric.duration}ms');
    }
  }
  
  // 清除指标
  static void clearMetrics() {
    _metrics.clear();
  }
}

class PerformanceMetric {
  final String name;
  final int duration;
  
  PerformanceMetric(this.name, this.duration);
}

void performanceTest() async {

// 帧率监控
class FrameRateMonitor {
  static int _frameCount = 0;
  static int _lastTimestamp = 0;
  static final List<int> _frameRates = [];
  
  // 开始监控
  static void startMonitoring() {
    WidgetsBinding.instance.addTimingsCallback((timings) {
      for (final timing in timings) {
        final duration = timing.totalSpan.inMilliseconds;
        if (duration > 16) { // 超过 60fps
          print('帧耗时: $duration ms');
        }
      }
    });
  }
  
  // 记录帧
  static void recordFrame() {
    _frameCount++;
    final now = DateTime.now().millisecondsSinceEpoch;
    if (now - _lastTimestamp >= 1000) {
      _frameRates.add(_frameCount);
      print('FPS: $_frameCount');
      _frameCount = 0;
      _lastTimestamp = now;
    }
  }
  
  // 获取平均帧率
  static double get averageFrameRate {
    if (_frameRates.isEmpty) return 0;
    return _frameRates.reduce((a, b) => a + b) / _frameRates.length;
  }
  
  // 打印帧率报告
  static void printReport() {
    print('帧率报告:');
    print('  平均帧率: $averageFrameRate FPS');
    print('  最低帧率: ${_frameRates.isNotEmpty ? _frameRates.reduce((a, b) => a < b ? a : b) : 0} FPS');
    print('  最高帧率: ${_frameRates.isNotEmpty ? _frameRates.reduce((a, b) => a > b ? a : b) : 0} FPS');
  }
}
```
